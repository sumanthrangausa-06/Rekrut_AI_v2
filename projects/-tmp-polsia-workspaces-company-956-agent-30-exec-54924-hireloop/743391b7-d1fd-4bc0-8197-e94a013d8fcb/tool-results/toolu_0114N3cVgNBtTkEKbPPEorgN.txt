   229→router.post('/answer', authMiddleware, async (req, res) => {
   230→  const client = await pool.connect();
   231→  try {
   232→    const userId = req.user.id;
   233→    const { sessionId, questionId, answer, timeTaken } = req.body;
   234→
   235→    await client.query('BEGIN');
   236→
   237→    // Get session
   238→    const sessionResult = await client.query(
   239→      'SELECT * FROM assessment_sessions WHERE id = $1 AND user_id = $2',
   240→      [sessionId, userId]
   241→    );
   242→
   243→    if (sessionResult.rows.length === 0) {
   244→      await client.query('ROLLBACK');
   245→      return res.status(404).json({ error: 'Session not found' });
   246→    }
   247→
   248→    const session = sessionResult.rows[0];
   249→
   250→    if (session.status !== 'in_progress') {
   251→      await client.query('ROLLBACK');
   252→      return res.status(400).json({ error: 'Session is not active' });
   253→    }
   254→
   255→    // Get question details
   256→    const questionResult = await client.query(
   257→      'SELECT * FROM assessment_questions WHERE id = $1',
   258→      [questionId]
   259→    );
   260→
   261→    if (questionResult.rows.length === 0) {
   262→      await client.query('ROLLBACK');
   263→      return res.status(404).json({ error: 'Question not found' });
   264→    }
   265→
   266→    const question = questionResult.rows[0];
   267→
   268→    // Evaluate answer
   269→    let isCorrect = false;
   270→    let scorePoints = 0;
   271→    let aiFeedback = null;
   272→
   273→    if (question.question_type === 'multiple_choice') {
   274→      isCorrect = answer === question.correct_answer;
   275→      scorePoints = isCorrect ? (question.difficulty_level || 2) * 10 : 0;
   276→    } else if (question.question_type === 'short_answer') {
   277→      // Use AI to evaluate short answer
   278→      const evaluation = await evaluateShortAnswer(question.question_text, answer, question.explanation);
   279→      isCorrect = evaluation.score >= 70;
   280→      scorePoints = Math.round((evaluation.score / 100) * (question.difficulty_level || 2) * 15);
   281→      aiFeedback = evaluation.feedback;
   282→    }
   283→
   284→    // Check for time anomaly (too fast)
   285→    const expectedMinTime = question.question_type === 'multiple_choice' ? 5 : 15;
   286→    const isTimeAnomaly = timeTaken < expectedMinTime;
   287→
   288→    // Update session
   289→    const questionsAsked = typeof session.questions_asked === 'string'
   290→      ? JSON.parse(session.questions_asked) : (session.questions_asked || []);
   291→    const answersGiven = typeof session.answers_given === 'string'
   292→      ? JSON.parse(session.answers_given) : (session.answers_given || []);
   293→
   294→    answersGiven.push({
   295→      questionId,
   296→      answer,
   297→      isCorrect,
   298→      scorePoints,
   299→      timeTaken,
   300→      timestamp: new Date(),
   301→      aiFeedback
   302→    });
   303→
   304→    const newScore = (session.score || 0) + scorePoints;
   305→    const currentQuestionIndex = (session.current_question_index || 0) + 1;
   306→
   307→    // Adaptive difficulty: increase if correct and no time anomaly, decrease if wrong
   308→    let newDifficulty = session.current_difficulty || 2;
   309→    if (isCorrect && !isTimeAnomaly && newDifficulty < 5) {
   310→      newDifficulty += 1;
   311→    } else if (!isCorrect && newDifficulty > 1) {
   312→      newDifficulty -= 1;
   313→    }
   314→
   315→    const maxDifficultyReached = Math.max(session.max_difficulty_reached || 2, newDifficulty);
   316→    const timeAnomalies = (session.time_anomalies || 0) + (isTimeAnomaly ? 1 : 0);
   317→
   318→    await client.query(`
   319→      UPDATE assessment_sessions
   320→      SET answers_given = $1, score = $2, current_question_index = $3,
   321→          current_difficulty = $4, max_difficulty_reached = $5,
   322→          time_anomalies = $6
   323→      WHERE id = $7
   324→    `, [
   325→      JSON.stringify(answersGiven), newScore, currentQuestionIndex,
   326→      newDifficulty, maxDifficultyReached, timeAnomalies, sessionId
   327→    ]);
   328→
   329→    // Check if assessment is complete (10 questions)
   330→    if (currentQuestionIndex >= 10) {
   331→      await completeAssessment(client, sessionId, userId);
   332→      await client.query('COMMIT');
   333→
   334→      return res.json({
   335→        completed: true,
   336→        score: newScore,
   337→        feedback: isCorrect ? 'Correct!' : 'Incorrect',
   338→        explanation: question.explanation,
   339→        aiFeedback
   340→      });
   341→    }
   342→
   343→    // Get skill for next question
   344→    const skillResult = await client.query(
   345→      'SELECT cs.* FROM candidate_skills cs JOIN assessment_sessions ass ON cs.id = ass.skill_id WHERE ass.id = $1',
   346→      [sessionId]
   347→    );
   348→
   349→    if (skillResult.rows.length === 0) {
   350→      await client.query('ROLLBACK');
   351→      return res.status(500).json({ error: 'Skill not found for session' });
   352→    }
   353→
   354→    const skill = skillResult.rows[0];
   355→
   356→    // Generate next question
   357→    const nextQuestion = await generateQuestion(skill.skill_name, skill.category, newDifficulty, client);
   358→
   359→    if (!nextQuestion) {
   360→      await client.query('ROLLBACK');
   361→      return res.status(500).json({ error: 'Failed to generate next question' });
   362→    }
   363→
   364→    // Record question asked
   365→    questionsAsked.push({
   366→      questionId: nextQuestion.id,
   367→      difficulty: nextQuestion.difficulty_level,
   368→      timestamp: new Date()
   369→    });
   370→
   371→    await client.query(
   372→      'UPDATE assessment_sessions SET questions_asked = $1 WHERE id = $2',
   373→      [JSON.stringify(questionsAsked), sessionId]
   374→    );
   375→
   376→    await client.query('COMMIT');
   377→
   378→    res.json({
   379→      completed: false,
   380→      feedback: isCorrect ? 'Correct!' : 'Incorrect',
   381→      explanation: question.explanation,
   382→      aiFeedback,
   383→      nextQuestion: {
   384→        id: nextQuestion.id,
   385→        text: nextQuestion.question_text,
   386→        type: nextQuestion.question_type,
   387→        options: typeof nextQuestion.options === 'string' ? JSON.parse(nextQuestion.options) : nextQuestion.options,
   388→        timeLimit: nextQuestion.time_limit_seconds || 120,
   389→        questionNumber: currentQuestionIndex + 1,
   390→        totalQuestions: 10
   391→      }

   393→  } catch (error) {
   394→    await client.query('ROLLBACK');

   444→// Helper: Generate or fetch question
   445→async function generateQuestion(skillName, category, difficulty, client) {
   446→  // Try to find existing question from bank
   447→  const existingResult = await client.query(`
   448→    SELECT * FROM assessment_questions
   449→    WHERE LOWER(skill_category) = LOWER($1) AND difficulty_level = $2
   450→    ORDER BY RANDOM()
   451→    LIMIT 1
   452→  `, [skillName, difficulty]);
   453→
   454→  if (existingResult.rows.length > 0) {
   455→    return existingResult.rows[0];
   456→  }
   457→
   458→  // Generate new question using AI
   459→  const questionType = Math.random() > 0.3 ? 'multiple_choice' : 'short_answer';
   460→  const prompt = `Generate a difficulty ${difficulty}/5 technical assessment question about "${skillName}" (category: ${category}).
   461→
   462→Question type: ${questionType}
   463→
   464→Return ONLY valid JSON (no markdown, no code blocks):
   465→{
   466→  "question_text": "The question text here",
   467→  "question_type": "${questionType}",
   468→  ${questionType === 'multiple_choice' ? '"options": ["Option A text", "Option B text", "Option C text", "Option D text"],\n  "correct_answer": "Option A text",' : ''}
   469→  "explanation": "Brief explanation of the correct answer",
   470→  "time_limit_seconds": ${questionType === 'multiple_choice' ? '90' : '180'}
   471→}
   472→
   473→Requirements:
   474→- Make it practical and relevant to real-world ${skillName} work
   475→- Difficulty ${difficulty}/5 (1=beginner, 5=expert)
   476→- ${questionType === 'multiple_choice' ? 'Provide exactly 4 options. correct_answer must exactly match one of the options.' : 'Ask a question that requires a detailed written response.'}
   477→- Keep question clear and unambiguous`;
   478→
   479→  try {
   480→    const response = await chat(prompt, { maxTokens: 1024 });
   481→
   482→    // Parse response - handle potential markdown code blocks
   483→    let cleanResponse = response.trim();
   484→    if (cleanResponse.startsWith('```')) {
   485→      cleanResponse = cleanResponse.replace(/^```(?:json)?\s*/, '').replace(/\s*```$/, '');
   486→    }
   487→
   488→    const questionData = JSON.parse(cleanResponse);
   489→
   490→    // Validate the response
   491→    if (!questionData.question_text) {
   492→      throw new Error('Missing question_text in AI response');
   493→    }
   494→
   495→    const qType = questionData.question_type || questionType;
   496→    const options = qType === 'multiple_choice' && questionData.options
   497→      ? (typeof questionData.options === 'string' ? questionData.options : JSON.stringify(questionData.options))
   498→      : null;
   499→
   500→    // Save to database
   501→    const result = await client.query(`
   502→      INSERT INTO assessment_questions
   503→      (skill_category, difficulty_level, question_type, question_text, options, correct_answer, explanation, time_limit_seconds)
   504→      VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
   505→      RETURNING *
   506→    `, [
   507→      skillName,
   508→      difficulty,
   509→      qType,
   510→      questionData.question_text,
   511→      options,
   512→      questionData.correct_answer || null,
   513→      questionData.explanation || 'See the correct answer above.',
   514→      questionData.time_limit_seconds || (qType === 'multiple_choice' ? 90 : 180)
   515→    ]);
   516→
   517→    return result.rows[0];
   518→  } catch (error) {
   519→    console.error('Error generating AI question:', error);
   520→
   521→    // Fallback: try any question from the bank
   522→    const fallbackResult = await client.query(`
   523→      SELECT * FROM assessment_questions
   524→      ORDER BY RANDOM()
   525→      LIMIT 1
   526→    `);
   527→
   528→    if (fallbackResult.rows.length > 0) {
   529→      return fallbackResult.rows[0];
   530→    }
   531→
   532→    // Last resort: create a hardcoded question

   586→// Helper: Complete assessment
   587→async function completeAssessment(client, sessionId, userId) {
   588→  const sessionResult = await client.query(
   589→    'SELECT * FROM assessment_sessions WHERE id = $1',
   590→    [sessionId]
   591→  );
   592→  const session = sessionResult.rows[0];
   593→
   594→  // Get skill name for the title
   595→  const skillResult = await client.query(
   596→    'SELECT skill_name FROM candidate_skills WHERE id = $1',
   597→    [session.skill_id]
   598→  );
   599→  const skillName = skillResult.rows.length > 0 ? skillResult.rows[0].skill_name : 'Unknown';
   600→
   601→  // Calculate anti-cheat score (100 = clean, lower = suspicious)
   602→  let antiCheatScore = 100;
   603→  antiCheatScore -= (session.tab_switches || 0) * 5;
   604→  antiCheatScore -= (session.copy_paste_attempts || 0) * 10;
   605→  antiCheatScore -= (session.time_anomalies || 0) * 5;
   606→  antiCheatScore = Math.max(0, antiCheatScore);
   607→
   608→  const passed = (session.score || 0) >= 60 && antiCheatScore >= 50;
   609→
   610→  // Mark session complete
   611→  await client.query(
   612→    "UPDATE assessment_sessions SET status = 'completed', completed_at = NOW() WHERE id = $1",
   613→    [sessionId]
   614→  );
   615→
   616→  // Create skill assessment record
   617→  const assessmentResult = await client.query(`
   618→    INSERT INTO skill_assessments
   619→    (user_id, skill_id, session_id, assessment_type, title, score, max_score, passed,
   620→     anti_cheat_score, behavioral_flags, duration_seconds, started_at, completed_at)
   621→    VALUES ($1, $2, $3, 'dynamic', $4, $5, 100, $6, $7, $8,
   622→            EXTRACT(EPOCH FROM (NOW() - $9::timestamp)), $9, NOW())
   623→    RETURNING id
   624→  `, [
   625→    userId,
   626→    session.skill_id,
   627→    sessionId,
   628→    `${skillName} Assessment`,
   629→    session.score || 0,
   630→    passed,
   631→    antiCheatScore,
   632→    JSON.stringify({
   633→      tab_switches: session.tab_switches || 0,
   634→      copy_paste: session.copy_paste_attempts || 0,
   635→      time_anomalies: session.time_anomalies || 0,
   636→      max_difficulty: session.max_difficulty_reached || 2
   637→    }),
   638→    session.started_at
   639→  ]);
   640→
   641→  // If passed, mark skill as verified
   642→  if (passed) {
   643→    await client.query(
   644→      'UPDATE candidate_skills SET is_verified = true, verified_at = NOW(), verified_score = $1 WHERE id = $2',
   645→      [session.score, session.skill_id]
   646→    );
   647→  }
   648→
   649→  return assessmentResult.rows[0].id;
   650→}
