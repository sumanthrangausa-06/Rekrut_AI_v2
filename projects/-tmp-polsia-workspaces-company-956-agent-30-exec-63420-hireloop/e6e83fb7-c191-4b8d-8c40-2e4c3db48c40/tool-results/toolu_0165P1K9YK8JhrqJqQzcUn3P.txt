    21→// Helper function to extract text from various file formats
    22→async function extractTextFromFile(buffer, mimetype) {
    23→  try {
    24→    // Handle PDF files
    25→    if (mimetype === 'application/pdf' || buffer.toString('utf-8', 0, 4) === '%PDF') {
    26→      const pdfData = await pdfParse(buffer);
    27→      return pdfData.text;
    28→    }
    29→
    30→    // Handle DOCX files (application/vnd.openxmlformats-officedocument.wordprocessingml.document)
    31→    if (mimetype === 'application/vnd.openxmlformats-officedocument.wordprocessingml.document' ||
    32→        buffer.toString('utf-8', 0, 2) === 'PK') {
    33→      const result = await mammoth.extractRawText({ buffer });
    34→      return result.value;
    35→    }
    36→
    37→    // Handle plain text files (fallback)
    38→    const text = buffer.toString('utf-8');
    39→    if (text.length > 0 && /[a-zA-Z]{3,}/.test(text.substring(0, 500))) {
    40→      return text;
    41→    }
    42→
    43→    throw new Error('Unsupported file format or unable to extract text');
    44→  } catch (error) {
    45→    console.error('Text extraction error:', error.message);
    46→    throw error;
    47→  }
    48→}
