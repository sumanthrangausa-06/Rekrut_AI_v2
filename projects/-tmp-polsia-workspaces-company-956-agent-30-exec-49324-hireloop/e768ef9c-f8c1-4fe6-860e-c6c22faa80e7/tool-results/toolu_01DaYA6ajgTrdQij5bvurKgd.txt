   212→    async function loadDashboard() {
   213→      try {
   214→        const response = await fetch('/api/payroll/dashboard', {
   215→          headers: { 'Authorization': `Bearer ${getToken()}` }
   216→        });
   217→
   218→        if (!response.ok) throw new Error('Failed to load dashboard');
   219→
   220→        dashboardData = await response.json();
   221→        renderDashboard();
   222→      } catch (err) {
   223→        console.error('Load dashboard error:', err);
   224→        showError('Failed to load payroll dashboard');
   225→      }
   226→    }
   227→
   228→    async function loadEmployees() {
   229→      try {
   230→        const response = await fetch('/api/payroll/employees', {
   231→          headers: { 'Authorization': `Bearer ${getToken()}` }
   232→        });
   233→
   234→        if (!response.ok) throw new Error('Failed to load employees');
   235→
   236→        const data = await response.json();
   237→        renderEmployees(data.employees);
   238→      } catch (err) {
   239→        console.error('Load employees error:', err);
   240→      }
   241→    }

   243→    function renderDashboard() {
   244→      document.getElementById('active-employees').textContent = dashboardData.activeEmployees;
   245→      document.getElementById('upcoming-count').textContent = dashboardData.upcomingPayrolls.length;
   246→      document.getElementById('monthly-total').textContent = `$${dashboardData.monthlyTotal.toLocaleString()}`;
   247→
   248→      renderUpcomingPayrolls(dashboardData.upcomingPayrolls);
   249→      renderRecentPayrolls(dashboardData.recentPayrolls);
   250→    }

   252→    function renderEmployees(employees) {
   253→      const tbody = document.getElementById('employees-tbody');
   254→      document.getElementById('employee-count').textContent = `${employees.length} employees`;
   255→
   256→      if (employees.length === 0) {
   257→        tbody.innerHTML = `
   258→          <tr>
   259→            <td colspan="7" style="text-align: center; padding: 40px; color: var(--text-muted);">
   260→              No employees found. Hire candidates to add them to payroll.
   261→            </td>
   262→          </tr>
   263→        `;
   264→        return;
   265→      }
   266→
   267→      tbody.innerHTML = employees.map(emp => `
   268→        <tr>
   269→          <td>
   270→            <div style="display: flex; align-items: center; gap: 10px;">
   271→              <div class="user-avatar">${emp.employee_name?.charAt(0) || '?'}</div>
   272→              <div>
   273→                <div style="font-weight: 500;">${emp.employee_name || 'Unknown'}</div>
   274→                <div style="font-size: 0.85rem; color: var(--text-muted);">${emp.employee_email || ''}</div>
   275→              </div>
   276→            </div>
   277→          </td>
   278→          <td>${emp.position || '-'}</td>
   279→          <td><span class="badge">${emp.salary_type || '-'}</span></td>
   280→          <td>$${parseFloat(emp.salary_amount || 0).toLocaleString()}</td>
   281→          <td>${emp.pay_frequency || '-'}</td>
   282→          <td><span class="badge badge-success">${emp.status || 'active'}</span></td>
   283→          <td>
   284→            <button onclick="viewEmployee(${emp.id})" class="btn btn-sm">View</button>
   285→          </td>
   286→        </tr>
   287→      `).join('');
   288→    }

   354→    async function handleCreatePayroll(e) {
   355→      e.preventDefault();
   356→      const form = e.target;
   357→      const formData = new FormData(form);
   358→
   359→      try {
   360→        const response = await fetch('/api/payroll/runs', {
   361→          method: 'POST',
   362→          headers: {
   363→            'Authorization': `Bearer ${getToken()}`,
   364→            'Content-Type': 'application/json'
   365→          },
   366→          body: JSON.stringify(Object.fromEntries(formData))
   367→        });
   368→
   369→        if (!response.ok) throw new Error('Failed to create payroll run');
   370→
   371→        const data = await response.json();
   372→        showSuccess('Payroll run created successfully');
   373→        closeModal();
   374→        form.reset();
   375→        loadDashboard();
   376→      } catch (err) {
   377→        console.error('Create payroll error:', err);
   378→        showError('Failed to create payroll run');
   379→      }
   380→    }
