     8→// Handle OAuth callback tokens from URL
     9→function handleOAuthCallback() {
    10→  const params = new URLSearchParams(window.location.search);
    11→  const token = params.get('token');
    12→  const refresh = params.get('refresh');
    13→
    14→  if (token) {
    15→    localStorage.setItem(TOKEN_KEY, token);
    16→    if (refresh) {
    17→      localStorage.setItem(REFRESH_KEY, refresh);
    18→    }
    19→    // Clean URL
    20→    window.history.replaceState({}, document.title, window.location.pathname);
    21→  }
    22→}
    23→
    24→// Refresh access token using refresh token
    25→async function refreshAccessToken() {
    26→  const refreshToken = localStorage.getItem(REFRESH_KEY);
    27→  if (!refreshToken) return null;
    28→
    29→  try {
    30→    const response = await fetch('/api/auth/refresh', {
    31→      method: 'POST',
    32→      headers: { 'Content-Type': 'application/json' },
    33→      body: JSON.stringify({ refreshToken })
    34→    });
    35→
    36→    if (response.ok) {
    37→      const data = await response.json();
    38→      if (data.success) {
    39→        localStorage.setItem(TOKEN_KEY, data.accessToken);
    40→        localStorage.setItem(REFRESH_KEY, data.refreshToken);
    41→        return data.accessToken;
    42→      }
    43→    }
    44→  } catch (err) {
    45→    console.error('Token refresh failed:', err);
    46→  }
    47→
    48→  // Refresh failed - clear tokens
    49→  localStorage.removeItem(TOKEN_KEY);
    50→  localStorage.removeItem(REFRESH_KEY);
    51→  return null;
    52→}

    54→// Check auth status on page load
    55→async function checkAuth() {
    56→  // Handle OAuth callback first
    57→  handleOAuthCallback();
    58→
    59→  const token = localStorage.getItem(TOKEN_KEY);
    60→  if (!token) return null;
    61→
    62→  try {
    63→    const response = await fetch('/api/auth/me', {
    64→      headers: { 'Authorization': `Bearer ${token}` }
    65→    });
    66→
    67→    if (response.ok) {
    68→      const data = await response.json();
    69→      currentUser = data.user;
    70→      updateNavForAuth();
    71→      return data.user;
    72→    }
    73→
    74→    // Token expired - try refresh
    75→    if (response.status === 401) {
    76→      const newToken = await refreshAccessToken();
    77→      if (newToken) {
    78→        return await checkAuth(); // Retry with new token
    79→      }
    80→    }
    81→  } catch (err) {
    82→    console.error('Auth check failed:', err);
    83→  }
    84→  return null;
    85→}

   100→    `;
   101→  }
   102→}

   104→// Registration
   105→async function handleRegister(event) {
   106→  event.preventDefault();
   107→  const form = event.target;
   108→  const submitBtn = form.querySelector('button[type="submit"]');
   109→  const errorEl = document.getElementById('register-error');
   110→
   111→  submitBtn.disabled = true;
   112→  submitBtn.textContent = 'Creating account...';
   113→  if (errorEl) errorEl.style.display = 'none';
   114→
   115→  try {
   116→    const response = await fetch('/api/auth/register', {
   117→      method: 'POST',
   118→      headers: { 'Content-Type': 'application/json' },
   119→      body: JSON.stringify({
   120→        email: form.email.value,
   121→        password: form.password.value,
   122→        name: form.name?.value || '',
   123→        role: form.role?.value || 'candidate',
   124→        company_name: form.company_name?.value || ''
   125→      })
   126→    });
   127→
   128→    const data = await response.json();
   129→
   130→    if (data.success) {
   131→      localStorage.setItem(TOKEN_KEY, data.accessToken || data.token);
   132→      if (data.refreshToken) {
   133→        localStorage.setItem(REFRESH_KEY, data.refreshToken);
   134→      }
   135→      // Redirect based on role
   136→      const role = form.role?.value || 'candidate';
   137→      window.location.href = role === 'recruiter'
   138→        ? '/recruiter-dashboard.html'
   139→        : '/candidate-dashboard.html';
   140→    } else {
   141→      if (errorEl) {
   142→        errorEl.textContent = data.error || 'Registration failed';
   143→        errorEl.style.display = 'block';
   144→      }
   145→    }
   146→  } catch (err) {
   147→    if (errorEl) {
   148→      errorEl.textContent = 'Network error. Please try again.';
   149→      errorEl.style.display = 'block';
   150→    }
   151→  } finally {
   152→    submitBtn.disabled = false;
   153→    submitBtn.textContent = 'Create Account';
   154→  }
   155→}

   206→// Logout
   207→async function handleLogout() {
   208→  const token = localStorage.getItem(TOKEN_KEY);
   209→  if (token) {
   210→    try {
   211→      await fetch('/api/auth/logout', {
   212→        method: 'POST',
   213→        headers: { 'Authorization': `Bearer ${token}` }
   214→      });
   215→    } catch (e) {
   216→      // Ignore errors during logout
   217→    }
   218→  }
   219→  localStorage.removeItem(TOKEN_KEY);
   220→  localStorage.removeItem(REFRESH_KEY);
   221→  window.location.href = '/';
   222→}

   224→// API helper with automatic token refresh
   225→async function apiCall(endpoint, options = {}) {
   226→  let token = localStorage.getItem(TOKEN_KEY);
   227→  const headers = {
   228→    'Content-Type': 'application/json',
   229→    ...(token && { 'Authorization': `Bearer ${token}` }),
   230→    ...options.headers
   231→  };
   232→
   233→  let response = await fetch(`/api${endpoint}`, {
   234→    ...options,
   235→    headers
   236→  });
   237→
   238→  // If 401 and we have a refresh token, try to refresh
   239→  if (response.status === 401) {
   240→    const newToken = await refreshAccessToken();
   241→    if (newToken) {
   242→      // Retry with new token
   243→      headers['Authorization'] = `Bearer ${newToken}`;
   244→      response = await fetch(`/api${endpoint}`, {
   245→        ...options,
   246→        headers
   247→      });
   248→    } else {
   249→      // Still unauthorized - redirect to login
   250→      window.location.href = '/login.html';
   251→      return null;
   252→    }
   253→  }
   254→
   255→  return response.json();
   256→}

   336→// Mobile Menu Toggle
   337→function initMobileMenu() {
   338→  const navLinks = document.getElementById('nav-links');
   339→  const navContainer = document.querySelector('.nav-container');
   340→
   341→  // Create hamburger button if it doesn't exist
   342→  let mobileToggle = document.querySelector('.mobile-menu-toggle');
   343→  if (!mobileToggle && navLinks && navContainer) {
   344→    mobileToggle = document.createElement('button');
   345→    mobileToggle.className = 'mobile-menu-toggle';
   346→    mobileToggle.innerHTML = '☰';
   347→    mobileToggle.setAttribute('aria-label', 'Toggle menu');
   348→
   349→    // Insert before nav-auth
   350→    const navAuth = document.getElementById('nav-auth');
   351→    if (navAuth) {
   352→      navContainer.insertBefore(mobileToggle, navAuth);
   353→    }
   354→  }
   355→
   356→  if (mobileToggle && navLinks) {
   357→    // Toggle mobile menu
   358→    mobileToggle.addEventListener('click', (e) => {
   359→      e.stopPropagation();
   360→      navLinks.classList.toggle('mobile-open');
   361→      mobileToggle.innerHTML = navLinks.classList.contains('mobile-open') ? '✕' : '☰';
   362→    });
   363→
   364→    // Close menu when clicking outside
   365→    document.addEventListener('click', (e) => {
   366→      if (!navLinks.contains(e.target) && !mobileToggle.contains(e.target)) {
   367→        navLinks.classList.remove('mobile-open');
   368→        mobileToggle.innerHTML = '☰';
   369→      }
   370→    });
   371→
   372→    // Handle dropdown toggles on mobile
   373→    const dropdowns = navLinks.querySelectorAll('.nav-dropdown');
   374→    dropdowns.forEach(dropdown => {
   375→      const toggle = dropdown.querySelector('.nav-dropdown-toggle');
   376→      if (toggle) {
   377→        toggle.addEventListener('click', (e) => {
   378→          if (window.innerWidth <= 768) {
   379→            e.preventDefault();
   380→            dropdown.classList.toggle('active');
   381→          }
   382→        });
   383→      }
   384→    });
   385→  }
   386→}

   415→    mobileToggle.innerHTML = sidebar.classList.contains('mobile-open') ? '✕' : '☰';
   416→  });
   417→
   418→  // Close sidebar when clicking overlay
   419→  overlay.addEventListener('click', () => {
   420→    sidebar.classList.remove('mobile-open');
   421→    overlay.classList.remove('active');
   422→    mobileToggle.innerHTML = '☰';

   425→  // Close sidebar when clicking nav items on mobile
   426→  const navItems = sidebar.querySelectorAll('.nav-item');
   427→  navItems.forEach(item => {
   428→    item.addEventListener('click', () => {
   429→      if (window.innerWidth <= 768) {
   430→        sidebar.classList.remove('mobile-open');
   431→        overlay.classList.remove('active');
   432→        mobileToggle.innerHTML = '☰';
   433→      }
   434→    });
   435→  });
