     1→import { useEffect, useState } from 'react'
     2→import { Link, useNavigate } from 'react-router-dom'
     3→import { apiCall } from '@/lib/api'
     4→import { Card, CardContent } from '@/components/ui/card'
     5→import { Button } from '@/components/ui/button'
     6→import { Badge } from '@/components/ui/badge'
     7→import { Input } from '@/components/ui/input'
     8→import { Select } from '@/components/ui/select'
     9→import {
    10→  Plus, Briefcase, MapPin, Users, Edit, Trash2, Search, Eye, EyeOff,
    11→  BarChart3, Clock,
    12→} from 'lucide-react'
    13→
    14→interface Job {
    15→  id: number
    16→  title: string
    17→  company: string
    18→  location: string
    19→  salary_range: string
    20→  job_type: string
    21→  status: string
    22→  created_at: string
    23→  views?: number
    24→  application_count?: number
    25→  interviews?: number
    26→  screening_questions?: string
    27→}
    28→
    29→const statusColors: Record<string, 'success' | 'warning' | 'secondary' | 'destructive'> = {
    30→  active: 'success',
    31→  paused: 'warning',
    32→  closed: 'secondary',
    33→  draft: 'secondary',
    34→}
    35→
    36→export function RecruiterJobsPage() {
    37→  const navigate = useNavigate()
    38→  const [jobs, setJobs] = useState<Job[]>([])
    39→  const [loading, setLoading] = useState(true)
    40→  const [deleting, setDeleting] = useState<number | null>(null)
    41→  const [search, setSearch] = useState('')
    42→  const [statusFilter, setStatusFilter] = useState('')
    43→
    44→  useEffect(() => {
    45→    loadJobs()
    46→  }, [])
    47→
    48→  async function loadJobs() {
    49→    try {
    50→      const data = await apiCall<{ jobs: Job[] }>('/recruiter/jobs')
    51→      setJobs(data.jobs || [])
    52→    } catch {
    53→      // silent
    54→    } finally {
    55→      setLoading(false)
    56→    }
    57→  }
    58→
    59→  async function toggleJobStatus(job: Job) {
    60→    const newStatus = job.status === 'active' ? 'paused' : 'active'
    61→    try {
    62→      await apiCall(`/recruiter/jobs/${job.id}`, {
    63→        method: 'PUT',
    64→        body: { status: newStatus },
    65→      })
    66→      loadJobs()
    67→    } catch {
    68→      // silent
    69→    }
    70→  }
    71→
    72→  async function deleteJob(id: number) {
    73→    if (!confirm('Are you sure you want to delete this job posting?')) return
    74→    setDeleting(id)
    75→    try {
    76→      await apiCall(`/jobs/${id}`, { method: 'DELETE' })
    77→      loadJobs()
    78→    } catch {
    79→      // silent
    80→    } finally {
    81→      setDeleting(null)
    82→    }
    83→  }
    84→
    85→  function timeAgo(dateStr: string) {
    86→    const diff = Date.now() - new Date(dateStr).getTime()
    87→    const days = Math.floor(diff / 86400000)
    88→    if (days === 0) return 'Today'
    89→    if (days === 1) return '1 day ago'
    90→    if (days < 30) return `${days} days ago`
    91→    return `${Math.floor(days / 30)}mo ago`
    92→  }
    93→
    94→  const filtered = jobs.filter(j => {
    95→    const matchSearch = !search ||
    96→      j.title?.toLowerCase().includes(search.toLowerCase()) ||
    97→      j.location?.toLowerCase().includes(search.toLowerCase())
    98→    const matchStatus = !statusFilter || j.status === statusFilter
    99→    return matchSearch && matchStatus
   100→  })
   101→
   102→  const activeJobs = jobs.filter(j => j.status === 'active')
   103→  const totalApps = jobs.reduce((sum, j) => sum + (j.application_count || 0), 0)
   104→
   105→  return (
   106→    <div className="space-y-6">
   107→      <div className="flex flex-col gap-3 sm:flex-row sm:items-center sm:justify-between">
   108→        <div>
   109→          <h1 className="font-heading text-2xl font-bold">Job Postings</h1>
   110→          <p className="text-muted-foreground text-sm">Manage your job listings</p>
   111→        </div>
   112→        <Link to="/recruiter/jobs/new">
   113→          <Button className="gap-2">
   114→            <Plus className="h-4 w-4" /> Post New Job
   115→          </Button>
   116→        </Link>
   117→      </div>
   118→
   119→      {/* Stats */}
   120→      <div className="grid gap-3 grid-cols-3">
   121→        <Card>
   122→          <CardContent className="p-4 text-center">
   123→            <p className="text-2xl font-bold text-emerald-600">{activeJobs.length}</p>
   124→            <p className="text-xs text-muted-foreground">Active</p>
   125→          </CardContent>
   126→        </Card>
   127→        <Card>
   128→          <CardContent className="p-4 text-center">
   129→            <p className="text-2xl font-bold">{jobs.length}</p>
   130→            <p className="text-xs text-muted-foreground">Total</p>
   131→          </CardContent>
   132→        </Card>
   133→        <Card>
   134→          <CardContent className="p-4 text-center">
   135→            <p className="text-2xl font-bold text-blue-600">{totalApps}</p>
   136→            <p className="text-xs text-muted-foreground">Applications</p>
   137→          </CardContent>
   138→        </Card>
   139→      </div>
   140→
   141→      {/* Filters */}
   142→      {jobs.length > 0 && (
   143→        <div className="flex flex-col gap-3 sm:flex-row">
   144→          <div className="relative flex-1">
   145→            <Search className="absolute left-3 top-1/2 h-4 w-4 -translate-y-1/2 text-muted-foreground" />
   146→            <Input
   147→              placeholder="Search jobs by title or location..."
   148→              value={search}
   149→              onChange={e => setSearch(e.target.value)}
   150→              className="pl-9"
   151→            />
   152→          </div>
   153→          <Select
   154→            value={statusFilter}
   155→            onChange={e => setStatusFilter(e.target.value)}
   156→            className="sm:w-36"
   157→          >
   158→            <option value="">All Status</option>
   159→            <option value="active">Active</option>
   160→            <option value="paused">Paused</option>
   161→            <option value="closed">Closed</option>
   162→            <option value="draft">Draft</option>
   163→          </Select>
   164→        </div>
   165→      )}
   166→
   167→      {loading ? (
   168→        <div className="flex items-center justify-center py-16">
   169→          <div className="h-8 w-8 animate-spin rounded-full border-2 border-primary border-t-transparent" />
   170→        </div>
   171→      ) : jobs.length === 0 ? (
   172→        <Card>
   173→          <CardContent className="py-16 text-center">
   174→            <Briefcase className="mx-auto mb-3 h-10 w-10 opacity-30" />
   175→            <p className="text-muted-foreground mb-4">No job postings yet</p>
   176→            <Link to="/recruiter/jobs/new">
   177→              <Button className="gap-2"><Plus className="h-4 w-4" /> Create Your First Job</Button>
   178→            </Link>
   179→          </CardContent>
   180→        </Card>
   181→      ) : filtered.length === 0 ? (
   182→        <Card>
   183→          <CardContent className="py-12 text-center">
   184→            <Search className="mx-auto mb-3 h-8 w-8 opacity-30" />
   185→            <p className="text-muted-foreground">No jobs match your filters</p>
   186→          </CardContent>
   187→        </Card>
   188→      ) : (
   189→        <div className="space-y-3">
   190→          {filtered.map(job => (
   191→            <Card key={job.id} className="transition-shadow hover:shadow-sm">
   192→              <CardContent className="p-4">
   193→                <div className="flex flex-col gap-3 sm:flex-row sm:items-center sm:justify-between">
   194→                  <div className="min-w-0 flex-1">
   195→                    <div className="flex items-center gap-2 mb-1">
   196→                      <h3 className="font-semibold truncate">{job.title}</h3>
   197→                      <Badge variant={statusColors[job.status] || 'secondary'}>
   198→                        {job.status}
   199→                      </Badge>
   200→                    </div>
   201→                    <div className="flex flex-wrap items-center gap-3 text-sm text-muted-foreground">
   202→                      {job.location && (
   203→                        <span className="flex items-center gap-1">
   204→                          <MapPin className="h-3.5 w-3.5" /> {job.location}
   205→                        </span>
   206→                      )}
   207→                      {job.job_type && <span>{job.job_type}</span>}
   208→                      {job.salary_range && <span>{job.salary_range}</span>}
   209→                      <span className="flex items-center gap-1">
   210→                        <Users className="h-3.5 w-3.5" /> {job.application_count || 0} applicants
   211→                      </span>
   212→                      <span className="flex items-center gap-1 text-xs">
   213→                        <Clock className="h-3 w-3" /> {timeAgo(job.created_at)}
   214→                      </span>
   215→                    </div>
   216→                  </div>
   217→                  <div className="flex items-center gap-1 shrink-0">
   218→                    <Button
   219→                      variant="ghost"
   220→                      size="sm"
   221→                      onClick={() => navigate(`/recruiter/jobs/${job.id}/applicants`)}
   222→                      className="gap-1"
   223→                      title="View applicants"
   224→                    >
   225→                      <Users className="h-3.5 w-3.5" />
   226→                      <span className="text-xs">{job.application_count || 0}</span>
   227→                    </Button>
   228→                    <Button
   229→                      variant="ghost"
   230→                      size="sm"
   231→                      onClick={() => navigate(`/recruiter/jobs/${job.id}/edit`)}
   232→                      title="Edit job"
   233→                    >
   234→                      <Edit className="h-3.5 w-3.5" />
   235→                    </Button>
   236→                    <Button
   237→                      variant="ghost"
   238→                      size="sm"
   239→                      onClick={() => toggleJobStatus(job)}
   240→                      title={job.status === 'active' ? 'Pause job' : 'Activate job'}
   241→                    >
   242→                      {job.status === 'active' ? (
   243→                        <EyeOff className="h-3.5 w-3.5" />
   244→                      ) : (
   245→                        <Eye className="h-3.5 w-3.5" />
   246→                      )}
   247→                    </Button>
   248→                    <Button
   249→                      variant="ghost"
   250→                      size="sm"
   251→                      onClick={() => deleteJob(job.id)}
   252→                      disabled={deleting === job.id}
   253→                      className="text-destructive hover:text-destructive"
   254→                      title="Delete job"
   255→                    >
   256→                      <Trash2 className="h-3.5 w-3.5" />
   257→                    </Button>
   258→                  </div>
   259→                </div>
   260→              </CardContent>
   261→            </Card>
   262→          ))}
   263→        </div>
   264→      )}
   265→    </div>
   266→  )
   267→}
