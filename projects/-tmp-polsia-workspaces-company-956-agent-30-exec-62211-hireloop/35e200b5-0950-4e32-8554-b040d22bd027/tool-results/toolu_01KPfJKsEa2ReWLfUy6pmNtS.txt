    58→
    59→  useEffect(() => {

    63→  async function loadApplications() {
    64→    try {
    65→      const data = await apiCall<{ success: boolean; applications: Application[] }>('/candidate/applications')

    68→      // silent
    69→    } finally {
    70→      setLoading(false)

    75→    if (!withdrawTarget) return

    78→      await apiCall(`/candidate/applications/${withdrawTarget.id}/withdraw`, {
    79→        method: 'PUT',
    80→      })

    82→        prev.map(a => a.id === withdrawTarget.id ? { ...a, status: 'withdrawn' } : a)
    83→      )
    84→      setWithdrawTarget(null)
    85→      setSelectedApp(null)

    87→      alert(err instanceof Error ? err.message : 'Failed to withdraw application')
    88→    } finally {
    89→      setWithdrawing(false)
    90→    }

    92→

    96→    if (days === 0) return 'Today'
    97→    if (days === 1) return 'Yesterday'

    99→    if (days < 30) return `${Math.floor(days / 7)} weeks ago`
   100→    return `${Math.floor(days / 30)} months ago`
   101→  }

   133→      </div>
   134→
   135→      {/* Stats */}
   136→      <div className="grid gap-3 sm:grid-cols-4">

   138→          <CardContent className="p-4 text-center">
   139→            <p className="text-2xl font-bold">{applications.length}</p>
   140→            <p className="text-xs text-muted-foreground">Total Applied</p>

   142→        </Card>
   143→        <Card>
   144→          <CardContent className="p-4 text-center">
   145→            <p className="text-2xl font-bold text-blue-600">
   146→              {applications.filter(a => !['rejected', 'withdrawn', 'hired'].includes(a.status)).length}
   147→            </p>
   148→            <p className="text-xs text-muted-foreground">Active</p>
   149→          </CardContent>
   150→        </Card>
   151→        <Card>
   152→          <CardContent className="p-4 text-center">
   153→            <p className="text-2xl font-bold text-amber-600">
   154→              {applications.filter(a => ['reviewing', 'interviewed'].includes(a.status)).length}
   155→            </p>
   156→            <p className="text-xs text-muted-foreground">In Progress</p>
   157→          </CardContent>
   158→        </Card>
   159→        <Card>
   160→          <CardContent className="p-4 text-center">
   161→            <p className="text-2xl font-bold text-emerald-600">
   162→              {applications.filter(a => ['offered', 'hired'].includes(a.status)).length}
   163→            </p>
   164→            <p className="text-xs text-muted-foreground">Offers / Hired</p>
   165→          </CardContent>
   166→        </Card>
   167→      </div>
   168→
   169→      {/* Status filter pills */}
   170→      <div className="flex flex-wrap gap-2">

   172→          variant={!statusFilter ? 'default' : 'outline'}
   173→          size="sm"
   174→          onClick={() => setStatusFilter('')}
   175→        >

   183→              size="sm"
   184→              onClick={() => setStatusFilter(s)}

   186→              {statusConfig[s]?.label || s} ({statusCounts[s]})

   190→      </div>
   191→
   192→      {loading ? (

   222→                    app={app}

   231→
   232→          {completed.length > 0 && (
   233→            <div>
   234→              <h2 className="font-medium text-sm text-muted-foreground mb-3">Completed ({completed.length})</h2>
   235→              <div className="space-y-3">
   236→                {completed.map(app => (
   237→                  <ApplicationCard
   238→                    key={app.id}
   239→                    app={app}
   240→                    timeAgo={timeAgo}
   241→                    onClick={() => setSelectedApp(app)}

   245→            </div>
   246→          )}
   247→        </div>

   249→
   250→      {/* Application detail dialog */}
   251→      {selectedApp && (
   252→        <Dialog open={true} onClose={() => setSelectedApp(null)} className="max-w-lg">
   253→          <DialogHeader>

   256→              {selectedApp.company || selectedApp.posted_by_company || 'Company'}

   259→          </DialogHeader>
   260→          <div className="space-y-4">
   261→            {/* Status badge */}
   262→            {(() => {
   263→              const config = statusConfig[selectedApp.status] || { label: selectedApp.status, variant: 'secondary' as const, icon: Clock }
   264→              return <Badge variant={config.variant} className="w-fit">{config.label}</Badge>
   265→            })()}
   266→

   268→            <div className="grid grid-cols-2 gap-3">
   269→              <div className="rounded-lg bg-muted/50 p-3">

   272→                </p>
   273→                <p className="font-medium">{new Date(selectedApp.applied_at).toLocaleDateString()}</p>
   274→              </div>
   275→              <div className="rounded-lg bg-muted/50 p-3">
   276→                <p className="text-xs text-muted-foreground flex items-center gap-1">
   277→                  <Clock className="h-3 w-3" /> Last Update
   278→                </p>
   279→                <p className="font-medium">{timeAgo(selectedApp.updated_at)}</p>
   280→              </div>

   288→              )}

   291→                  <p className="text-xs text-muted-foreground">Match Score</p>
   292→                  <p className="font-medium text-primary">{selectedApp.match_score}%</p>
   293→                </div>
   294→              )}
   295→            </div>
   296→
   297→            {/* Progress timeline */}
   298→            <ApplicationTimeline status={selectedApp.status} appliedAt={selectedApp.applied_at} updatedAt={selectedApp.updated_at} />
   299→
   300→            {/* Cover letter */}
   301→            {selectedApp.cover_letter && (
   302→              <div>
   303→                <h4 className="font-medium text-sm mb-1">Your Cover Letter</h4>
   304→                <div className="rounded-lg bg-muted/50 p-3 text-sm whitespace-pre-wrap max-h-40 overflow-y-auto">
   305→                  {selectedApp.cover_letter}
   306→                </div>
   307→              </div>
   308→            )}

   310→            {/* Screening answers */}

   315→              <Link to={`/candidate/jobs/${selectedApp.job_id}`} className="flex-1">
   316→                <Button variant="outline" className="gap-2 w-full">
   317→                  <ExternalLink className="h-4 w-4" /> View Job
   318→                </Button>
   319→              </Link>

   321→                <Button
   322→                  variant="outline"
   323→                  onClick={() => {
   324→                    setSelectedApp(null)
   325→                    setWithdrawTarget(selectedApp)
   326→                  }}
   327→                  className="gap-2 text-destructive hover:text-destructive"
   328→                >
   329→                  <XCircle className="h-4 w-4" /> Withdraw

   331→              )}
   332→            </div>
   333→          </div>
   334→        </Dialog>
   335→      )}
   336→
   337→      {/* Withdraw confirmation dialog */}
   338→      {withdrawTarget && (

   341→            <DialogTitle className="flex items-center gap-2">
   342→              <AlertTriangle className="h-5 w-5 text-amber-500" />
   343→              Withdraw Application?
   344→            </DialogTitle>
   345→          </DialogHeader>
   346→          <div className="space-y-4">
   347→            <p className="text-sm text-muted-foreground">
   348→              Are you sure you want to withdraw your application for{' '}
   349→              <strong>{withdrawTarget.title}</strong> at{' '}
   350→              <strong>{withdrawTarget.company || withdrawTarget.posted_by_company || 'this company'}</strong>?
   351→            </p>
   352→            <div className="rounded-lg bg-amber-50 border border-amber-200 p-3 text-sm text-amber-800">
   353→              This action cannot be undone. You may need to reapply if you change your mind.
   354→            </div>
   355→            <div className="flex gap-2 pt-2">
   356→              <Button
   357→                variant="outline"
   358→                onClick={() => setWithdrawTarget(null)}
   359→                disabled={withdrawing}
   360→                className="flex-1"
   361→              >
   362→                Keep Application
   363→              </Button>
   364→              <Button
   365→                variant="destructive"
   366→                onClick={withdrawApplication}
   367→                disabled={withdrawing}
   368→                className="gap-2 flex-1"
   369→              >
   370→                {withdrawing ? (
   371→                  <div className="h-4 w-4 animate-spin rounded-full border-2 border-white border-t-transparent" />
   372→                ) : (
   373→                  <XCircle className="h-4 w-4" />
   374→                )}
   375→                Withdraw
   376→              </Button>
   377→            </div>
   378→          </div>
   379→        </Dialog>
   380→      )}
   381→    </div>
   382→  )
   383→}

   385→function ScreeningAnswersSection({
   386→  app,
   387→  parseScreeningData,
   388→}: {
   389→  app: Application
   390→  parseScreeningData: (app: Application) => { answers: Record<string, string> | null; questions: ScreeningQuestion[] }
   391→}) {
   392→  const { answers, questions } = parseScreeningData(app)
   393→
   394→  if (!answers || Object.keys(answers).length === 0) return null
   395→
   396→  return (
   397→    <div>
   398→      <h4 className="font-medium text-sm mb-2 flex items-center gap-1.5">
   399→        <ClipboardList className="h-4 w-4 text-primary" />
   400→        Your Screening Answers
   401→      </h4>
   402→      <div className="space-y-2">
   403→        {Object.entries(answers).map(([key, value], i) => {
   404→          const q = questions[i]
   405→          const questionText = q?.question || q || `Question ${i + 1}`
   406→          return (
   407→            <div key={key} className="rounded-lg bg-muted/50 p-3">
   408→              <p className="text-xs text-muted-foreground mb-1 font-medium">
   409→                {String(questionText)}
   410→              </p>
   411→              <p className="text-sm">{String(value)}</p>
   412→            </div>
   413→          )
   414→        })}
   415→      </div>
   416→    </div>
   417→  )
   418→}

   420→function ApplicationTimeline({ status, appliedAt, updatedAt }: { status: string; appliedAt: string; updatedAt: string }) {
   421→  const steps = [
   422→    { key: 'applied', label: 'Applied', icon: FileText },
   423→    { key: 'reviewing', label: 'Review', icon: Eye },
   424→    { key: 'interviewed', label: 'Interview', icon: Briefcase },
   425→    { key: 'offered', label: 'Offer', icon: CheckCircle },
   426→  ]
   427→
   428→  const stepIndex = {
   429→    applied: 0, reviewing: 1, shortlisted: 1, interviewed: 2,
   430→    offered: 3, hired: 4, rejected: -1, withdrawn: -1,
   431→  }[status] ?? 0
   432→
   433→  // Special terminal states
   434→  if (status === 'rejected' || status === 'withdrawn') {
   435→    const config = statusConfig[status]
   436→    return (
   437→      <div>
   438→        <h4 className="font-medium text-sm mb-3">Status Timeline</h4>
   439→        <div className="space-y-3">
   440→          <div className="flex items-center gap-3">
   441→            <div className="flex h-8 w-8 items-center justify-center rounded-full border-2 border-primary bg-primary/10">
   442→              <FileText className="h-4 w-4 text-primary" />
   443→            </div>
   444→            <div>
   445→              <p className="text-sm font-medium">Applied</p>
   446→              <p className="text-xs text-muted-foreground">{new Date(appliedAt).toLocaleDateString()}</p>
   447→            </div>
   448→          </div>
   449→          <div className="ml-4 h-4 w-px bg-muted" />
   450→          <div className="flex items-center gap-3">
   451→            <div className={`flex h-8 w-8 items-center justify-center rounded-full border-2 ${
   452→              status === 'rejected' ? 'border-destructive bg-destructive/10' : 'border-muted bg-muted/50'
   453→            }`}>
   454→              <XCircle className={`h-4 w-4 ${status === 'rejected' ? 'text-destructive' : 'text-muted-foreground'}`} />
   455→            </div>
   456→            <div>
   457→              <p className="text-sm font-medium">{config.label}</p>
   458→              <p className="text-xs text-muted-foreground">{new Date(updatedAt).toLocaleDateString()}</p>
   459→            </div>
   460→          </div>
   461→        </div>
   462→      </div>
   463→    )
   464→  }
   465→
   466→  if (stepIndex < 0) return null
   467→
   468→  return (
   469→    <div>
   470→      <h4 className="font-medium text-sm mb-3">Progress</h4>
   471→      <div className="flex items-center gap-1">
   472→        {steps.map((step, i) => {
   473→          const isComplete = i <= stepIndex
   474→          const isCurrent = i === stepIndex
   475→          const Icon = step.icon
   476→          return (
   477→            <div key={step.key} className="flex items-center gap-1 flex-1">
   478→              <div className={`flex items-center gap-1.5 ${isComplete ? 'text-primary' : 'text-muted-foreground/50'}`}>
   479→                <div className={`flex h-7 w-7 items-center justify-center rounded-full border-2 transition-all ${
   480→                  isCurrent ? 'border-primary bg-primary text-white' :
   481→                  isComplete ? 'border-primary bg-primary/10' : 'border-muted'
   482→                }`}>
   483→                  <Icon className="h-3.5 w-3.5" />
   484→                </div>
   485→                <span className="text-[11px] font-medium hidden sm:inline">{step.label}</span>
   486→              </div>
   487→              {i < steps.length - 1 && (
   488→                <div className={`h-0.5 flex-1 mx-1 ${i < stepIndex ? 'bg-primary' : 'bg-muted'}`} />
   489→              )}
   490→            </div>
   491→          )
   492→        })}
   493→      </div>
   494→      {/* Date labels */}
   495→      <div className="flex justify-between mt-2 px-1">
   496→        <span className="text-[10px] text-muted-foreground">
   497→          {new Date(appliedAt).toLocaleDateString()}
   498→        </span>
   499→        {stepIndex > 0 && (
   500→          <span className="text-[10px] text-muted-foreground">
   501→            Updated {new Date(updatedAt).toLocaleDateString()}
   502→          </span>
   503→        )}
   504→      </div>
   505→    </div>
   506→  )
   507→}

   509→function ApplicationCard({
   510→  app,
   511→  timeAgo,
   512→  onWithdraw,
   513→  onClick,
   514→}: {
   515→  app: Application
   516→  timeAgo: (d: string) => string
   517→  onWithdraw?: () => void
   518→  onClick?: () => void
   519→}) {
   520→  const config = statusConfig[app.status] || { label: app.status, variant: 'secondary' as const, icon: Clock }
   521→
   522→  // Progress steps
   523→  const steps = ['Applied', 'Reviewing', 'Interview', 'Offer']
   524→  const currentStep = {
   525→    applied: 0, reviewing: 1, shortlisted: 1, interviewed: 2,
   526→    offered: 3, hired: 4, rejected: -1, withdrawn: -1,
   527→  }[app.status] ?? 0
   528→
   529→  // Check if has screening answers
   530→  let hasScreening = false
   531→  try {
   532→    const answers = typeof app.screening_answers === 'string' ? JSON.parse(app.screening_answers) : app.screening_answers
   533→    hasScreening = answers && Object.keys(answers).length > 0
   534→  } catch { /* ignore */ }
   535→
   536→  return (
   537→    <Card
   538→      className="cursor-pointer transition-shadow hover:shadow-md"
   539→      onClick={onClick}
   540→    >
   541→      <CardContent className="p-4">
   542→        <div className="flex flex-col gap-3 sm:flex-row sm:items-start sm:justify-between">
