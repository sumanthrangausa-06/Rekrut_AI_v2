   206→    let currentFilter = 'upcoming';

   209→    document.addEventListener('DOMContentLoaded', async () => {
   210→      const user = await checkAuth();
   211→      if (!user) {
   212→        window.location.href = '/login.html';
   213→        return;
   214→      }
   215→      // Verify recruiter/employer role - must have company_id and appropriate role
   216→      const allowedRoles = ['employer', 'recruiter', 'hiring_manager', 'admin'];
   217→      if (!user.company_id || !allowedRoles.includes(user.role)) {
   218→        window.location.href = '/candidate-dashboard.html';
   219→        return;
   220→      }
   221→
   222→      const name = user.name || user.email.split('@')[0];
   223→      document.getElementById('user-name').textContent = name;
   224→      document.getElementById('user-avatar').textContent = name.charAt(0).toUpperCase();
   225→      document.getElementById('company-name').textContent = user.company_name || 'Company';
   226→      document.getElementById('dropdown-user-name').textContent = name;
   227→      document.getElementById('dropdown-user-email').textContent = user.email;
   228→
   229→      // Setup user menu dropdown
   230→      const userMenuToggle = document.getElementById('user-menu-toggle');
   231→      const userDropdownMenu = document.getElementById('user-dropdown-menu');
   232→      if (userMenuToggle && userDropdownMenu) {
   233→        userMenuToggle.addEventListener('click', (e) => {
   234→          e.stopPropagation();
   235→          userDropdownMenu.classList.toggle('active');
   236→        });
   237→        document.addEventListener('click', () => {
   238→          userDropdownMenu.classList.remove('active');
   239→        });
   240→      }
   241→
   242→      // Setup filter tabs
   243→      document.querySelectorAll('.filter-tab').forEach(tab => {
   244→        tab.addEventListener('click', () => {
   245→          document.querySelectorAll('.filter-tab').forEach(t => t.classList.remove('active'));
   246→          tab.classList.add('active');
   247→          currentFilter = tab.dataset.status;
   248→          renderInterviews();
   249→        });
   250→      });
   251→
   252→      // Check URL params for schedule action
   253→      const params = new URLSearchParams(window.location.search);
   254→      if (params.get('schedule')) {
   255→        loadCandidates().then(() => {
   256→          document.getElementById('candidate-select').value = params.get('schedule');
   257→          showScheduleModal();
   258→        });
   259→      }
   260→
   261→      loadCandidates();
   262→      loadInterviews();
   263→    });

   265→    async function loadCandidates() {
   266→      try {
   267→        const data = await apiCall('/recruiter/applications');
   268→        if (data.applications) {
   269→          const select = document.getElementById('candidate-select');
   270→          data.applications
   271→            .filter(a => a.status !== 'rejected')
   272→            .forEach(app => {
   273→              const option = document.createElement('option');
   274→              option.value = app.id;
   275→              option.textContent = `${app.candidate_name || 'Unknown'} - ${app.job_title}`;
   276→              select.appendChild(option);
   277→            });
   278→        }
   279→      } catch (err) {
   280→        console.error('Failed to load candidates:', err);
   281→      }
   282→    }
   283→
   284→    async function loadInterviews() {
   285→      try {
   286→        const data = await apiCall('/recruiter/interviews');
   287→        if (data.interviews) {
   288→          interviews = data.interviews;
   289→          updateStats(interviews);
   290→          renderInterviews();
   291→        } else {
   292→          renderInterviews();
   293→        }
   294→      } catch (err) {
   295→        console.error('Failed to load interviews:', err);
   296→        document.getElementById('interviews-list').innerHTML = `
   297→          <div class="empty-state">
   298→            <p>No interviews scheduled yet</p>
   299→            <button onclick="showScheduleModal()" class="btn btn-primary btn-sm">Schedule an Interview</button>
   300→          </div>
   301→        `;
   302→      }
   303→    }

   305→    function updateStats(interviewList) {
   306→      const today = new Date().toDateString();
   307→      const weekFromNow = new Date();
   308→      weekFromNow.setDate(weekFromNow.getDate() + 7);
   309→
   310→      document.getElementById('today-count').textContent =
   311→        interviewList.filter(i => new Date(i.scheduled_at).toDateString() === today).length;
   312→
   313→      document.getElementById('upcoming-count').textContent =
   314→        interviewList.filter(i => {
   315→          const date = new Date(i.scheduled_at);
   316→          return date > new Date() && date <= weekFromNow;
   317→        }).length;
   318→
   319→      document.getElementById('completed-count').textContent =
   320→        interviewList.filter(i => i.status === 'completed').length;
   321→
   322→      document.getElementById('video-count').textContent =
   323→        interviewList.filter(i => i.interview_type === 'video').length;
   324→    }

   326→    function renderInterviews() {
   327→      const container = document.getElementById('interviews-list');
   328→      const now = new Date();
   329→      const today = now.toDateString();
   330→
   331→      let filtered = interviews;
   332→      if (currentFilter === 'upcoming') {
   333→        filtered = filtered.filter(i => new Date(i.scheduled_at) > now && i.status !== 'completed');
   334→      } else if (currentFilter === 'today') {
   335→        filtered = filtered.filter(i => new Date(i.scheduled_at).toDateString() === today);
   336→      } else if (currentFilter === 'completed') {
   337→        filtered = filtered.filter(i => i.status === 'completed');
   338→      }
   339→
   340→      if (filtered.length === 0) {
   341→        container.innerHTML = `
   342→          <div class="empty-state">
   343→            <p>No interviews ${currentFilter === 'all' ? '' : currentFilter}</p>
   344→            <button onclick="showScheduleModal()" class="btn btn-primary btn-sm">Schedule an Interview</button>
   345→          </div>
   346→        `;
   347→        return;
   348→      }
   349→
   350→      // Sort by date
   351→      filtered.sort((a, b) => new Date(a.scheduled_at) - new Date(b.scheduled_at));
   352→
   353→      container.innerHTML = filtered.map(interview => {
   354→        const date = new Date(interview.scheduled_at);
   355→        const isPast = date < now;
   356→        return `
   357→          <div class="interview-card ${isPast ? 'past' : ''}">
   358→            <div class="interview-time-block">
   359→              <span class="interview-date">${date.toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' })}</span>
   360→              <span class="interview-time">${date.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })}</span>
   361→            </div>
   362→            <div class="interview-details">
   363→              <div class="interview-candidate">
   364→                <div class="candidate-avatar">${(interview.candidate_name || 'U').charAt(0).toUpperCase()}</div>
   365→                <div>
   366→                  <h3>${interview.candidate_name || 'Unknown Candidate'}</h3>
   367→                  <p>${interview.job_title || 'Position'}</p>
   368→                </div>
   369→              </div>
   370→            </div>
   371→            <div class="interview-meta">
   372→              <span class="interview-type-badge ${interview.interview_type}">${interview.interview_type || 'Interview'}</span>
   373→              <span class="interview-duration">${interview.duration || 60} min</span>
   374→            </div>
   375→            <div class="interview-actions">
   376→              ${!isPast ? `
   377→                <button onclick="joinInterview('${interview.id}')" class="btn btn-primary btn-sm">Join</button>
   378→                <button onclick="reschedule('${interview.id}')" class="btn btn-ghost btn-sm">Reschedule</button>
   379→              ` : `
   380→                <button onclick="markCompleted('${interview.id}')" class="btn btn-ghost btn-sm">Mark Complete</button>
   381→              `}
   382→              <button onclick="cancelInterview('${interview.id}')" class="btn btn-ghost btn-sm danger">Cancel</button>
   383→            </div>
   384→          </div>
   385→        `;
   386→      }).join('');
   387→    }

   389→    function showScheduleModal() {
   390→      document.getElementById('schedule-modal').style.display = 'flex';

   394→    }

   401→    async function scheduleInterview(e) {
   402→      e.preventDefault();
   403→      const applicationId = document.getElementById('candidate-select').value;
   404→      const type = document.getElementById('interview-type').value;
   405→      const date = document.getElementById('interview-date').value;
   406→      const time = document.getElementById('interview-time').value;
   407→      const duration = document.getElementById('interview-duration').value;
   408→      const notes = document.getElementById('interview-notes').value;
   409→
   410→      const scheduledAt = new Date(`${date}T${time}`);
   411→
   412→      try {
   413→        const result = await apiCall('/recruiter/interviews', {
   414→          method: 'POST',
   415→          body: JSON.stringify({
   416→            application_id: applicationId,
   417→            interview_type: type,
   418→            scheduled_at: scheduledAt.toISOString(),
   419→            duration: parseInt(duration),
   420→            notes
   421→          })
   422→        });
   423→
   424→        if (result.success) {
   425→          closeScheduleModal();
   426→          loadInterviews();
   427→        } else {
   428→          alert(result.error || 'Failed to schedule interview');
   429→        }
   430→      } catch (err) {
   431→        alert('Failed to schedule interview');
   432→      }
   433→    }

   458→    async function cancelInterview(id) {

   460→      try {
   461→        const result = await apiCall(`/recruiter/interviews/${id}`, {
   462→          method: 'DELETE'

   464→        if (result.success) {
   465→          loadInterviews();
   466→        }
   467→      } catch (err) {
   468→        alert('Failed to cancel interview');
   469→      }
