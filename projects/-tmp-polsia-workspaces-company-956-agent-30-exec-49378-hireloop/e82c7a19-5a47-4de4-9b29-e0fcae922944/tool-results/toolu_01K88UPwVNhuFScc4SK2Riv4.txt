   322→    function formatDate(dateStr) {
   323→      const date = new Date(dateStr);
   324→      const now = new Date();
   325→      const diff = now - date;
   326→      const days = Math.floor(diff / (1000 * 60 * 60 * 24));
   327→      if (days === 0) return 'Today';
   328→      if (days === 1) return 'Yesterday';
   329→      if (days < 7) return `${days} days ago`;
   330→      return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
   331→    }

   333→    async function updateStatus(appId, status) {
   334→      try {
   335→        const result = await apiCall(`/recruiter/applications/${appId}/status`, {
   336→          method: 'PUT',
   337→          body: JSON.stringify({ status })
   338→        });
   339→        if (result.success) {
   340→          loadApplications();
   341→        }
   342→      } catch (err) {
   343→        alert('Failed to update status');
   344→      }
   345→    }
