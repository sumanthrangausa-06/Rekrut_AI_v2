   546→async function analyzeFeedbackWithAI(responses, comments) {
   547→  try {
   548→    const prompt = `Analyze this employee feedback and provide insights:
   549→
   550→Responses: ${JSON.stringify(responses)}
   551→Comments: ${comments}
   552→
   553→Provide:
   554→1. Sentiment (positive/neutral/negative)
   555→2. Key themes identified
   556→3. Action items for management
   557→4. Risk level (low/medium/high) if negative patterns detected`;
   558→
   559→    const analysis = await polsiaAI.chat([
   560→      { role: 'user', content: prompt }
   561→    ], { max_tokens: 1024 });
   562→
   563→    return { analysis: analysis, analyzed_at: new Date().toISOString() };
   564→  } catch (err) {
   565→    console.error('Error analyzing feedback:', err);
   566→    return { error: 'Analysis failed' };
   567→  }

   570→async function getOnboardingAssistantResponse(messages, policyContext, user) {
   571→  try {
   572→    const systemPrompt = `You are an onboarding assistant for new employees. You help them understand company policies, complete paperwork, and answer questions about their first days.
   573→
   574→Company Policies and Information:
   575→${policyContext}
   576→
   577→Employee name: ${user.name || 'there'}
   578→
   579→Be friendly, helpful, and direct. If you don't know something, say so and suggest they contact HR.`;
   580→
   581→    const response = await polsiaAI.chat([
   582→      { role: 'system', content: systemPrompt },
   583→      ...messages.map(m => ({ role: m.role, content: m.content }))
   584→    ], { max_tokens: 512 });
   585→
   586→    return response;
   587→  } catch (err) {
   588→    console.error('Error getting AI response:', err);
   589→    return "I'm having trouble connecting right now. Please try again or contact HR directly.";
   590→  }
