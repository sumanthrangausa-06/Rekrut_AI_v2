     8→async function chat(message, options = {}) {
     9→  const response = await anthropic.messages.create({
    10→    max_tokens: options.maxTokens || 8192,
    11→    messages: [{ role: 'user', content: message }],
    12→    system: options.system,
    13→  }, {
    14→    headers: options.subscriptionId ? { 'X-Subscription-ID': options.subscriptionId } : {}
    15→  });
    16→  return response.content[0].text;
    17→}
