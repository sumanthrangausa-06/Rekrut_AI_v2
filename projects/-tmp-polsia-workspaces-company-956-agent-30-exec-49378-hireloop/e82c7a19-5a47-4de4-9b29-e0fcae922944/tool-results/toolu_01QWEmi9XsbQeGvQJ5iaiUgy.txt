   123→// Complete interview and get overall feedback
   124→router.post('/:id/complete', authMiddleware, async (req, res) => {
   125→  try {
   126→    const interview = await pool.query(
   127→      'SELECT * FROM interviews WHERE id = $1 AND user_id = $2',
   128→      [req.params.id, req.user.id]
   129→    );
   130→
   131→    if (interview.rows.length === 0) {
   132→      return res.status(404).json({ error: 'Interview not found' });
   133→    }
   134→
   135→    const interviewData = interview.rows[0];
   136→    const responses = interviewData.responses || [];
   137→    const validResponses = responses.filter(r => r && r.analysis);
   138→
   139→    if (validResponses.length === 0) {
   140→      return res.status(400).json({ error: 'No responses to evaluate' });
   141→    }
   142→
   143→    // Generate overall feedback
   144→    const overallFeedback = await generateOverallFeedback(
   145→      validResponses.map(r => r.analysis),
   146→      { subscriptionId: req.user.stripe_subscription_id }
   147→    );
   148→
   149→    // Calculate duration
   150→    const startTime = new Date(interviewData.created_at);
   151→    const duration = Math.floor((Date.now() - startTime.getTime()) / 1000);
   152→
   153→    // Update interview record
   154→    await pool.query(
   155→      `UPDATE interviews SET
   156→        status = 'completed',
   157→        ai_feedback = $1,
   158→        overall_score = $2,
   159→        duration_seconds = $3,
   160→        completed_at = NOW()
   161→       WHERE id = $4`,
   162→      [JSON.stringify(overallFeedback), overallFeedback.overall_score, duration, req.params.id]
   163→    );
   164→
   165→    // Update OmniScore with interview results
   166→    let omniscoreUpdate = null;
   167→    try {
   168→      omniscoreUpdate = await omniscoreService.addInterviewComponent(
   169→        req.user.id,
   170→        req.params.id,
   171→        overallFeedback.overall_score
   172→      );
   173→
   174→      // Update role-specific score if job title is available
   175→      if (interviewData.job_id) {
   176→        const job = await pool.query('SELECT title FROM jobs WHERE id = $1', [interviewData.job_id]);
   177→        if (job.rows.length > 0) {
   178→          await omniscoreService.updateRoleScore(
   179→            req.user.id,
   180→            job.rows[0].title,
   181→            overallFeedback.overall_score
   182→          );
   183→        }
   184→      }
   185→    } catch (scoreErr) {
   186→      console.error('OmniScore update error:', scoreErr);
   187→      // Don't fail the request if OmniScore update fails
   188→    }
   189→
   190→    res.json({
   191→      success: true,
   192→      overall_feedback: overallFeedback,
   193→      duration_seconds: duration,
   194→      response_count: validResponses.length,
   195→      omniscore: omniscoreUpdate
   196→    });
   197→  } catch (err) {
   198→    console.error('Complete interview error:', err);
   199→    res.status(500).json({ error: 'Failed to complete interview' });
   200→  }
   201→});

   541→// Get practice statistics
   542→router.get('/practice/stats', authMiddleware, async (req, res) => {
   543→  try {
   544→    const stats = await pool.query(
   545→      `SELECT
   546→        COUNT(*) as total_questions,
   547→        AVG(score) as average_score,
   548→        MAX(created_at) as last_practice
   549→       FROM practice_sessions
   550→       WHERE user_id = $1`,
   551→      [req.user.id]
   552→    );
   553→
   554→    // Calculate improvement (compare first half vs second half of sessions)
   555→    const improvement = await pool.query(
   556→      `WITH ordered_sessions AS (
   557→        SELECT score, ROW_NUMBER() OVER (ORDER BY created_at) as rn,
   558→               COUNT(*) OVER () as total
   559→        FROM practice_sessions
   560→        WHERE user_id = $1
   561→      )
   562→      SELECT
   563→        AVG(CASE WHEN rn <= total/2 THEN score END) as first_half_avg,
   564→        AVG(CASE WHEN rn > total/2 THEN score END) as second_half_avg
   565→      FROM ordered_sessions`,
   566→      [req.user.id]
   567→    );
   568→
   569→    let improvementPercent = null;
   570→    if (improvement.rows[0].first_half_avg && improvement.rows[0].second_half_avg) {
   571→      const firstHalf = parseFloat(improvement.rows[0].first_half_avg);
   572→      const secondHalf = parseFloat(improvement.rows[0].second_half_avg);
   573→      improvementPercent = ((secondHalf - firstHalf) / firstHalf) * 100;
   574→    }
   575→
   576→    // Calculate day streak
   577→    const streakResult = await pool.query(
   578→      `WITH RECURSIVE date_series AS (
   579→        SELECT CURRENT_DATE::date as check_date, 0 as days_back
   580→        UNION ALL
   581→        SELECT (check_date - INTERVAL '1 day')::date, days_back + 1
   582→        FROM date_series
   583→        WHERE days_back < 30
   584→      )
   585→      SELECT COUNT(*) as streak
   586→      FROM date_series d
   587→      WHERE EXISTS (
   588→        SELECT 1 FROM practice_sessions
   589→        WHERE user_id = $1 AND DATE(created_at) = d.check_date
   590→      )
   591→      AND d.check_date <= CURRENT_DATE
   592→      AND NOT EXISTS (
   593→        SELECT 1 FROM date_series d2
   594→        WHERE d2.check_date > d.check_date
   595→        AND d2.check_date <= CURRENT_DATE
   596→        AND NOT EXISTS (
   597→          SELECT 1 FROM practice_sessions
   598→          WHERE user_id = $1 AND DATE(created_at) = d2.check_date
   599→        )
   600→      )`,
   601→      [req.user.id]
   602→    );
   603→
   604→    res.json({
   605→      success: true,
