    17→// Robust JSON parser — handles malformed LLM output (markdown fences, trailing commas, unescaped chars)
    18→function safeParseJSON(text) {
    19→  if (!text || typeof text !== 'string') return null;
    20→
    21→  // Step 1: Strip markdown code fences
    22→  let cleaned = text.replace(/```(?:json)?\s*/gi, '').replace(/```\s*/g, '');
    23→
    24→  // Step 2: Extract the outermost JSON object or array
    25→  const match = cleaned.match(/(\{[\s\S]*\}|\[[\s\S]*\])/);
    26→  if (!match) return null;
    27→  cleaned = match[0];
    28→
    29→  // Step 3: Try direct parse first
    30→  try { return JSON.parse(cleaned); } catch (e) { /* continue to repair */ }
    31→
    32→  // Step 4: Remove trailing commas before } or ]
    33→  let repaired = cleaned.replace(/,\s*([}\]])/g, '$1');
    34→
    35→  // Step 5: Strip control characters (except \n \r \t) within strings
    36→  repaired = repaired.replace(/[\x00-\x08\x0B\x0C\x0E-\x1F]/g, '');
    37→
    38→  try { return JSON.parse(repaired); } catch (e) { /* continue */ }
    39→
    40→  // Step 6: Fix unescaped newlines inside JSON strings (replace literal newlines with \n)
    41→  repaired = repaired.replace(/"([^"]*?)"/g, (match, content) => {
    42→    return '"' + content.replace(/\n/g, '\\n').replace(/\r/g, '\\r').replace(/\t/g, '\\t') + '"';
    43→  });
    44→
    45→  try { return JSON.parse(repaired); } catch (e) { /* continue */ }
    46→
    47→  // Step 7: Last resort — try to fix common quote escaping issues
    48→  // Replace curly/smart quotes with straight quotes
    49→  repaired = repaired.replace(/[\u201C\u201D\u201E\u201F\u2033\u2036]/g, '"');
    50→  repaired = repaired.replace(/[\u2018\u2019\u201A\u201B\u2032\u2035]/g, "'");
    51→
    52→  try { return JSON.parse(repaired); } catch (e) {
    53→    console.warn('[safeParseJSON] All repair attempts failed:', e.message);
    54→    return null;
    55→  }
    56→}

    58→// Upload a base64 image frame to R2 and return the CDN URL

    60→async function uploadFrameToR2(base64Data, index) {
    61→  try {
    62→    // Strip data URI prefix if present
    63→    const raw = base64Data.replace(/^data:image\/\w+;base64,/, '');
    64→    const buffer = Buffer.from(raw, 'base64');
    65→
    66→    const formData = new FormData();
    67→    formData.append('file', buffer, {
    68→      filename: `video-frame-${Date.now()}-${index}.jpg`,
    69→      contentType: 'image/jpeg'
    70→    });
    71→
    72→    const res = await fetch('https://polsia.com/api/proxy/r2/upload', {
    73→      method: 'POST',
    74→      headers: {
    75→        'Authorization': `Bearer ${process.env.POLSIA_API_KEY}`,
    76→        ...formData.getHeaders()
    77→      },
    78→      body: formData
    79→    });
    80→
    81→    const result = await res.json();
    82→    if (!result.success) {
    83→      console.error(`R2 frame upload failed (frame ${index}):`, result.error);
    84→      return null;
    85→    }
    86→    return result.file.url;
    87→  } catch (err) {
    88→    console.error(`R2 frame upload error (frame ${index}):`, err.message);
    89→    return null;

    95→async function chat(message, options = {}) {
    96→  // Handle both string messages and arrays of message objects
    97→  let messages;
    98→  if (Array.isArray(message)) {
    99→    // Filter out system messages and extract them
   100→    const systemMessages = message.filter(m => m.role === 'system');
   101→    const nonSystemMessages = message.filter(m => m.role !== 'system');
   102→
   103→    // Use the first system message content as the system prompt if not already set
   104→    if (systemMessages.length > 0 && !options.system) {
   105→      options.system = systemMessages[0].content;
   106→    }
   107→
   108→    // Ensure messages have proper format (content must be string or array)
   109→    messages = nonSystemMessages.map(m => ({
   110→      role: m.role,
   111→      content: typeof m.content === 'string' ? m.content : String(m.content)
   112→    }));
   113→  } else {
   114→    // Simple string message
   115→    messages = [{ role: 'user', content: String(message) }];
   116→  }
   117→
   118→  // Ensure we have at least one message
   119→  if (messages.length === 0) {
   120→    messages = [{ role: 'user', content: 'Hello' }];
   121→  }
   122→
   123→  // Use the provider service with automatic fallback
   124→  return aiProvider.chatCompletion(messages, options);
   125→}

   497→// Normalize vision analysis result — different AI models use different field names
   498→// Maps common variations to the expected field names
   499→function normalizeVisionResult(parsed) {
   500→  if (!parsed || typeof parsed !== 'object') return parsed;
   501→  const result = { ...parsed };
   502→
   503→  // Normalize field name variations
   504→  const fieldMappings = {
   505→    eye_contact: ['eye_contact', 'eyeContact', 'eye_contact_score', 'gaze'],
   506→    facial_expressions: ['facial_expressions', 'facialExpressions', 'expressions', 'facial_expression'],
   507→    body_language: ['body_language', 'bodyLanguage', 'body_language_score', 'posture'],
   508→    professional_appearance: ['professional_appearance', 'professionalAppearance', 'appearance', 'camera_setup'],
   509→    overall_presentation: ['overall_presentation', 'overallPresentation', 'overall_score', 'overall'],
   510→  };
   511→
   512→  for (const [target, alternatives] of Object.entries(fieldMappings)) {
   513→    if (!result[target]) {
   514→      for (const alt of alternatives) {
   515→        if (result[alt] !== undefined) {
   516→          result[target] = result[alt];
   517→          break;
   518→        }
   519→      }
   520→    }
   521→    // Normalize score-only values into {score, feedback} objects
   522→    if (target !== 'overall_presentation' && result[target] !== undefined) {
   523→      if (typeof result[target] === 'number') {
   524→        result[target] = { score: result[target], feedback: '' };
   525→      } else if (typeof result[target] === 'object' && result[target] !== null) {
   526→        // Ensure score is a number
   527→        if (result[target].score === undefined && result[target].rating !== undefined) {
   528→          result[target].score = result[target].rating;
   529→        }
   530→        if (typeof result[target].score !== 'number') {
   531→          result[target].score = parseInt(result[target].score) || 5;
   532→        }
   533→        result[target].feedback = result[target].feedback || result[target].assessment || result[target].comment || '';
   534→      }
   535→    }
   536→  }
   537→
   538→  // Normalize overall_presentation to a number
   539→  if (typeof result.overall_presentation === 'object' && result.overall_presentation !== null) {
   540→    result.overall_presentation = result.overall_presentation.score || result.overall_presentation.rating || 5;
   541→  }
   542→  if (typeof result.overall_presentation !== 'number') {
   543→    result.overall_presentation = parseInt(result.overall_presentation) || 5;
   544→  }
   545→
   546→  // Ensure timestamped_notes is an array
   547→  if (!Array.isArray(result.timestamped_notes)) {
   548→    result.timestamped_notes = result.timestamped_notes ? [result.timestamped_notes] : [];
   549→  }
   550→
   551→  return result;
   552→}

   748→// Analyze speech transcription for communication quality
   749→function analyzeSpeechPatterns(transcription, durationSeconds) {
   750→  const words = transcription.trim().split(/\s+/).filter(w => w.length > 0);
   751→  const wordCount = words.length;
   752→  const wpm = durationSeconds > 0 ? Math.round((wordCount / durationSeconds) * 60) : 0;
   753→
   754→  // Filler word detection
   755→  const fillerWords = ['um', 'uh', 'like', 'you know', 'so', 'basically', 'actually', 'literally', 'right', 'okay so', 'i mean'];
   756→  const fillerCounts = {};
   757→  let totalFillers = 0;
   758→  const lowerTranscription = transcription.toLowerCase();
   759→
   760→  for (const filler of fillerWords) {
   761→    const regex = new RegExp(`\\b${filler}\\b`, 'gi');
   762→    const matches = lowerTranscription.match(regex);
   763→    if (matches && matches.length > 0) {
   764→      fillerCounts[filler] = matches.length;
   765→      totalFillers += matches.length;
   766→    }
   767→  }
   768→
   769→  const fillerRate = wordCount > 0 ? (totalFillers / wordCount) * 100 : 0;
   770→
   771→  // Pace analysis
   772→  let paceAssessment = 'good';
   773→  let paceFeedback = '';
   774→  if (wpm < 100) {
   775→    paceAssessment = 'too_slow';
   776→    paceFeedback = 'Your speaking pace is quite slow. Try to speak more naturally and with more energy. Aim for 120-150 words per minute.';
   777→  } else if (wpm < 120) {
   778→    paceAssessment = 'slightly_slow';
   779→    paceFeedback = 'Your pace is a bit slow. Try to increase your speaking speed slightly for more engaging delivery.';
   780→  } else if (wpm <= 160) {
   781→    paceAssessment = 'good';
   782→    paceFeedback = 'Your speaking pace is natural and easy to follow. Well done!';
   783→  } else if (wpm <= 180) {
   784→    paceAssessment = 'slightly_fast';
   785→    paceFeedback = 'You\'re speaking a bit quickly. Try to slow down slightly and pause between key points.';
   786→  } else {
   787→    paceAssessment = 'too_fast';
   788→    paceFeedback = 'You\'re speaking too fast — this can signal nervousness. Take deliberate pauses between sentences and breathe.';
   789→  }
   790→
   791→  // Sentence structure analysis
   792→  const sentences = transcription.split(/[.!?]+/).filter(s => s.trim().length > 0);
   793→  const avgSentenceLength = sentences.length > 0 ? Math.round(wordCount / sentences.length) : 0;
   794→
   795→  // Communication score (1-10)
   796→  let communicationScore = 7;
   797→  // Penalize for filler words
   798→  if (fillerRate > 5) communicationScore -= 2;
   799→  else if (fillerRate > 3) communicationScore -= 1;
   800→  // Penalize for bad pace
   801→  if (paceAssessment === 'too_slow' || paceAssessment === 'too_fast') communicationScore -= 1.5;
   802→  else if (paceAssessment === 'slightly_slow' || paceAssessment === 'slightly_fast') communicationScore -= 0.5;
   803→  // Reward for good sentence structure
   804→  if (avgSentenceLength >= 10 && avgSentenceLength <= 25) communicationScore += 0.5;
   805→  // Penalize for very short responses
   806→  if (wordCount < 50) communicationScore -= 2;
   807→  else if (wordCount < 100) communicationScore -= 1;
   808→  // Reward for substantial responses
   809→  if (wordCount >= 150 && wordCount <= 400) communicationScore += 0.5;
   810→
   811→  communicationScore = Math.max(1, Math.min(10, Math.round(communicationScore * 10) / 10));
   812→
   813→  return {
   814→    word_count: wordCount,
   815→    duration_seconds: durationSeconds,
   816→    words_per_minute: wpm,
   817→    filler_words: fillerCounts,
   818→    total_fillers: totalFillers,
   819→    filler_rate: Math.round(fillerRate * 10) / 10,
   820→    pace: {
   821→      assessment: paceAssessment,
   822→      wpm,
   823→      feedback: paceFeedback
   824→    },
   825→    sentence_count: sentences.length,
   826→    avg_sentence_length: avgSentenceLength,
   827→    communication_score: communicationScore,
   828→    tips: generateSpeechTips(fillerRate, wpm, wordCount, totalFillers, fillerCounts)
   829→  };
   830→}

   862→async function transcribeAudioWithWhisper(audioBase64, options = {}) {
   863→  try {
   864→    // Strip full data URL prefix including codec params (e.g., data:audio/webm;codecs=opus;base64,)
   865→    const base64Data = audioBase64.replace(/^data:audio\/[^;]+(?:;[^;]*)*;base64,/, '');
   866→    const buffer = Buffer.from(base64Data, 'base64');
   867→
   868→    // Extract MIME type (e.g., audio/webm from data:audio/webm;codecs=opus;base64,)
   869→    const mimeMatch = audioBase64.match(/^data:(audio\/[^;]+)/);
   870→    const mimeType = mimeMatch ? mimeMatch[1] : 'audio/webm';
   871→    const ext = mimeType.includes('mp4') ? 'mp4' : mimeType.includes('ogg') ? 'ogg' : 'webm';
   872→    console.log(`[whisper] Audio format: ${mimeType}, ext: ${ext}, buffer size: ${buffer.length} bytes`);

   874→    // Use provider service with automatic fallback (Whisper → Parakeet)
   875→    return await aiProvider.transcribeAudio(buffer, `recording.${ext}`, mimeType, {
   876→      subscriptionId: options.subscriptionId,
   877→      module: 'mock_interview', feature: 'transcription',
   878→    });
   879→  } catch (err) {
   880→    console.error('Whisper transcription error:', err.message || err);

  1457→// Fallback chain: Cache → OpenAI TTS → null (browser Web Speech API fallback)
  1458→async function textToSpeech(text, options = {}) {
  1459→  const pool = require('./db');
  1460→  const crypto = require('crypto');
  1461→
  1462→  try {
  1463→    const voice = options.voice || 'nova';
  1464→    const speed = options.speed || 1.0;
  1465→
  1466→    // Truncate text to ~4000 chars (TTS API limit is 4096)
  1467→    const truncatedText = text.length > 4000 ? text.substring(0, 4000) : text;
  1468→
  1469→    // Generate cache key from text + voice
  1470→    const cacheKey = crypto.createHash('sha256').update(`${voice}:${truncatedText}`).digest('hex');
  1471→
  1472→    // Check cache first — serves audio even when daily token limit is exhausted
  1473→    try {
  1474→      const cached = await pool.query(
  1475→        'SELECT audio_data FROM tts_cache WHERE text_hash = $1',
  1476→        [cacheKey]
  1477→      );
  1478→      if (cached.rows.length > 0 && cached.rows[0].audio_data) {
  1479→        const buf = Buffer.from(cached.rows[0].audio_data);
  1480→        console.log(`[tts] Cache HIT: ${truncatedText.length} chars, ${buf.length} bytes (voice: ${voice})`);
  1481→        return buf;
  1482→      }
  1483→    } catch (cacheErr) {
  1484→      // Cache miss or table doesn't exist yet — continue to API
  1485→      console.log(`[tts] Cache check failed (${cacheErr.message}), proceeding to API`);
  1486→    }
  1487→
  1488→    console.log(`[tts] Cache MISS — generating speech: ${truncatedText.length} chars, voice: ${voice}`);
  1489→
  1490→    // Use provider service with automatic fallback
  1491→    const buf = await aiProvider.textToSpeech(truncatedText, {
  1492→      voice,
  1493→      speed,
  1494→      subscriptionId: options.subscriptionId,
  1495→      module: 'mock_interview', feature: 'tts',
  1496→    });
  1497→
  1498→    if (!buf) {
  1499→      console.log('[tts] All providers failed, returning null for browser fallback');
  1500→      return null;
  1501→    }
  1502→
  1503→    console.log(`[tts] Generated ${buf.length} bytes of MP3 audio`);
  1504→
  1505→    // Cache the audio for future use (non-blocking)
  1506→    if (buf.length > 100) {
  1507→      pool.query(
  1508→        `INSERT INTO tts_cache (text_hash, voice, audio_data, text_preview)
  1509→         VALUES ($1, $2, $3, $4)
  1510→         ON CONFLICT (text_hash) DO NOTHING`,
  1511→        [cacheKey, voice, buf, truncatedText.substring(0, 100)]
  1512→      ).then(() => {
  1513→        console.log(`[tts] Cached audio for: "${truncatedText.substring(0, 50)}..."`);
  1514→      }).catch(err => {
  1515→        // Non-fatal — caching failure doesn't break TTS
  1516→        console.warn(`[tts] Cache write failed: ${err.message}`);
  1517→      });
  1518→    }
  1519→
  1520→    return buf;
  1521→  } catch (err) {
  1522→    console.error('[tts] Error:', err.message || err);
  1523→    return null;
  1524→  }
  1525→}
