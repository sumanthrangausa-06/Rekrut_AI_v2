   186→// Complete interview and get overall feedback
   187→router.post('/:id/complete', authMiddleware, async (req, res) => {
   188→  try {
   189→    const interview = await pool.query(
   190→      'SELECT * FROM interviews WHERE id = $1 AND user_id = $2',
   191→      [req.params.id, req.user.id]
   192→    );
   193→
   194→    if (interview.rows.length === 0) {
   195→      return res.status(404).json({ error: 'Interview not found' });
   196→    }
   197→
   198→    const interviewData = interview.rows[0];
   199→    const responses = interviewData.responses || [];
   200→    const validResponses = responses.filter(r => r && r.analysis);
   201→
   202→    if (validResponses.length === 0) {
   203→      return res.status(400).json({ error: 'No responses to evaluate' });
   204→    }
   205→
   206→    // Generate overall feedback
   207→    const overallFeedback = await generateOverallFeedback(
   208→      validResponses.map(r => r.analysis),
   209→      { subscriptionId: req.user.stripe_subscription_id }
   210→    );
   211→
   212→    // Calculate duration
   213→    const startTime = new Date(interviewData.created_at);
   214→    const duration = Math.floor((Date.now() - startTime.getTime()) / 1000);
   215→
   216→    // Update interview record
   217→    await pool.query(
   218→      `UPDATE interviews SET
   219→        status = 'completed',
   220→        ai_feedback = $1,
   221→        overall_score = $2,
   222→        duration_seconds = $3,
   223→        completed_at = NOW()
   224→       WHERE id = $4`,
   225→      [JSON.stringify(overallFeedback), overallFeedback.overall_score, duration, req.params.id]
   226→    );
   227→
   228→    // Update OmniScore with interview results
   229→    let omniscoreUpdate = null;
   230→    try {
   231→      omniscoreUpdate = await omniscoreService.addInterviewComponent(
   232→        req.user.id,
   233→        req.params.id,
   234→        overallFeedback.overall_score
   235→      );
   236→
   237→      // Update role-specific score if job title is available
   238→      if (interviewData.job_id) {
   239→        const job = await pool.query('SELECT title FROM jobs WHERE id = $1', [interviewData.job_id]);
   240→        if (job.rows.length > 0) {
   241→          await omniscoreService.updateRoleScore(
   242→            req.user.id,
   243→            job.rows[0].title,
   244→            overallFeedback.overall_score
   245→          );
   246→        }
   247→      }
   248→    } catch (scoreErr) {
   249→      console.error('OmniScore update error:', scoreErr);
   250→      // Don't fail the request if OmniScore update fails
   251→    }
   252→
   253→    res.json({
   254→      success: true,
   255→      overall_feedback: overallFeedback,
   256→      duration_seconds: duration,
   257→      response_count: validResponses.length,
   258→      omniscore: omniscoreUpdate
   259→    });
   260→  } catch (err) {
   261→    console.error('Complete interview error:', err);
   262→    res.status(500).json({ error: 'Failed to complete interview' });
   263→  }

   705→      improvementPercent = ((secondHalf - firstHalf) / firstHalf) * 100;
   706→    }

   708→    // Calculate day streak
   709→    const streakResult = await pool.query(
   710→      `WITH RECURSIVE date_series AS (

   727→        AND d2.check_date <= CURRENT_DATE
   728→        AND NOT EXISTS (

  1662→    const WHISPER_HALLUCINATIONS = [
  1663→      'ご視聴ありがとうございました', '視聴ありがとうございました', 'ありがとうございました',
  1664→      'ご視聴ありがとうございます', '字幕', 'サブスクライブ', 'チャンネル登録',
  1665→      '谢谢观看', '感谢观看', 'Sous-titres', 'Sottotitoli', 'Untertitel',
  1666→      'Thanks for watching', 'Thank you for watching', 'Please subscribe', 'Like and subscribe',
  1667→    ];
  1668→    const isHallucination = !usedFallback && WHISPER_HALLUCINATIONS.some(phrase =>
  1669→      transcribedText.toLowerCase().includes(phrase.toLowerCase())
  1670→    );
  1671→    if (isHallucination) {
  1672→      console.log(`[voice-respond] Filtered Whisper hallucination: "${transcribedText}"`);
  1673→      // Try client transcript before rejecting
  1674→      if (clientTranscript.length >= 10) {
  1675→        transcribedText = clientTranscript;
  1676→        usedFallback = true;
  1677→      } else {
  1678→        return res.status(400).json({ error: 'I didn\'t catch that. Could you please repeat your answer.' });
  1679→      }
  1680→    }
  1681→
  1682→    if (!transcribedText || transcribedText.length < 5) {
  1683→      return res.status(400).json({ error: 'Could not transcribe your response. Please try speaking louder and more clearly.' });
  1684→    }
  1685→
  1686→    // Step 2: Add candidate response to conversation
  1687→    const candidateMessage = {
  1688→      role: 'candidate',
  1689→      text: transcribedText,
  1690→      timestamp: new Date().toISOString()
  1691→    };
  1692→    conversation.push(candidateMessage);
  1693→
  1694→    // Step 3: Get base questions and AI response
  1695→    const questionsResult = await pool.query(
  1696→      'SELECT * FROM question_bank WHERE id = ANY($1)',
  1697→      [questionIds]
  1698→    );
  1699→    const questionMap = {};
  1700→    questionsResult.rows.forEach(q => { questionMap[q.id] = q; });
  1701→    const baseQuestions = questionIds.map(id => questionMap[id]).filter(Boolean);
  1702→
  1703→    const aiTurn = await conductInterviewTurn(
  1704→      conversation,
  1705→      baseQuestions,
  1706→      session.current_question_index,
  1707→      session.target_role,
  1708→      { subscriptionId: req.user.stripe_subscription_id }
  1709→    );
  1710→
  1711→    // Build interviewer message
  1712→    let interviewerText = aiTurn.reaction || '';
  1713→    if (aiTurn.question) {
  1714→      interviewerText += (interviewerText ? '\n\n' : '') + aiTurn.question;
  1715→    }
  1716→
  1717→    const interviewerMessage = {
  1718→      role: 'interviewer',
  1719→      text: interviewerText,
  1720→      action: aiTurn.action || 'transition',
  1721→      score_hint: aiTurn.score_hint || null,
  1722→      notes: aiTurn.notes || null,
  1723→      timestamp: new Date().toISOString()
  1724→    };
  1725→    conversation.push(interviewerMessage);
  1726→
  1727→    // Step 4: Update session in DB
  1728→    const isWrappingUp = aiTurn.action === 'wrap_up';
  1729→    const isTransition = aiTurn.action === 'transition';
  1730→    const newQuestionIndex = isTransition
  1731→      ? Math.min(session.current_question_index + 1, baseQuestions.length)
  1732→      : session.current_question_index;
  1733→    const newFollowUps = (aiTurn.action === 'follow_up' || aiTurn.action === 'challenge')
  1734→      ? (session.follow_ups_asked || 0) + 1
  1735→      : session.follow_ups_asked || 0;
  1736→
  1737→    if (isTransition && newQuestionIndex < baseQuestions.length) {
  1738→      const nextQ = baseQuestions[newQuestionIndex];
  1739→      if (nextQ) {
  1740→        await pool.query('UPDATE question_bank SET times_used = times_used + 1 WHERE id = $1', [nextQ.id]);
  1741→      }
  1742→    }
  1743→
  1744→    await pool.query(
  1745→      `UPDATE mock_interview_sessions
  1746→       SET conversation = $1, current_question_index = $2,
  1747→           questions_asked = $3, follow_ups_asked = $4
  1748→       WHERE id = $5`,
  1749→      [
  1750→        JSON.stringify(conversation),
  1751→        newQuestionIndex,
  1752→        isTransition ? (session.questions_asked || 0) + 1 : session.questions_asked || 0,
  1753→        newFollowUps,
  1754→        sessionId
  1755→      ]
  1756→    );
  1757→
  1758→    // BUG FIX: Pre-generate feedback in background after 3+ candidate turns (same as text respond)
  1759→    const voiceCandidateTurnCount = conversation.filter(t => t.role === 'candidate').length;
  1760→    if (voiceCandidateTurnCount >= 3 && !isWrappingUp) {
  1761→      generateSessionFeedback(conversation, session.target_role, { subscriptionId: req.user.stripe_subscription_id })
  1762→        .then(feedback => {
  1763→          pool.query(
  1764→            `UPDATE mock_interview_sessions SET cached_feedback = $1 WHERE id = $2 AND status = 'in_progress'`,
  1765→            [JSON.stringify(feedback), sessionId]
  1766→          ).catch(() => {});
  1767→          console.log(`[voice-respond] Background feedback cached for session ${sessionId} (${voiceCandidateTurnCount} turns)`);
  1768→        })
  1769→        .catch(err => console.warn('[voice-respond] Background feedback generation failed:', err.message));
  1770→    }
  1771→
  1772→    // Step 5: Generate TTS for interviewer response
  1773→    let audioBase64 = null;
  1774→    try {
  1775→      const audioBuffer = await textToSpeech(interviewerText, {
  1776→        voice: 'nova',
  1777→        subscriptionId: req.user.stripe_subscription_id
  1778→      });
  1779→      if (audioBuffer) {
  1780→        audioBase64 = audioBuffer.toString('base64');
  1781→      }
  1782→    } catch (ttsErr) {
  1783→      console.error('TTS generation failed for voice-respond:', ttsErr.message);
  1784→      // Non-fatal — response still works without audio
  1785→    }
  1786→
  1787→    res.json({
  1788→      success: true,
  1789→      transcribed_text: transcribedText,
  1790→      interviewer_message: interviewerMessage,
  1791→      interviewer_audio_base64: audioBase64,
  1792→      action: aiTurn.action,
  1793→      questions_asked: isTransition ? (session.questions_asked || 0) + 1 : session.questions_asked || 0,
  1794→      is_wrapping_up: isWrappingUp
  1795→    });
  1796→  } catch (err) {
  1797→    console.error('Voice respond error:', err);
  1798→    res.status(500).json({ error: 'Failed to process voice response. Please try again.' });
  1799→  }
  1800→});
