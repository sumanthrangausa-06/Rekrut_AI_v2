   165→// Submit practice response for coaching
   166→async function submitPracticeResponse() {
   167→  const response = document.getElementById('practice-response').value.trim();
   168→
   169→  if (!response) {
   170→    alert('Please provide a response before getting coaching.');
   171→    return;
   172→  }
   173→
   174→  if (response.length < 50) {
   175→    alert('Please provide a more detailed response (at least 50 characters).');
   176→    return;
   177→  }
   178→
   179→  showLoading('AI is analyzing your response...');
   180→
   181→  try {
   182→    const data = await apiCall('/interviews/practice/submit', {
   183→      method: 'POST',
   184→      body: JSON.stringify({
   185→        question_id: practiceState.currentQuestion.id,
   186→        question: practiceState.currentQuestion.question,
   187→        category: practiceState.currentQuestion.category,
   188→        response_text: response
   189→      })
   190→    });
   191→
   192→    if (!data || !data.success) {
   193→      throw new Error(data?.error || 'Failed to get coaching');
   194→    }
   195→
   196→    hideLoading();
   197→    displayCoachingResult(data.coaching);
   198→
   199→    // Reload stats
   200→    await loadStats();
   201→
   202→  } catch (err) {
   203→    hideLoading();
   204→    alert('Failed to analyze response: ' + err.message);
   205→  }
   206→}

   392→// Format date helper
   393→function formatDate(dateString) {
   394→  const date = new Date(dateString);
   395→  const now = new Date();
   396→  const diffMs = now - date;
   397→  const diffMins = Math.floor(diffMs / 60000);
   398→  const diffHours = Math.floor(diffMs / 3600000);
   399→  const diffDays = Math.floor(diffMs / 86400000);
   400→
   401→  if (diffMins < 1) return 'Just now';
   402→  if (diffMins < 60) return `${diffMins}m ago`;
   403→  if (diffHours < 24) return `${diffHours}h ago`;
   404→  if (diffDays < 7) return `${diffDays}d ago`;
   405→
   406→  return date.toLocaleDateString();
   407→}
