     8→async function chat(message, options = {}) {
     9→  // Handle both string messages and arrays of message objects
    10→  let messages;
    11→  if (Array.isArray(message)) {
    12→    // Filter out system messages and extract them
    13→    const systemMessages = message.filter(m => m.role === 'system');
    14→    const nonSystemMessages = message.filter(m => m.role !== 'system');
    15→
    16→    // Use the first system message content as the system prompt if not already set
    17→    if (systemMessages.length > 0 && !options.system) {
    18→      options.system = systemMessages[0].content;
    19→    }
    20→
    21→    // Ensure messages have proper format (content must be string or array)
    22→    messages = nonSystemMessages.map(m => ({
    23→      role: m.role,
    24→      content: typeof m.content === 'string' ? m.content : String(m.content)
    25→    }));
    26→  } else {
    27→    // Simple string message
    28→    messages = [{ role: 'user', content: String(message) }];
    29→  }
    30→
    31→  // Ensure we have at least one message
    32→  if (messages.length === 0) {
    33→    messages = [{ role: 'user', content: 'Hello' }];
    34→  }
    35→
    36→  const response = await anthropic.messages.create({
    37→    max_tokens: options.maxTokens || options.max_tokens || 8192,
    38→    messages: messages,
    39→    system: options.system,
    40→  }, {
    41→    headers: options.subscriptionId ? { 'X-Subscription-ID': options.subscriptionId } : {}
    42→  });
    43→  return response.content[0].text;
    44→}
