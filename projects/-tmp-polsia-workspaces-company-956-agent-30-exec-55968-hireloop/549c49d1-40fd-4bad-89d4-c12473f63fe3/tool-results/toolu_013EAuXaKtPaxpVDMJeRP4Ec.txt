   140→  async function loadProgress() {
   141→    try {
   142→      setLoading(true)
   143→      const data = await apiCall<ProgressResponse>('/onboarding/wizard/progress')
   144→      setProgress(data)
   145→
   146→      if (data.has_onboarding && data.wizard) {
   147→        const w = data.wizard
   148→        // Restore form data
   149→        setFirstName(w.legal_first_name || '')
   150→        setMiddleName(w.legal_middle_name || '')
   151→        setLastName(w.legal_last_name || '')
   152→        setDob(w.date_of_birth ? w.date_of_birth.split('T')[0] : '')
   153→        setAddress1(w.address_line1 || '')
   154→        setAddress2(w.address_line2 || '')
   155→        setCity(w.city || '')
   156→        setState(w.state || '')
   157→        setZip(w.zip_code || '')
   158→        setPhone(w.phone || '')
   159→        setEcName(w.emergency_contact_name || '')
   160→        setEcRelationship(w.emergency_contact_relationship || '')
   161→        setEcPhone(w.emergency_contact_phone || '')
   162→        setEcEmail(w.emergency_contact_email || '')
   163→        setBankName(w.bank_name || '')
   164→        setAccountType(w.account_type || 'checking')
   165→        setFilingStatus(w.w4_filing_status || 'single')
   166→        // SSN/routing/account are encrypted — don't prefill
   167→
   168→        if (w.wizard_status === 'completed') {
   169→          setCurrentStep(5)
   170→        } else {
   171→          setCurrentStep(Math.max(w.current_step || 1, 1))
   172→        }
   173→      }
   174→      if (data.documents) {
   175→        setDocuments(data.documents)
   176→      }
   177→    } catch (err: any) {
   178→      setError(err.message)
   179→    } finally {
   180→      setLoading(false)
   181→    }
   182→  }
   183→
   184→  function validateStep(step: number): boolean {
   185→    const errors: Record<string, string> = {}
   186→
   187→    if (step === 1) {
   188→      if (!firstName.trim()) errors.firstName = 'Required'
   189→      if (!lastName.trim()) errors.lastName = 'Required'
   190→      if (!dob) errors.dob = 'Required'
   191→      if (!ssn.trim() && !progress?.wizard?.ssn_encrypted) errors.ssn = 'Required'
   192→      if (ssn && !/^\d{3}-?\d{2}-?\d{4}$/.test(ssn.replace(/\s/g, ''))) errors.ssn = 'Invalid SSN format (XXX-XX-XXXX)'
   193→      if (!address1.trim()) errors.address1 = 'Required'
   194→      if (!city.trim()) errors.city = 'Required'
   195→      if (!state) errors.state = 'Required'
   196→      if (!zip.trim()) errors.zip = 'Required'
   197→      if (zip && !/^\d{5}(-\d{4})?$/.test(zip)) errors.zip = 'Invalid ZIP'
   198→      if (!phone.trim()) errors.phone = 'Required'
   199→    }
   200→
   201→    if (step === 2) {
   202→      if (!ecName.trim()) errors.ecName = 'Required'
   203→      if (!ecRelationship.trim()) errors.ecRelationship = 'Required'
   204→      if (!ecPhone.trim()) errors.ecPhone = 'Required'
   205→    }
   206→
   207→    if (step === 3) {
   208→      if (!bankName.trim()) errors.bankName = 'Required'
   209→      if (!routingNumber.trim() && !progress?.wizard?.routing_number_encrypted) errors.routingNumber = 'Required'
   210→      if (routingNumber && !/^\d{9}$/.test(routingNumber)) errors.routingNumber = 'Must be 9 digits'
   211→      if (!accountNumber.trim() && !progress?.wizard?.account_number_encrypted) errors.accountNumber = 'Required'
   212→    }
   213→
   214→    setStepErrors(errors)
   215→    return Object.keys(errors).length === 0
   216→  }
   217→
   218→  async function saveStep(step: number) {
   219→    if (!validateStep(step)) return false
   220→    if (!progress?.checklist) return false
   221→
   222→    setSaving(true)
   223→    setError('')
   224→    try {
   225→      let data: any = {}
   226→
   227→      if (step === 1) {
   228→        data = {
   229→          legal_first_name: firstName.trim(),
   230→          legal_middle_name: middleName.trim() || null,
   231→          legal_last_name: lastName.trim(),
   232→          date_of_birth: dob,
   233→          ssn: ssn.replace(/[^0-9]/g, '') || undefined,
   234→          address_line1: address1.trim(),
   235→          address_line2: address2.trim() || null,
   236→          city: city.trim(),
   237→          state,
   238→          zip_code: zip.trim(),
   239→          phone: phone.trim(),
   240→        }
   241→      } else if (step === 2) {
   242→        data = {
   243→          emergency_contact_name: ecName.trim(),
   244→          emergency_contact_relationship: ecRelationship.trim(),
   245→          emergency_contact_phone: ecPhone.trim(),
   246→          emergency_contact_email: ecEmail.trim() || null,
   247→        }
   248→      } else if (step === 3) {
   249→        data = {
   250→          bank_name: bankName.trim(),
   251→          routing_number: routingNumber || undefined,
   252→          account_number: accountNumber || undefined,
   253→          account_type: accountType,
   254→          w4_filing_status: filingStatus,
   255→        }
   256→      }
   257→
   258→      await apiCall('/onboarding/wizard/save-step', {
   259→        method: 'POST',
   260→        body: { checklist_id: progress.checklist.id, step, data },
   261→      })
   262→
   263→      return true
   264→    } catch (err: any) {
   265→      setError(err.message)
   266→      return false
   267→    } finally {
   268→      setSaving(false)
   269→    }
   270→  }
   271→
   272→  async function handleNext() {
   273→    if (currentStep <= 3) {
   274→      const success = await saveStep(currentStep)
   275→      if (success) {
   276→        if (currentStep === 3) {
   277→          // After step 3, generate documents automatically
   278→          await generateDocuments()
   279→        }
   280→        setCurrentStep(currentStep + 1)
   281→      }
   282→    }
   283→  }
   284→
   285→  async function handleBack() {
   286→    if (currentStep > 1) {
   287→      setCurrentStep(currentStep - 1)
   288→      setStepErrors({})
   289→    }
   290→  }
   291→
   292→  async function generateDocuments() {
   293→    if (!progress?.checklist) return
   294→    setGeneratingDocs(true)
   295→    try {
   296→      const res = await apiCall<{ documents: OnboardingDoc[] }>('/onboarding/wizard/generate-documents', {
   297→        method: 'POST',
   298→        body: { checklist_id: progress.checklist.id },
   299→      })
   300→      setDocuments(res.documents)
   301→    } catch (err: any) {
   302→      setError(err.message)
   303→    } finally {
   304→      setGeneratingDocs(false)
   305→    }
   306→  }
   307→
   308→  async function signAllDocuments() {
   309→    if (!progress?.checklist) return
   310→    if (!signatureText.trim() && !hasSignature) {
   311→      setStepErrors({ signature: 'Please type your name or draw your signature' })
   312→      return
   313→    }
   314→
   315→    setSigning(true)
   316→    setError('')
   317→    try {
   318→      const sigData = hasSignature && canvasRef.current
   319→        ? canvasRef.current.toDataURL()
   320→        : `text:${signatureText}`
   321→
   322→      await apiCall('/onboarding/wizard/sign-all', {
   323→        method: 'POST',
   324→        body: {
   325→          checklist_id: progress.checklist.id,
   326→          signature_data: sigData,
   327→        },
   328→      })
   329→      setCurrentStep(5)
   330→      await loadProgress()
   331→    } catch (err: any) {
   332→      setError(err.message)
   333→    } finally {
   334→      setSigning(false)
   335→    }
   336→  }

   338→  // Canvas drawing helpers
   339→  function startDrawing(e: React.MouseEvent<HTMLCanvasElement>) {
   340→    const canvas = canvasRef.current
   341→    if (!canvas) return
   342→    const ctx = canvas.getContext('2d')
   343→    if (!ctx) return
   344→    const rect = canvas.getBoundingClientRect()
   345→    ctx.beginPath()
   346→    ctx.moveTo(e.clientX - rect.left, e.clientY - rect.top)
   347→    setIsDrawing(true)
   348→  }
   349→
   350→  function draw(e: React.MouseEvent<HTMLCanvasElement>) {
   351→    if (!isDrawing) return
   352→    const canvas = canvasRef.current
   353→    if (!canvas) return
   354→    const ctx = canvas.getContext('2d')
   355→    if (!ctx) return
   356→    const rect = canvas.getBoundingClientRect()
   357→    ctx.lineWidth = 2
   358→    ctx.lineCap = 'round'
   359→    ctx.strokeStyle = '#1e293b'
   360→    ctx.lineTo(e.clientX - rect.left, e.clientY - rect.top)
   361→    ctx.stroke()
   362→    setHasSignature(true)
   363→  }
   364→
   365→  function stopDrawing() {
   366→    setIsDrawing(false)
   367→  }
   368→
   369→  function clearCanvas() {
   370→    const canvas = canvasRef.current
   371→    if (!canvas) return
   372→    const ctx = canvas.getContext('2d')
   373→    if (!ctx) return
   374→    ctx.clearRect(0, 0, canvas.width, canvas.height)
   375→    setHasSignature(false)
   376→  }

   528→                  <Icon className="h-4 w-4" />

   532→                <span className="hidden sm:inline">{step.label}</span>

   540→      {/* Error */}
   541→      {error && (

   543→          <AlertCircle className="h-4 w-4 shrink-0" />

   545→        </div>
   546→      )}
   547→
   548→      {/* Step content */}
   549→      <Card>
   550→        <CardContent className="p-6 sm:p-8">
   551→          {/* Step 1: Personal Info */}
   552→          {currentStep === 1 && (
   553→            <div className="space-y-6">
