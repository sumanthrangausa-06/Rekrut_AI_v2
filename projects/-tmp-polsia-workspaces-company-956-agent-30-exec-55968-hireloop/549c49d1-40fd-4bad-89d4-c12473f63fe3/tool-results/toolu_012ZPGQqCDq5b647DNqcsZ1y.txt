    42→// AI-generate professional offer letter
    43→router.post('/offers/:id/generate-letter', authMiddleware, async (req, res) => {
    44→  try {
    45→    // Get offer with all related data
    46→    const offerResult = await pool.query(
    47→      `SELECT o.*,
    48→        u.name as candidate_name,
    49→        u.email as candidate_email,
    50→        j.title as job_title,
    51→        j.description as job_description,
    52→        j.location as job_location,
    53→        r.name as recruiter_name
    54→      FROM offers o
    55→      JOIN users u ON o.candidate_id = u.id
    56→      LEFT JOIN jobs j ON o.job_id = j.id
    57→      LEFT JOIN users r ON o.recruiter_id = r.id
    58→      WHERE o.id = $1 AND o.company_id = $2`,
    59→      [req.params.id, req.user.company_id]
    60→    );
    61→
    62→    if (offerResult.rows.length === 0) {
    63→      return res.status(404).json({ error: 'Offer not found' });
    64→    }
    65→
    66→    const offer = offerResult.rows[0];
    67→
    68→    const prompt = `Generate a professional, formal employment offer letter. Return ONLY the HTML content (no markdown, no code fences). Use clean, semantic HTML with inline styles for professional formatting.
    69→
    70→OFFER DETAILS:
    71→- Company Name: ${offer.company_name || 'The Company'}
    72→- Candidate Name: ${offer.candidate_name}
    73→- Job Title: ${offer.job_title || offer.title}
    74→- Annual Salary: $${Number(offer.salary).toLocaleString()}
    75→- Start Date: ${offer.start_date ? new Date(offer.start_date).toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }) : 'To be determined'}
    76→- Employment Type: ${offer.employment_type || 'Full-time'}
    77→- Location: ${offer.location || offer.job_location || 'To be discussed'}
    78→- Reporting To: ${offer.reporting_to || 'Department Manager'}
    79→- Benefits: ${offer.benefits || 'Standard company benefits package'}
    80→- Recruiter/HR Contact: ${offer.recruiter_name || 'HR Department'}
    81→
    82→REQUIREMENTS:
    83→1. Professional business letter format with company letterhead area at top
    84→2. Today's date: ${new Date().toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })}
    85→3. Formal greeting to the candidate by name
    86→4. Opening paragraph expressing excitement about extending the offer
    87→5. Position details section (title, department, reporting structure, start date)
    88→6. Compensation section (salary, pay frequency)
    89→7. Benefits section (list the provided benefits professionally)
    90→8. Employment terms (at-will employment, contingencies like background check)
    91→9. Acceptance section with a placeholder line for signature and date
    92→10. Warm closing from the company
    93→11. Use inline CSS styles for clean formatting (the HTML will be rendered directly)
    94→12. Use a professional color scheme (dark navy #1e3a5f for headers, #333 for body text)
    95→13. Include proper spacing, margins, and a clean border/outline for the document
    96→14. The document should look like a real PDF offer letter when rendered
    97→
    98→FORMAT RULES:
    99→- Wrap everything in a single <div> with max-width: 800px and margin: 0 auto
   100→- Use a subtle border and padding to frame the letter
   101→- Company name at top should be bold and prominent
   102→- Include horizontal rules to separate sections
   103→- The signature block should have clear lines for name, signature, and date`;
   104→
   105→    const letterHtml = await polsiaAI.chat(prompt, {
   106→      system: 'You are an expert HR document writer. Generate professional, legally appropriate employment offer letters. Return ONLY clean HTML with inline styles. No markdown, no code blocks, no explanations.'
   107→    });
   108→
   109→    // Clean up any accidental code fences
   110→    let cleanHtml = letterHtml.trim();
   111→    if (cleanHtml.startsWith('```')) {
   112→      cleanHtml = cleanHtml.replace(/^```(?:html)?\n?/, '').replace(/\n?```$/, '');
   113→    }
   114→
   115→    // Store the generated letter
   116→    await pool.query(
   117→      `UPDATE offers SET
   118→        offer_letter_html = $1,
   119→        offer_letter_generated_at = NOW(),
   120→        updated_at = NOW()
   121→      WHERE id = $2`,
   122→      [cleanHtml, req.params.id]
   123→    );
   124→
   125→    res.json({
   126→      success: true,
   127→      offer_letter_html: cleanHtml,
   128→      generated_at: new Date().toISOString()
   129→    });
   130→  } catch (err) {
   131→    console.error('Error generating offer letter:', err);
   132→    res.status(500).json({ error: 'Failed to generate offer letter' });
   133→  }
   134→});

   261→// Accept offer (with e-signature)
   262→router.post('/offers/:id/accept', authMiddleware, async (req, res) => {
   263→  try {
   264→    const { signature_url, signature_data } = req.body;
   265→    const signerIp = req.headers['x-forwarded-for'] || req.connection?.remoteAddress || 'unknown';
   266→
   267→    const result = await pool.query(
   268→      `UPDATE offers
   269→       SET status = 'accepted', accepted_at = NOW(), signature_url = $3,
   270→           candidate_signature = $4, candidate_signed_at = NOW(), candidate_sign_ip = $5,
   271→           updated_at = NOW()
   272→       WHERE id = $1 AND candidate_id = $2
   273→       RETURNING *`,
   274→      [req.params.id, req.user.id, signature_url, signature_data || null, signerIp]
   275→    );
   276→
   277→    if (result.rows.length === 0) {
   278→      return res.status(404).json({ error: 'Offer not found' });
   279→    }
   280→
   281→    // Create default onboarding checklist
   282→    const offer = result.rows[0];
   283→    const defaultItems = [
   284→      { id: 1, task: 'Complete I-9 form', required: true },
   285→      { id: 2, task: 'Upload government-issued ID', required: true },
   286→      { id: 3, task: 'Set up direct deposit', required: true },
   287→      { id: 4, task: 'Complete tax withholding (W-4)', required: true },
   288→      { id: 5, task: 'Sign employee handbook acknowledgment', required: true },
   289→      { id: 6, task: 'Submit emergency contact information', required: true },
   290→      { id: 7, task: 'Complete IT setup (email, laptop)', required: false },
   291→      { id: 8, task: 'Schedule first day orientation', required: true }
   292→    ];
   293→
   294→    await pool.query(
   295→      `INSERT INTO onboarding_checklists
   296→       (offer_id, candidate_id, title, description, items, due_date)
   297→       VALUES ($1, $2, $3, $4, $5, $6)`,
   298→      [
   299→        offer.id,
   300→        offer.candidate_id,
   301→        'Pre-Onboarding Checklist',
   302→        'Complete these tasks before your first day',
   303→        JSON.stringify(defaultItems),
   304→        offer.start_date
   305→      ]
   306→    );
   307→
   308→    // Auto-create employee record for payroll (bridges hiring → payroll pipeline)
   309→    try {
   310→      const empNum = 'EMP-' + String(offer.candidate_id).padStart(4, '0');
   311→      const empResult = await pool.query(`
   312→        INSERT INTO employees (user_id, employer_id, company_id, employee_number, position, employment_type, start_date, status)
   313→        VALUES ($1, $2, $3, $4, $5, 'full-time', $6, 'active')
   314→        ON CONFLICT DO NOTHING
   315→        RETURNING id
   316→      `, [
   317→        offer.candidate_id,
   318→        offer.recruiter_id,
   319→        offer.company_id,
   320→        empNum,
   321→        offer.title,
   322→        offer.start_date || new Date()
   323→      ]);
   324→
   325→      if (empResult.rows.length > 0) {
   326→        const annualSalary = parseFloat(offer.salary || 50000);
   327→        await pool.query(`
   328→          INSERT INTO payroll_configs (employee_id, salary_type, salary_amount, pay_frequency, payment_method, tax_filing_status)
   329→          VALUES ($1, 'salary', $2, 'bi-weekly', 'direct_deposit', 'single')
   330→          ON CONFLICT (employee_id) DO NOTHING
   331→        `, [empResult.rows[0].id, annualSalary]);
   332→      }
   333→    } catch (empErr) {
   334→      console.error('Auto-create employee record failed (non-blocking):', empErr.message);
   335→    }
   336→
   337→    res.json(result.rows[0]);
   338→  } catch (err) {
   339→    console.error('Error accepting offer:', err);
   340→    res.status(500).json({ error: 'Failed to accept offer' });
   341→  }
   342→});

  1547→function escapeHtmlServer(text) {
  1548→  if (!text) return '';
  1549→  const map = { '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;' };
  1550→  return String(text).replace(/[&<>"']/g, m => map[m]);
  1551→}
