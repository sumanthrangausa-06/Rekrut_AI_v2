    11→// Middleware to require recruiter role
    12→function requireRecruiter(req, res, next) {
    13→  if (!req.user.company_id || !['recruiter', 'hiring_manager', 'employer', 'admin'].includes(req.user.role)) {
    14→    return res.status(403).json({ error: 'Recruiter access required' });
    15→  }
    16→  next();
    17→}

    19→// Dashboard overview
    20→router.get('/dashboard', authMiddleware, requireRecruiter, async (req, res) => {
    21→  try {
    22→    const companyId = req.user.company_id;
    23→
    24→    // Get TrustScore
    25→    const trustScore = await trustscoreService.calculateTrustScore(companyId);
    26→
    27→    // Get job stats
    28→    const jobStats = await pool.query(`
    29→      SELECT
    30→        COUNT(*) FILTER (WHERE status = 'active') as active_jobs,
    31→        COUNT(*) FILTER (WHERE status = 'paused') as paused_jobs,
    32→        COUNT(*) FILTER (WHERE status = 'closed') as closed_jobs
    33→      FROM jobs WHERE company_id = $1
    34→    `, [companyId]);
    35→
    36→    // Get application stats
    37→    const appStats = await pool.query(`
    38→      SELECT
    39→        COUNT(*) as total_applications,
    40→        COUNT(*) FILTER (WHERE status = 'applied') as new_applications,
    41→        COUNT(*) FILTER (WHERE status = 'reviewing') as reviewing,
    42→        COUNT(*) FILTER (WHERE status = 'interviewed') as interviewed,
    43→        COUNT(*) FILTER (WHERE status = 'offered') as offered,
    44→        COUNT(*) FILTER (WHERE status = 'hired') as hired
    45→      FROM job_applications WHERE company_id = $1
    46→    `, [companyId]);
    47→
    48→    // Get upcoming interviews
    49→    const upcomingInterviews = await pool.query(`
    50→      SELECT si.id, si.scheduled_at, si.interview_type, si.status,
    51→             u.name as candidate_name, u.email as candidate_email,
    52→             j.title as job_title
    53→      FROM scheduled_interviews si
    54→      JOIN users u ON si.candidate_id = u.id
    55→      JOIN jobs j ON si.job_id = j.id
    56→      WHERE si.company_id = $1
    57→        AND si.scheduled_at > NOW()
    58→        AND si.status = 'scheduled'
    59→      ORDER BY si.scheduled_at
    60→      LIMIT 5
    61→    `, [companyId]);
    62→
    63→    // Get recent applications
    64→    const recentApps = await pool.query(`
    65→      SELECT ja.id, ja.status, ja.applied_at, ja.omniscore_at_apply,
    66→             u.name as candidate_name, u.email as candidate_email,
    67→             j.title as job_title, j.id as job_id
    68→      FROM job_applications ja
    69→      JOIN users u ON ja.candidate_id = u.id
    70→      JOIN jobs j ON ja.job_id = j.id
    71→      WHERE ja.company_id = $1
    72→      ORDER BY ja.applied_at DESC
    73→      LIMIT 10
    74→    `, [companyId]);
    75→
    76→    res.json({
    77→      success: true,
    78→      trust_score: trustScore,
    79→      job_stats: jobStats.rows[0],
    80→      application_stats: appStats.rows[0],
    81→      upcoming_interviews: upcomingInterviews.rows,
    82→      recent_applications: recentApps.rows
    83→    });
    84→  } catch (err) {
    85→    console.error('Dashboard error:', err);
    86→    res.status(500).json({ error: 'Failed to load dashboard' });
    87→  }
    88→});

   123→// Create job with optional AI optimization
   124→router.post('/jobs', authMiddleware, requireRecruiter, async (req, res) => {
   125→  try {
   126→    const {
   127→      title, description, requirements, location, salary_range, job_type,
   128→      optimize = false // Flag to run AI optimization
   129→    } = req.body;
   130→
   131→    if (!title) {
   132→      return res.status(400).json({ error: 'Job title is required' });
   133→    }
   134→
   135→    let finalDescription = description;
   136→    let finalRequirements = requirements;
   137→    let optimizationResult = null;
   138→
   139→    // Run AI optimization if requested
   140→    if (optimize) {
   141→      try {
   142→        optimizationResult = await jobOptimizer.optimizeJobDescription({
   143→          title, description, requirements, location, salary_range, job_type,
   144→          company: req.user.company_name
   145→        });
   146→        finalDescription = optimizationResult.optimized_description || description;
   147→        finalRequirements = optimizationResult.optimized_requirements || requirements;
   148→      } catch (e) {
   149→        console.error('Job optimization error:', e);
   150→        // Continue without optimization
   151→      }
   152→    }
   153→
   154→    // Create job
   155→    const result = await pool.query(
   156→      `INSERT INTO jobs (user_id, company_id, title, company, description, requirements, location, salary_range, job_type)
   157→       VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
   158→       RETURNING *`,
   159→      [req.user.id, req.user.company_id, title, req.user.company_name,
   160→       finalDescription, finalRequirements, location, salary_range, job_type || 'full-time']
   161→    );
   162→
   163→    const job = result.rows[0];
   164→
   165→    // Create analytics entry
   166→    await pool.query(
   167→      'INSERT INTO job_analytics (job_id) VALUES ($1)',
   168→      [job.id]
   169→    );
   170→
   171→    // Analyze and add authenticity score
   172→    try {
   173→      const analysis = await jobOptimizer.analyzeJobPosting({
   174→        title, description: finalDescription, requirements: finalRequirements,
   175→        location, salary_range, job_type, company: req.user.company_name
   176→      });
   177→
   178→      await trustscoreService.addJobAuthenticityComponent(
   179→        req.user.company_id,
   180→        job.id,
   181→        analysis.overall_score
   182→      );
   183→
   184→      res.json({
   185→        success: true,
   186→        job,
   187→        optimization: optimizationResult,
   188→        analysis,
   189→        message: optimize ? 'Job created with AI optimization!' : 'Job created successfully'
   190→      });
   191→    } catch (e) {
   192→      // Return job without analysis
   193→      res.json({ success: true, job });
   194→    }
   195→
   196→  } catch (err) {
   197→    console.error('Create job error:', err);
   198→    res.status(500).json({ error: 'Failed to create job' });
   199→  }
   200→});

   510→        await pool.query(
   511→          'UPDATE job_analytics SET interviews_scheduled = interviews_scheduled + 1 WHERE job_id = $1',

   516→          'UPDATE job_analytics SET offers_made = offers_made + 1 WHERE job_id = $1',

   520→        // Update hiring ratio score

   524→          'UPDATE job_analytics SET offers_accepted = offers_accepted + 1 WHERE job_id = $1',
   525→          [app.job_id]

   528→    }
   529→
   530→    // Audit log: Application status change
   531→    if (status) {
   532→      await AuditLogger.log({
   533→        actionType: 'application_status_changed',
   534→        userId: req.user.id,
