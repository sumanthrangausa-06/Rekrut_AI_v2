   813→    function formatDateRange(start, end, isCurrent) {
   814→      const formatDate = (d) => d ? new Date(d).toLocaleDateString('en-US', { month: 'short', year: 'numeric' }) : '';
   815→      const startStr = formatDate(start);
   816→      const endStr = isCurrent ? 'Present' : formatDate(end);
   817→      if (!startStr && !endStr) return '';
   818→      return `${startStr} - ${endStr}`;
   819→    }
