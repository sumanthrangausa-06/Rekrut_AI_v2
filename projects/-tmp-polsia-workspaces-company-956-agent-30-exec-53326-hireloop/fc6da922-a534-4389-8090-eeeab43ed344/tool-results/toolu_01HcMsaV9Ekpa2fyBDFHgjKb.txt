   292→    // Load results
   293→    async function loadResults() {
   294→      await checkAuth();
   295→      await loadUserInfo();
   296→
   297→      const urlParams = new URLSearchParams(window.location.search);
   298→      const sessionId = urlParams.get('session');
   299→      const assessmentId = urlParams.get('id');
   300→
   301→      if (!sessionId && !assessmentId) {
   302→        alert('No results found');
   303→        window.location.href = '/skill-assessments.html';
   304→        return;
   305→      }
   306→
   307→      try {
   308→        // For now, we'll fetch session data from the results endpoint
   309→        // In a complete implementation, you'd have a dedicated session endpoint
   310→        const response = await fetch('/api/assessments/results', {
   311→          headers: { 'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}` }
   312→        });
   313→
   314→        if (!response.ok) throw new Error('Failed to load results');
   315→
   316→        const data = await response.json();
   317→        
   318→        // Get the most recent result (or specific one if assessmentId provided)
   319→        let result = assessmentId 
   320→          ? data.results.find(r => r.id === parseInt(assessmentId))
   321→          : data.results[0];
   322→
   323→        if (!result) {
   324→          throw new Error('Result not found');
   325→        }
   326→
   327→        displayResults(result);
   328→      } catch (error) {
   329→        console.error('Error loading results:', error);
   330→        document.getElementById('result-status').textContent = 'Error Loading Results';
   331→        document.getElementById('result-subtitle').textContent = 'Please try again';
   332→      }
   333→    }
   334→
   335→    // Display results
   336→    function displayResults(result) {
   337→      // Score animation
   338→      const score = result.score || 0;
   339→      const scoreCircle = document.getElementById('score-circle');
   340→      const circumference = 2 * Math.PI * 80;
   341→      const offset = circumference - (score / 100) * circumference;
   342→      
   343→      setTimeout(() => {
   344→        scoreCircle.style.strokeDashoffset = offset;
   345→      }, 100);
   346→
   347→      document.getElementById('score-value').textContent = score;
   348→
   349→      // Status
   350→      const passed = result.passed;
   351→      document.getElementById('result-status').textContent = passed ? 'Passed!' : 'Not Passed';
   352→      document.getElementById('result-subtitle').textContent = passed 
   353→        ? 'Great job! Your skill has been verified.'
   354→        : 'Keep practicing and try again.';
   355→
   356→      // Stats
   357→      const answersGiven = JSON.parse(result.answers_given || '[]');
   358→      const correctCount = answersGiven.filter(a => a.isCorrect).length;
   359→      
   360→      document.getElementById('stat-correct').textContent = `${correctCount}/10`;
   361→      document.getElementById('stat-difficulty').textContent = `${result.max_difficulty_reached || 2}/5`;
   362→      
   363→      const duration = result.duration_seconds || 0;
   364→      const minutes = Math.floor(duration / 60);
   365→      const seconds = Math.floor(duration % 60);
   366→      document.getElementById('stat-time').textContent = `${minutes}m ${seconds}s`;
   367→
   368→      // Integrity
   369→      const integrityScore = result.anti_cheat_score || 100;
   370→      const integrityEl = document.getElementById('integrity-score');
   371→      integrityEl.textContent = integrityScore;
   372→      
   373→      integrityEl.className = 'integrity-score-value';
   374→      if (integrityScore >= 90) integrityEl.classList.add('excellent');
   375→      else if (integrityScore >= 70) integrityEl.classList.add('good');
   376→      else if (integrityScore >= 50) integrityEl.classList.add('fair');
   377→      else integrityEl.classList.add('poor');
   378→
   379→      document.getElementById('tab-switches').textContent = result.tab_switches || 0;
   380→      document.getElementById('copy-paste').textContent = result.copy_paste_attempts || 0;
   381→      document.getElementById('time-anomalies').textContent = result.time_anomalies || 0;
   382→
   383→      // Badge
   384→      if (passed) {
   385→        document.getElementById('badge-section').style.display = 'block';
   386→        document.getElementById('skill-name').textContent = result.skill_name || 'Skill';
   387→      }
   388→    }
   389→
   390→    // Initialize
   391→    document.addEventListener('DOMContentLoaded', loadResults);
   392→  </script>
