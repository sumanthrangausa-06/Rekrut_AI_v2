   164→    // Check OAuth status and setup buttons
   165→    async function setupOAuth() {
   166→      try {
   167→        const res = await fetch('/api/auth/oauth/status');
   168→        const status = await res.json();
   169→
   170→        const googleBtn = document.getElementById('google-signup');
   171→        const linkedinBtn = document.getElementById('linkedin-signup');
   172→
   173→        if (status.google?.configured) {
   174→          googleBtn.disabled = false;
   175→          googleBtn.addEventListener('click', async () => {
   176→            try {
   177→              const res = await fetch('/api/auth/google/url');
   178→              const data = await res.json();
   179→              if (data.url) {
   180→                window.location.href = data.url;
   181→              }
   182→            } catch (e) {
   183→              console.error('Google OAuth error:', e);
   184→            }
   185→          });
   186→        }
   187→
   188→        if (status.linkedin?.configured) {
   189→          linkedinBtn.disabled = false;
   190→          linkedinBtn.addEventListener('click', async () => {
   191→            try {
   192→              const res = await fetch('/api/auth/linkedin/url');
   193→              const data = await res.json();
   194→              if (data.url) {
   195→                window.location.href = data.url;
   196→              }
   197→            } catch (e) {
   198→              console.error('LinkedIn OAuth error:', e);
   199→            }
   200→          });
   201→        }
   202→
   203→        // Show note if neither is configured
   204→        if (!status.google?.configured && !status.linkedin?.configured) {
   205→          document.getElementById('oauth-note').style.display = 'block';
   206→        }
   207→      } catch (e) {
   208→        console.error('Failed to check OAuth status:', e);
   209→      }
   210→    }
   211→
   212→    setupOAuth();
   213→  </script>
