   237→  <script>
   238→    let sessionId = null;
   239→    let currentQuestion = null;
   240→    let selectedAnswer = null;
   241→    let questionStartTime = Date.now();
   242→    let timerInterval = null;
   243→    let timeRemaining = 120; // seconds
   244→
   245→    // Anti-cheat tracking
   246→    let isTabActive = true;
   247→    let tabSwitchCount = 0;
   248→    let copyPasteCount = 0;
   249→
   250→    // Initialize assessment
   251→    async function initAssessment() {
   252→      await checkAuth();
   253→
   254→      const skillId = localStorage.getItem('assessment_skill_id');
   255→      if (!skillId) {
   256→        alert('No assessment selected');
   257→        window.location.href = '/skill-assessments.html';
   258→        return;
   259→      }
   260→
   261→      try {
   262→        const response = await fetch('/api/assessments/start', {
   263→          method: 'POST',
   264→          headers: {
   265→            'Content-Type': 'application/json',
   266→            'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}`
   267→          },
   268→          body: JSON.stringify({ skillId: parseInt(skillId) })
   269→        });
   270→
   271→        if (!response.ok) throw new Error('Failed to start assessment');
   272→
   273→        const data = await response.json();
   274→        sessionId = data.sessionId;
   275→        currentQuestion = data.question;
   276→        
   277→        displayQuestion();
   278→        startTimer(currentQuestion.timeLimit);
   279→      } catch (error) {
   280→        console.error('Error starting assessment:', error);
   281→        alert('Failed to start assessment');
   282→        window.location.href = '/skill-assessments.html';
   283→      }
   284→    }
   285→
   286→    // Display current question
   287→    function displayQuestion() {
   288→      document.getElementById('question-num').textContent = currentQuestion.questionNumber;
   289→      document.getElementById('total-questions').textContent = currentQuestion.totalQuestions;
   290→      document.getElementById('question-text').textContent = currentQuestion.text;
   291→      
   292→      // Update progress bar
   293→      const progress = (currentQuestion.questionNumber / currentQuestion.totalQuestions) * 100;
   294→      document.getElementById('progress-fill').style.width = `${progress}%`;
   295→
   296→      // Render answer area based on question type
   297→      const answerArea = document.getElementById('answer-area');
   298→      
   299→      if (currentQuestion.type === 'multiple_choice') {
   300→        answerArea.innerHTML = `
   301→          <div class="options-grid">
   302→            ${currentQuestion.options.map((option, index) => `
   303→              <button class="option-button" data-value="${option}" onclick="selectOption(this)">
   304→                ${option}
   305→              </button>
   306→            `).join('')}
   307→          </div>
   308→        `;
   309→      } else if (currentQuestion.type === 'short_answer') {
   310→        answerArea.innerHTML = `
   311→          <textarea class="textarea-answer" id="short-answer" placeholder="Type your answer here..."
   312→                    onpaste="logCopyPaste(event)"></textarea>
   313→        `;
   314→      }
   315→
   316→      // Reset state
   317→      selectedAnswer = null;
   318→      questionStartTime = Date.now();
   319→    }
   320→
   321→    // Select multiple choice option
   322→    function selectOption(button) {
   323→      document.querySelectorAll('.option-button').forEach(btn => {
   324→        btn.classList.remove('selected');
   325→      });
   326→      button.classList.add('selected');
   327→      selectedAnswer = button.dataset.value;
   328→    }
   329→
   330→    // Submit answer
   331→    async function submitAnswer() {
   332→      // Get answer
   333→      if (currentQuestion.type === 'multiple_choice') {
   334→        if (!selectedAnswer) {
   335→          alert('Please select an answer');
   336→          return;
   337→        }
   338→      } else {
   339→        const textarea = document.getElementById('short-answer');
   340→        selectedAnswer = textarea.value.trim();
   341→        if (!selectedAnswer) {
   342→          alert('Please provide an answer');
   343→          return;
   344→        }
   345→      }
   346→
   347→      // Calculate time taken
   348→      const timeTaken = Math.floor((Date.now() - questionStartTime) / 1000);
   349→
   350→      // Stop timer
   351→      clearInterval(timerInterval);
   352→
   353→      // Submit to API
   354→      try {
   355→        const response = await fetch('/api/assessments/answer', {
   356→          method: 'POST',
   357→          headers: {
   358→            'Content-Type': 'application/json',
   359→            'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}`
   360→          },
   361→          body: JSON.stringify({
   362→            sessionId,
   363→            questionId: currentQuestion.id,
   364→            answer: selectedAnswer,
   365→            timeTaken
   366→          })
   367→        });
   368→
   369→        if (!response.ok) throw new Error('Failed to submit answer');
   370→
   371→        const data = await response.json();
   372→
   373→        // Show feedback
   374→        showFeedback(data);
   375→
   376→        // Store next question if not completed
   377→        if (!data.completed) {
   378→          currentQuestion = data.nextQuestion;
   379→        } else {
   380→          // Assessment completed
   381→          setTimeout(() => {
   382→            window.location.href = `/assessment-results.html?session=${sessionId}`;
   383→          }, 3000);
   384→        }
   385→      } catch (error) {
   386→        console.error('Error submitting answer:', error);
   387→        alert('Failed to submit answer');
   388→      }
   389→    }
   390→
   391→    // Show feedback modal
   392→    function showFeedback(data) {
   393→      const modal = document.getElementById('feedback-modal');
   394→      const icon = document.getElementById('feedback-icon');
   395→      const title = document.getElementById('feedback-title');
   396→      const text = document.getElementById('feedback-text');
   397→
   398→      const isCorrect = data.feedback === 'Correct!';
   399→      icon.textContent = isCorrect ? '✅' : '❌';
   400→      title.textContent = data.feedback;
   401→      
   402→      let feedbackHTML = data.explanation ? `<p>${data.explanation}</p>` : '';
   403→      if (data.aiFeedback) {
   404→        feedbackHTML += `<p style="margin-top: 12px; padding-top: 12px; border-top: 1px solid var(--border-color);">${data.aiFeedback}</p>`;
   405→      }
   406→      text.innerHTML = feedbackHTML;
   407→
   408→      modal.classList.add('show');
   409→    }
   410→
   411→    // Next question
   412→    function nextQuestion() {
   413→      document.getElementById('feedback-modal').classList.remove('show');
   414→      displayQuestion();
   415→      timeRemaining = currentQuestion.timeLimit;
   416→      startTimer(currentQuestion.timeLimit);
   417→    }
   418→
   419→    // Timer
   420→    function startTimer(seconds) {
   421→      timeRemaining = seconds;
   422→      updateTimerDisplay();
   423→
   424→      timerInterval = setInterval(() => {
   425→        timeRemaining--;
   426→        updateTimerDisplay();
   427→
   428→        if (timeRemaining <= 0) {
   429→          clearInterval(timerInterval);
   430→          // Auto-submit empty answer
   431→          submitAnswer();
   432→        }
   433→      }, 1000);
   434→    }
   435→
   436→    function updateTimerDisplay() {
   437→      const minutes = Math.floor(timeRemaining / 60);
   438→      const seconds = timeRemaining % 60;
   439→      const display = `${minutes}:${seconds.toString().padStart(2, '0')}`;
   440→      
   441→      const timerEl = document.getElementById('timer');
   442→      timerEl.textContent = display;
   443→      
   444→      timerEl.className = 'timer';
   445→      if (timeRemaining <= 10) {
   446→        timerEl.classList.add('danger');
   447→      } else if (timeRemaining <= 30) {
   448→        timerEl.classList.add('warning');
   449→      }
   450→    }

   452→    // Anti-cheat: Tab visibility
   453→    document.addEventListener('visibilitychange', () => {
   454→      if (document.hidden) {
   455→        isTabActive = false;
   456→        tabSwitchCount++;
   457→        
   458→        // Log event
   459→        logEvent('tab_switch', { count: tabSwitchCount });
   460→        
   461→        // Show warning
   462→        const warning = document.getElementById('proctoring-warning');
   463→        warning.classList.add('show');
   464→        setTimeout(() => warning.classList.remove('show'), 3000);
   465→      } else {
   466→        isTabActive = true;
   467→      }
   468→    });
   469→
   470→    // Anti-cheat: Copy/paste detection
   471→    function logCopyPaste(event) {
   472→      copyPasteCount++;
   473→      logEvent('copy_paste', { count: copyPasteCount });
   474→    }
   475→
   476→    // Log anti-cheat event
   477→    async function logEvent(eventType, eventData) {
   478→      try {
   479→        await fetch('/api/assessments/event', {
   480→          method: 'POST',
   481→          headers: {
   482→            'Content-Type': 'application/json',
   483→            'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}`
   484→          },
   485→          body: JSON.stringify({
   486→            sessionId,
   487→            eventType,
   488→            eventData
   489→          })
   490→        });
   491→      } catch (error) {
   492→        console.error('Failed to log event:', error);
   493→      }
   494→    }
