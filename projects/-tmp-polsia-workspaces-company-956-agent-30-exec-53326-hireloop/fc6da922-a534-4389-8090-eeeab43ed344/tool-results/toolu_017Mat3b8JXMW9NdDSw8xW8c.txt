   143→    const TOKEN_KEY = 'rekrutai_token';
   144→    let offersData = [];
   145→    let candidatesData = [];
   146→    let jobsData = [];
   147→
   148→    function getToken() {
   149→      return localStorage.getItem(TOKEN_KEY);
   150→    }
   151→
   152→    async function checkAuth() {
   153→      const token = getToken();
   154→      if (!token) {
   155→        window.location.href = '/login.html';
   156→        return null;
   157→      }
   158→
   159→      try {
   160→        const response = await fetch('/api/auth/me', {
   161→          headers: { 'Authorization': `Bearer ${token}` }
   162→        });
   163→
   164→        if (!response.ok) {
   165→          window.location.href = '/login.html';
   166→          return null;
   167→        }
   168→
   169→        return await response.json();
   170→      } catch (err) {
   171→        console.error('Auth check failed:', err);
   172→        window.location.href = '/login.html';
   173→        return null;
   174→      }
   175→    }
   176→
   177→    async function loadData() {
   178→      const token = getToken();
   179→      if (!token) {
   180→        window.location.href = '/login.html';
   181→        return;
   182→      }
   183→
   184→      const headers = { 'Authorization': `Bearer ${token}` };
   185→
   186→      try {
   187→        const [offersRes, candidatesRes, jobsRes] = await Promise.all([
   188→          fetch('/api/onboarding/offers', { headers }),
   189→          fetch('/api/recruiter/candidates', { headers }),
   190→          fetch('/api/jobs', { headers })
   191→        ]);
   192→
   193→        if (!offersRes.ok || !candidatesRes.ok || !jobsRes.ok) {
   194→          console.error('API error - offers:', offersRes.status, 'candidates:', candidatesRes.status, 'jobs:', jobsRes.status);
   195→          if (offersRes.status === 401 || candidatesRes.status === 401) {
   196→            window.location.href = '/login.html';
   197→            return;
   198→          }
   199→        }
   200→
   201→        const offersJson = await offersRes.json();
   202→        const candidatesJson = await candidatesRes.json();
   203→        const jobsJson = await jobsRes.json();
   204→
   205→        // Handle both array and object responses
   206→        offersData = Array.isArray(offersJson) ? offersJson : (offersJson.offers || []);
   207→        candidatesData = Array.isArray(candidatesJson) ? candidatesJson : (candidatesJson.candidates || []);
   208→        jobsData = Array.isArray(jobsJson) ? jobsJson : (jobsJson.jobs || []);
   209→
   210→        renderOffers();
   211→        updateStats();
   212→        populateDropdowns();
   213→      } catch (err) {
   214→        console.error('Error loading data:', err);
   215→      }
   216→    }
   217→
   218→    function renderOffers() {
   219→      const tbody = document.getElementById('offers-tbody');
   220→
   221→      if (offersData.length === 0) {
   222→        tbody.innerHTML = `
   223→          <tr>
   224→            <td colspan="6" class="px-6 py-12 text-center text-gray-400">
   225→              <i class="fas fa-inbox text-4xl mb-4 block"></i>
   226→              <p>No offers created yet</p>
   227→            </td>
   228→          </tr>
   229→        `;
   230→        return;
   231→      }
   232→
   233→      tbody.innerHTML = offersData.map(offer => `
   234→        <tr class="hover:bg-white/5 transition">
   235→          <td class="px-6 py-4">
   236→            <div class="font-medium text-white">${offer.candidate_name || 'Unknown'}</div>
   237→            <div class="text-sm text-gray-400">${offer.candidate_email || ''}</div>
   238→          </td>
   239→          <td class="px-6 py-4 text-gray-300">${offer.title || offer.job_title || 'N/A'}</td>
   240→          <td class="px-6 py-4 text-gray-300">$${parseInt(offer.salary).toLocaleString()}</td>
   241→          <td class="px-6 py-4 text-gray-300">${offer.start_date ? new Date(offer.start_date).toLocaleDateString() : 'TBD'}</td>
   242→          <td class="px-6 py-4">
   243→            ${getStatusBadge(offer.status)}
   244→          </td>
   245→          <td class="px-6 py-4">
   246→            ${getActionButtons(offer)}
   247→          </td>
   248→        </tr>
   249→      `).join('');
   250→    }
   251→
   252→    function getStatusBadge(status) {
   253→      const badges = {
   254→        draft: '<span class="px-3 py-1 bg-gray-600 text-white text-xs rounded-full">Draft</span>',
   255→        sent: '<span class="px-3 py-1 bg-yellow-600 text-white text-xs rounded-full">Sent</span>',
   256→        accepted: '<span class="px-3 py-1 bg-green-600 text-white text-xs rounded-full">Accepted</span>',
   257→        declined: '<span class="px-3 py-1 bg-red-600 text-white text-xs rounded-full">Declined</span>'
   258→      };
   259→      return badges[status] || badges.draft;
   260→    }
   261→
   262→    function getActionButtons(offer) {
   263→      if (offer.status === 'draft') {
   264→        return `<button onclick="sendOffer(${offer.id})" class="text-purple-400 hover:text-purple-300"><i class="fas fa-paper-plane"></i> Send</button>`;
   265→      }
   266→      if (offer.status === 'sent') {
   267→        return `<span class="text-gray-400">Awaiting response</span>`;
   268→      }
   269→      return `<span class="text-gray-500">-</span>`;
   270→    }
   271→
   272→    function updateStats() {
   273→      document.getElementById('stat-total').textContent = offersData.length;
   274→      document.getElementById('stat-pending').textContent = offersData.filter(o => o.status === 'sent').length;
   275→      document.getElementById('stat-accepted').textContent = offersData.filter(o => o.status === 'accepted').length;
   276→      document.getElementById('stat-declined').textContent = offersData.filter(o => o.status === 'declined').length;
   277→    }
   278→
   279→    function populateDropdowns() {
   280→      const candidateSelect = document.getElementById('candidate_id');
   281→      const jobSelect = document.getElementById('job_id');
   282→
   283→      candidateSelect.innerHTML = '<option value="">Select candidate...</option>' +
   284→        candidatesData.map(c => `<option value="${c.id}">${c.name} - ${c.email}</option>`).join('');
   285→
   286→      jobSelect.innerHTML = '<option value="">Select job...</option>' +
   287→        jobsData.map(j => `<option value="${j.id}">${j.title}</option>`).join('');
   288→    }

   290→    function showCreateOfferModal() {

   292→      document.getElementById('createOfferModal').classList.add('flex');
   293→    }

   295→    function closeCreateOfferModal() {
   296→      document.getElementById('createOfferModal').classList.add('hidden');
   297→      document.getElementById('createOfferModal').classList.remove('flex');

   306→    async function createOffer(sendImmediately) {
   307→      const formData = {
   308→        candidate_id: parseInt(document.getElementById('candidate_id').value),
   309→        job_id: parseInt(document.getElementById('job_id').value),
   310→        title: document.getElementById('title').value,
   311→        salary: parseFloat(document.getElementById('salary').value),
   312→        start_date: document.getElementById('start_date').value,
   313→        benefits: document.getElementById('benefits').value
   314→      };
   315→
   316→      try {
   317→        const res = await fetch('/api/onboarding/offers', {
   318→          method: 'POST',
   319→          headers: {
   320→            'Content-Type': 'application/json',
   321→            'Authorization': `Bearer ${getToken()}`
   322→          },
   323→          body: JSON.stringify(formData)
   324→        });
   325→
   326→        if (!res.ok) throw new Error('Failed to create offer');
   327→
   328→        const offer = await res.json();
   329→
   330→        if (sendImmediately) {
   331→          await sendOffer(offer.id);
   332→        }
   333→
   334→        closeCreateOfferModal();
   335→        loadData();
   336→      } catch (err) {
   337→        console.error('Error creating offer:', err);
   338→        alert('Failed to create offer. Please try again.');
   339→      }
   340→    }
   341→
   342→    async function createAndSendOffer() {
   343→      await createOffer(true);
   344→    }
   345→
   346→    async function sendOffer(offerId) {
   347→      if (!confirm('Send this offer to the candidate?')) return;
   348→
   349→      try {
   350→        const res = await fetch(`/api/onboarding/offers/${offerId}/send`, {
   351→          method: 'POST',
   352→          headers: {
   353→            'Authorization': `Bearer ${getToken()}`
   354→          }
   355→        });
   356→
   357→        if (!res.ok) throw new Error('Failed to send offer');
   358→
   359→        alert('Offer sent successfully!');
   360→        loadData();
   361→      } catch (err) {
   362→        console.error('Error sending offer:', err);
   363→        alert('Failed to send offer. Please try again.');
   364→      }
   365→    }

   369→      const authData = await checkAuth();
