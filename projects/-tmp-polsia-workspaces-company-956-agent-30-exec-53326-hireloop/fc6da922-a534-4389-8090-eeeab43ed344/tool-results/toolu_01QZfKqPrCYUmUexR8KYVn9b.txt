    22→
    23→    if (!email || !password) {
    24→      return res.status(400).json({ error: 'Email and password are required' });
    25→    }
    26→
    27→    // Check if user exists
    28→    const existing = await pool.query('SELECT id FROM users WHERE email = $1', [email]);
    29→    if (existing.rows.length > 0) {
    30→      return res.status(400).json({ error: 'Email already registered' });
    31→    }
    32→
    33→    // Hash password
    34→    const password_hash = await bcrypt.hash(password, 10);
    35→
    36→    // Create user
    37→    const result = await pool.query(
    38→      `INSERT INTO users (email, password_hash, name, role, company_name)
    39→       VALUES ($1, $2, $3, $4, $5)
    40→       RETURNING id, email, name, role, company_name, created_at`,
    41→      [email, password_hash, name, role, company_name]
    42→    );
    43→
    44→    const user = result.rows[0];
    45→    const accessToken = generateToken(user);
    46→    const { token: refreshToken } = await generateRefreshToken(user.id);
    47→
    48→    // Track signup completion
    49→    try {
    50→      await pool.query(
    51→        'INSERT INTO events (event_type, user_id, metadata) VALUES ($1, $2, $3)',
    52→        [`signup_complete_${role}`, user.id, JSON.stringify({ email: user.email, role: role })]
    53→      );
    54→    } catch (e) {
    55→      console.error('Failed to log signup event:', e);
    56→    }
    57→
    58→    res.json({
    59→      success: true,
    60→      user: { id: user.id, email: user.email, name: user.name, role: user.role },
    61→      token: accessToken,
    62→      accessToken,
    63→      refreshToken
    64→    });
    65→  } catch (err) {
    66→    console.error('Registration error:', err);
    67→
    68→    if (err.code === '23505') {
    69→      return res.status(400).json({ error: 'This email is already registered' });
    70→    }
    71→    if (err.code === '42703') {
    72→      return res.status(500).json({ error: 'Database schema error. Please try again in a few minutes.' });
    73→    }
    74→
    75→    res.status(500).json({ error: 'Registration failed. Please try again.' });
    76→  }

   232→// Google OAuth callback
   233→router.get('/google/callback', async (req, res) => {
   234→  try {
   235→    const { code, state, error } = req.query;
   236→
   237→    if (error) {
   238→      return res.redirect(`/login.html?error=${encodeURIComponent(error)}`);
   239→    }
   240→
   241→    if (!code) {
   242→      return res.redirect('/login.html?error=No authorization code received');
   243→    }
   244→
   245→    // Exchange code for tokens
   246→    const tokenResponse = await fetch('https://oauth2.googleapis.com/token', {
   247→      method: 'POST',
   248→      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
   249→      body: new URLSearchParams({
   250→        client_id: process.env.GOOGLE_CLIENT_ID,
   251→        client_secret: process.env.GOOGLE_CLIENT_SECRET,
   252→        code,
   253→        grant_type: 'authorization_code',
   254→        redirect_uri: process.env.GOOGLE_REDIRECT_URI || `${req.protocol}://${req.get('host')}/api/auth/google/callback`
   255→      })
   256→    });
   257→
   258→    const tokens = await tokenResponse.json();
   259→
   260→    if (tokens.error) {
   261→      console.error('Google token error:', tokens);
   262→      return res.redirect('/login.html?error=Failed to authenticate with Google');
   263→    }
   264→
   265→    // Get user info
   266→    const userInfoResponse = await fetch('https://www.googleapis.com/oauth2/v2/userinfo', {
   267→      headers: { Authorization: `Bearer ${tokens.access_token}` }
   268→    });
   269→
   270→    const googleUser = await userInfoResponse.json();
   271→
   272→    if (!googleUser.email) {
   273→      return res.redirect('/login.html?error=Could not retrieve email from Google');
   274→    }
   275→
   276→    // Find or create user
   277→    let user;
   278→    const existingUser = await pool.query(
   279→      'SELECT * FROM users WHERE google_id = $1 OR email = $2',
   280→      [googleUser.id, googleUser.email]
   281→    );
   282→
   283→    if (existingUser.rows.length > 0) {
   284→      user = existingUser.rows[0];
   285→      // Update Google ID if not set
   286→      if (!user.google_id) {
   287→        await pool.query(
   288→          'UPDATE users SET google_id = $1, avatar_url = COALESCE(avatar_url, $2) WHERE id = $3',
   289→          [googleUser.id, googleUser.picture, user.id]
   290→        );
   291→      }
   292→    } else {
   293→      // Create new user
   294→      const result = await pool.query(
   295→        `INSERT INTO users (email, name, google_id, avatar_url, oauth_provider, role)
   296→         VALUES ($1, $2, $3, $4, 'google', 'candidate')
   297→         RETURNING *`,
   298→        [googleUser.email, googleUser.name, googleUser.id, googleUser.picture]
   299→      );
   300→      user = result.rows[0];
   301→    }
   302→
   303→    // Store OAuth connection
   304→    await pool.query(`
   305→      INSERT INTO oauth_connections (user_id, provider, provider_user_id, access_token, refresh_token, profile_data)
   306→      VALUES ($1, 'google', $2, $3, $4, $5)
   307→      ON CONFLICT (provider, provider_user_id) DO UPDATE SET
   308→        access_token = $3, refresh_token = $4, profile_data = $5, updated_at = NOW()
   309→    `, [user.id, googleUser.id, tokens.access_token, tokens.refresh_token, JSON.stringify(googleUser)]);
   310→
   311→    // Generate tokens
   312→    const accessToken = generateToken(user);
   313→    const { token: refreshToken } = await generateRefreshToken(user.id);
   314→
   315→    // Redirect with tokens
   316→    const redirectUrl = user.role === 'recruiter' ? '/recruiter-dashboard.html' : '/candidate-dashboard.html';
   317→    res.redirect(`${redirectUrl}?token=${accessToken}&refresh=${refreshToken}`);
   318→  } catch (err) {
   319→    console.error('Google OAuth error:', err);
   320→    res.redirect('/login.html?error=Authentication failed');
   321→  }
