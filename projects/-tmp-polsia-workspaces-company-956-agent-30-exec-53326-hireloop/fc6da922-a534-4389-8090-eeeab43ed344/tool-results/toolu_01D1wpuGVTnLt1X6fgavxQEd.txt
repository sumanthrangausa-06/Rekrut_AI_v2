   129→  <script>
   130→    document.getElementById('recruiter-register-form').addEventListener('submit', async (e) => {
   131→      e.preventDefault();
   132→      const form = e.target;
   133→      const submitBtn = form.querySelector('button[type="submit"]');
   134→      const errorEl = document.getElementById('register-error');
   135→
   136→      submitBtn.disabled = true;
   137→      submitBtn.textContent = 'Creating company account...';
   138→      errorEl.style.display = 'none';
   139→
   140→      try {
   141→        const response = await fetch('/api/company/register', {
   142→          method: 'POST',
   143→          headers: { 'Content-Type': 'application/json' },
   144→          body: JSON.stringify({
   145→            email: form.email.value,
   146→            password: form.password.value,
   147→            name: form.name.value,
   148→            company_name: form.company_name.value,
   149→            industry: form.industry.value,
   150→            company_size: form.company_size.value,
   151→            website: form.website.value,
   152→            headquarters: form.headquarters.value
   153→          })
   154→        });
   155→
   156→        const data = await response.json();
   157→
   158→        if (data.success) {
   159→          localStorage.setItem('rekrutai_token', data.token);
   160→          localStorage.setItem('rekrutai_company_id', data.company.id);
   161→          window.location.href = '/recruiter-dashboard.html';
   162→        } else {
   163→          errorEl.textContent = data.error || 'Registration failed';
   164→          errorEl.style.display = 'block';
   165→        }
   166→      } catch (err) {
   167→        errorEl.textContent = 'Network error. Please try again.';
   168→        errorEl.style.display = 'block';
   169→      } finally {
   170→        submitBtn.disabled = false;
   171→        submitBtn.textContent = 'Create Company Account';
   172→      }
   173→    });
   174→  </script>
