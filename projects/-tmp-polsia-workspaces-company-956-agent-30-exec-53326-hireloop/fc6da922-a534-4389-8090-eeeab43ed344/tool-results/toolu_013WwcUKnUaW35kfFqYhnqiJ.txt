   160→    // Check OAuth status and setup buttons
   161→    async function setupOAuth() {
   162→      try {
   163→        const res = await fetch('/api/auth/oauth/status');
   164→        const status = await res.json();
   165→
   166→        const googleBtn = document.getElementById('google-login');
   167→        const linkedinBtn = document.getElementById('linkedin-login');
   168→
   169→        if (status.google?.configured) {
   170→          googleBtn.disabled = false;
   171→          googleBtn.addEventListener('click', async () => {
   172→            try {
   173→              const res = await fetch('/api/auth/google/url');
   174→              const data = await res.json();
   175→              if (data.url) {
   176→                window.location.href = data.url;
   177→              }
   178→            } catch (e) {
   179→              console.error('Google OAuth error:', e);
   180→            }
   181→          });
   182→        }
   183→
   184→        if (status.linkedin?.configured) {
   185→          linkedinBtn.disabled = false;
   186→          linkedinBtn.addEventListener('click', async () => {
   187→            try {
   188→              const res = await fetch('/api/auth/linkedin/url');
   189→              const data = await res.json();
   190→              if (data.url) {
   191→                window.location.href = data.url;
   192→              }
   193→            } catch (e) {
   194→              console.error('LinkedIn OAuth error:', e);
   195→            }
   196→          });
   197→        }
   198→
   199→        // Show note if neither is configured
   200→        if (!status.google?.configured && !status.linkedin?.configured) {
   201→          document.getElementById('oauth-note').style.display = 'block';
   202→        }
   203→      } catch (e) {
   204→        console.error('Failed to check OAuth status:', e);
   205→      }
   206→    }
   207→
   208→    setupOAuth();
