   344→    async function loadDashboard() {
   345→      try {
   346→        // Load dashboard stats

   389→    function renderOffers(offers) {
   390→      const container = document.getElementById('offers-list');
   391→
   392→      container.innerHTML = offers.map(offer => {
   393→        const statusColors = {
   394→          draft: { bg: 'rgba(107,114,128,0.15)', color: '#6b7280', label: 'Draft' },

   403→            <div class="offer-info">

   418→    }

   432→    function renderScheduledInterviews(interviews) {
   433→      const container = document.getElementById('scheduled-interviews-list');
   434→
   435→      container.innerHTML = interviews.map(interview => {
   436→        const date = new Date(interview.scheduled_at);
   437→        const now = new Date();
   438→        const isUpcoming = date > now;
   439→        const isSoon = date - now < 30 * 60 * 1000; // Within 30 minutes
   440→
   441→        return `
   442→          <div class="scheduled-interview-card ${isSoon ? 'soon' : ''}">
   443→            <div class="interview-time-badge">
   444→              <span class="interview-date">${date.toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' })}</span>
   445→              <span class="interview-time">${date.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })}</span>
   446→            </div>
   447→            <div class="interview-info">
   448→              <h3>${interview.job_title || 'Interview'}</h3>
   449→              <p>${interview.company_name || 'Company'}</p>
   450→              <span class="interview-type-badge ${interview.interview_type || 'video'}">${interview.interview_type || 'Video Interview'}</span>
   451→            </div>
   452→            <div class="interview-action">
   453→              ${isUpcoming ? `
   454→                ${interview.meeting_link ?
   455→                  `<a href="${interview.meeting_link}" target="_blank" class="btn btn-primary ${isSoon ? 'pulse' : ''}">
   456→                    ${interview.interview_type === 'video' ? '🎥 Join Video Call' : '📞 Join'}
   457→                  </a>` :
   458→                  `<span class="interview-pending">Link will be available soon</span>`
   459→                }
   460→              ` : `<span class="interview-status">Completed</span>`}
   461→            </div>
   462→          </div>
   463→        `;
   464→      }).join('');
   465→    }

   489→    function renderStats(stats) {
   490→      // Profile completeness
   491→      const completeness = stats.profile_completeness || 0;
   492→      document.getElementById('completeness-value').textContent = completeness + '%';
   493→      document.getElementById('completeness-path').setAttribute('stroke-dasharray', `${completeness}, 100`);

   495→      if (completeness >= 80) {
   496→        document.getElementById('completeness-message').textContent = 'Your profile is looking great!';
   497→      } else if (completeness >= 50) {
   498→        document.getElementById('completeness-message').textContent = 'Almost there! Add more details.';
   499→      }
   500→
   501→      // OmniScore
   502→      if (stats.omniscore) {
   503→        document.getElementById('omniscore-value').textContent = stats.omniscore.total_score;
   504→        document.getElementById('omniscore-tier').textContent = stats.omniscore.tier_label;
   505→        document.getElementById('omniscore-tier').className = 'omniscore-badge ' + (stats.omniscore.tier || '').replace('_', '-');
   506→      }
   507→

   634→    function formatDate(dateStr) {
   635→      const date = new Date(dateStr);
   636→      const now = new Date();
   637→      const diff = now - date;
   638→      const days = Math.floor(diff / (1000 * 60 * 60 * 24));
   639→
   640→      if (days === 0) return 'Today';
   641→      if (days === 1) return 'Yesterday';
   642→      if (days < 7) return `${days} days ago`;
   643→      return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
   644→    }
