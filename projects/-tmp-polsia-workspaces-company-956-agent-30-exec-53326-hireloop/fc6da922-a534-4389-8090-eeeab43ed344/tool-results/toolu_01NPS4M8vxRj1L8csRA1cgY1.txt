   453→    const BACKEND_URL = '/api';
   454→    let currentCandidateId = null;

   456→    async function loadOnboardingDocuments() {
   457→      try {
   458→        const response = await fetch(`${BACKEND_URL}/onboarding/recruiter/summary`, {
   459→          headers: { 'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}` }
   460→        });
   461→
   462→        if (!response.ok) throw new Error('Failed to load documents');
   463→
   464→        const data = await response.json();
   465→        displayCandidates(data);
   466→      } catch (error) {
   467→        console.error('Error:', error);
   468→        document.getElementById('empty-state').style.display = 'block';
   469→        document.getElementById('candidates-container').innerHTML = '';
   470→      }
   471→    }
   472→
   473→    function displayCandidates(candidates) {
   474→      const container = document.getElementById('candidates-container');
   475→
   476→      if (!candidates || candidates.length === 0) {
   477→        document.getElementById('empty-state').style.display = 'block';
   478→        container.innerHTML = '';
   479→        return;
   480→      }
   481→
   482→      document.getElementById('empty-state').style.display = 'none';
   483→
   484→      container.innerHTML = candidates.map(candidate => {
   485→        const total = parseInt(candidate.total_documents) || 0;
   486→        const signed = parseInt(candidate.signed_documents) || 0;
   487→        const pct = total > 0 ? Math.round((signed / total) * 100) : 0;
   488→        const barColor = pct === 100 ? '#10b981' : pct > 0 ? '#f59e0b' : '#e5e7eb';
   489→
   490→        return `
   491→        <div class="candidate-card" onclick="viewCandidateDetails(${candidate.candidate_id})">
   492→          <div class="candidate-header">
   493→            <div class="candidate-info">
   494→              <h3>${escapeHtml(candidate.candidate_name)}</h3>
   495→              <div class="candidate-email">${escapeHtml(candidate.candidate_email)}</div>
   496→              ${candidate.job_title ? `<div class="candidate-job">${escapeHtml(candidate.job_title)}</div>` : ''}
   497→            </div>
   498→            <span class="onboarding-status status-${candidate.onboarding_status || 'pending'}">
   499→              ${formatStatus(candidate.onboarding_status)}
   500→            </span>
   501→          </div>
   502→
   503→          <!-- Progress bar -->
   504→          <div style="margin-top: 12px;">
   505→            <div style="display: flex; justify-content: space-between; align-items: center; margin-bottom: 4px;">
   506→              <span style="font-size: 12px; color: var(--text-muted);">Document Progress</span>
   507→              <span style="font-size: 12px; font-weight: 600; color: ${barColor};">${pct}%</span>
   508→            </div>
   509→            <div style="height: 6px; background: var(--border-color); border-radius: 3px; overflow: hidden;">
   510→              <div style="height: 100%; width: ${pct}%; background: ${barColor}; border-radius: 3px; transition: width 0.3s;"></div>
   511→            </div>
   512→          </div>
   513→
   514→          <div class="docs-summary">
   515→            <div class="summary-item">
   516→              <strong>${signed}</strong>
   517→              <span>Signed</span>
   518→            </div>
   519→            <div class="summary-item">
   520→              <strong>${total}</strong>
   521→              <span>Total Docs</span>
   522→            </div>
   523→          </div>
   524→
   525→          ${candidate.due_date ? `
   526→            <div style="margin-top: 12px; padding-top: 12px; border-top: 1px solid var(--border-color); font-size: 12px; color: var(--text-muted);">
   527→              Due: ${formatDate(candidate.due_date)}
   528→            </div>
   529→          ` : ''}
   530→
   531→          <div style="margin-top: 12px; text-align: right;">
   532→            <a class="view-link" onclick="event.stopPropagation(); viewCandidateDetails(${candidate.candidate_id})">View Details →</a>
   533→          </div>
   534→        </div>
   535→      `}).join('');
   536→    }
   537→
   538→    async function viewCandidateDetails(candidateId) {
   539→      currentCandidateId = candidateId;
   540→
   541→      try {
   542→        // Load candidate details and documents
   543→        const response = await fetch(
   544→          `${BACKEND_URL}/onboarding/recruiter/candidate/${candidateId}/documents`,
   545→          { headers: { 'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}` } }
   546→        );
   547→
   548→        if (!response.ok) throw new Error('Failed to load documents');
   549→
   550→        const documents = await response.json();
   551→        const candidateData = documents[0];
   552→
   553→        // Update modal header
   554→        document.getElementById('modal-candidate-name').textContent = candidateData.candidate_name;
   555→        document.getElementById('modal-candidate-email').textContent = candidateData.candidate_email;
   556→
   557→        // Update details
   558→        document.getElementById('modal-job').textContent = candidateData.job_title || '—';
   559→        document.getElementById('modal-start-date').textContent = candidateData.start_date ? formatDate(candidateData.start_date) : '—';
   560→
   561→        // Display documents
   562→        const docsContainer = document.getElementById('modal-documents');
   563→        if (documents.length === 0) {
   564→          docsContainer.innerHTML = '<p style="color: var(--text-muted);">No documents yet</p>';
   565→        } else {
   566→          docsContainer.innerHTML = documents.map(doc => `
   567→            <div class="doc-item" style="border-left-color: ${doc.signed_at ? '#10b981' : doc.uploaded_at ? '#3b82f6' : '#e5e7eb'};">
   568→              <div class="doc-item-header">
   569→                <span class="doc-item-type">${escapeHtml(doc.document_type)}</span>
   570→                <span class="doc-item-date">${formatDate(doc.created_at)}</span>
   571→              </div>
   572→              <div style="display: flex; align-items: center; justify-content: space-between; margin-top: 8px;">
   573→                <div>
   574→                  ${doc.signed_at ? `
   575→                    <span class="document-status status-signed">✓ Signed ${formatDate(doc.signed_at)}</span>
   576→                  ` : doc.uploaded_at ? `
   577→                    <span class="document-status status-uploaded">✓ Generated</span>
   578→                  ` : `
   579→                    <span class="document-status status-pending-doc">Pending</span>
   580→                  `}
   581→                </div>
   582→                <div style="display: flex; gap: 8px;">
   583→                  ${doc.document_content ? `<a class="view-link" onclick="viewDocContent(${doc.id}, '${escapeHtml(doc.document_type)}')">View Details</a>` : ''}

   630→    function escapeHtml(text) {
