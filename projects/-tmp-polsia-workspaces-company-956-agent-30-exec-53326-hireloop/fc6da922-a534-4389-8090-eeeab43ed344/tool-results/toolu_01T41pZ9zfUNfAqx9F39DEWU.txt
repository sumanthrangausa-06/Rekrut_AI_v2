   139→    );
   140→
   141→    if (result.rows.length === 0) {
   142→      return res.status(404).json({ error: 'Offer not found' });
   143→    }
   144→
   145→    // Create default onboarding checklist
   146→    const offer = result.rows[0];
   147→    const defaultItems = [
   148→      { id: 1, task: 'Complete I-9 form', required: true },
   149→      { id: 2, task: 'Upload government-issued ID', required: true },
   150→      { id: 3, task: 'Set up direct deposit', required: true },
   151→      { id: 4, task: 'Complete tax withholding (W-4)', required: true },
   152→      { id: 5, task: 'Sign employee handbook acknowledgment', required: true },
   153→      { id: 6, task: 'Submit emergency contact information', required: true },
   154→      { id: 7, task: 'Complete IT setup (email, laptop)', required: false },
   155→      { id: 8, task: 'Schedule first day orientation', required: true }
   156→    ];
   157→
   158→    await pool.query(
   159→      `INSERT INTO onboarding_checklists
   160→       (offer_id, candidate_id, title, description, items, due_date)

   171→
   172→    // Auto-create employee record for payroll (bridges hiring → payroll pipeline)
   173→    try {
   174→      const empNum = 'EMP-' + String(offer.candidate_id).padStart(4, '0');
   175→      const empResult = await pool.query(`
   176→        INSERT INTO employees (user_id, employer_id, company_id, employee_number, position, employment_type, start_date, status)
   177→        VALUES ($1, $2, $3, $4, $5, 'full-time', $6, 'active')
   178→        ON CONFLICT DO NOTHING
   179→        RETURNING id
   180→      `, [
   181→        offer.candidate_id,
   182→        offer.recruiter_id,
   183→        offer.company_id,
   184→        empNum,
   185→        offer.title,
   186→        offer.start_date || new Date()
   187→      ]);
   188→
   189→      if (empResult.rows.length > 0) {
   190→        const annualSalary = parseFloat(offer.salary || 50000);
   191→        await pool.query(`
   192→          INSERT INTO payroll_configs (employee_id, salary_type, salary_amount, pay_frequency, payment_method, tax_filing_status)
   193→          VALUES ($1, 'salary', $2, 'bi-weekly', 'direct_deposit', 'single')
   194→          ON CONFLICT (employee_id) DO NOTHING
   195→        `, [empResult.rows[0].id, annualSalary]);
   196→      }
   197→    } catch (empErr) {
   198→      console.error('Auto-create employee record failed (non-blocking):', empErr.message);
   199→    }
   200→
   201→    res.json(result.rows[0]);
   202→  } catch (err) {
   203→    console.error('Error accepting offer:', err);
   204→    res.status(500).json({ error: 'Failed to accept offer' });
   205→  }

   473→// Start or continue chat session
   474→router.post('/assistant/chat', authMiddleware, async (req, res) => {
   475→  try {
   476→    const { message, checklist_id } = req.body;
   477→
   478→    // Get or create chat session
   479→    let session = await pool.query(
   480→      `SELECT * FROM onboarding_chats
   481→       WHERE candidate_id = $1 AND checklist_id = $2 AND is_active = true
   482→       ORDER BY session_started DESC LIMIT 1`,
   483→      [req.user.id, checklist_id || null]
   484→    );
   485→
   486→    let sessionId;
   487→    let messages = [];
   488→
   489→    if (session.rows.length === 0) {
   490→      const newSession = await pool.query(
   491→        `INSERT INTO onboarding_chats (candidate_id, checklist_id, messages)
   492→         VALUES ($1, $2, $3)
   493→         RETURNING *`,
   494→        [req.user.id, checklist_id || null, JSON.stringify([])]
   495→      );
   496→      sessionId = newSession.rows[0].id;
   497→    } else {
   498→      sessionId = session.rows[0].id;
   499→      messages = session.rows[0].messages || [];
   500→    }
   501→
   502→    // Get company policies for context
   503→    const policies = await pool.query(
   504→      `SELECT category, title, content
   505→       FROM company_policies
   506→       WHERE is_active = true
   507→       ORDER BY category`
   508→    );
   509→
   510→    // Build context from policies
   511→    const policyContext = policies.rows.map(p =>
   512→      `${p.category}: ${p.title}\n${p.content}`
   513→    ).join('\n\n');
   514→
   515→    // Add user message
   516→    messages.push({
   517→      role: 'user',
   518→      content: message,
   519→      timestamp: new Date().toISOString()
   520→    });
   521→
   522→    // Get AI response
   523→    const aiResponse = await getOnboardingAssistantResponse(
   524→      messages,
   525→      policyContext,
   526→      req.user
   527→    );
   528→
   529→    // Add AI response
   530→    messages.push({
   531→      role: 'assistant',
   532→      content: aiResponse,
   533→      timestamp: new Date().toISOString()
   534→    });
   535→
   536→    // Update session
   537→    await pool.query(
   538→      `UPDATE onboarding_chats
   539→       SET messages = $1, last_activity = NOW()
   540→       WHERE id = $2`,
   541→      [JSON.stringify(messages), sessionId]
   542→    );
   543→
   544→    res.json({ response: aiResponse, session_id: sessionId });
   545→  } catch (err) {
   546→    console.error('Error in assistant chat:', err);
   547→    res.status(500).json({ error: 'Failed to process chat message' });
   548→  }
   549→});

  1389→function escapeHtmlServer(text) {
  1390→  if (!text) return '';
  1391→  const map = { '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;' };
  1392→  return String(text).replace(/[&<>"']/g, m => map[m]);
  1393→}
