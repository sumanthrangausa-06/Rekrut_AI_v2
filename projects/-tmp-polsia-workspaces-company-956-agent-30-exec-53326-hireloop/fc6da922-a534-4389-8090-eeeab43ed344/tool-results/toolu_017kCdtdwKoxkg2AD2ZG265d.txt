   113→    let currentFilter = 'all';
   114→
   115→    document.addEventListener('DOMContentLoaded', async () => {
   116→      const user = await checkAuth();
   117→      if (!user) {
   118→        window.location.href = '/login.html';
   119→        return;
   120→      }
   121→      // Verify recruiter/employer role - must have company_id and appropriate role
   122→      const allowedRoles = ['employer', 'recruiter', 'hiring_manager', 'admin'];
   123→      if (!user.company_id || !allowedRoles.includes(user.role)) {
   124→        window.location.href = '/candidate-dashboard.html';
   125→        return;
   126→      }
   127→
   128→      const name = user.name || user.email.split('@')[0];
   129→      document.getElementById('user-name').textContent = name;
   130→      document.getElementById('user-avatar').textContent = name.charAt(0).toUpperCase();
   131→      document.getElementById('company-name').textContent = user.company_name || 'Company';
   132→
   133→      loadJobs();
   134→
   135→      // Filter tabs
   136→      document.querySelectorAll('.filter-tab').forEach(tab => {
   137→        tab.addEventListener('click', () => {
   138→          document.querySelectorAll('.filter-tab').forEach(t => t.classList.remove('active'));
   139→          tab.classList.add('active');
   140→          currentFilter = tab.dataset.status;
   141→          loadJobs();
   142→        });
   143→      });
   144→    });
   145→
   146→    async function loadJobs() {
   147→      try {
   148→        const url = currentFilter === 'all' ? '/recruiter/jobs' : `/recruiter/jobs?status=${currentFilter}`;
   149→        const data = await apiCall(url);
   150→        if (!data) return;
   151→
   152→        renderJobs(data.jobs || []);
   153→      } catch (err) {
   154→        console.error('Failed to load jobs:', err);
   155→      }
   156→    }
   157→
   158→    function renderJobs(jobs) {
   159→      const container = document.getElementById('jobs-list');
   160→
   161→      if (jobs.length === 0) {
   162→        container.innerHTML = `
   163→          <div class="empty-state">
   164→            <div class="empty-icon">💼</div>
   165→            <h3>No jobs found</h3>
   166→            <p>Create your first job posting to start receiving applications.</p>
   167→            <a href="/job-create.html" class="btn btn-primary">Post New Job</a>
   168→          </div>
   169→        `;
   170→        return;
   171→      }
   172→
   173→      container.innerHTML = jobs.map(job => `
   174→        <div class="job-card">
   175→          <div class="job-header">
   176→            <div class="job-title-row">
   177→              <h3>${job.title}</h3>
   178→              <span class="status-badge ${job.status}">${job.status}</span>
   179→            </div>
   180→            <div class="job-meta">
   181→              ${job.location ? `<span>📍 ${job.location}</span>` : ''}
   182→              ${job.salary_range ? `<span>💰 ${job.salary_range}</span>` : ''}
   183→              <span>📅 ${new Date(job.created_at).toLocaleDateString()}</span>
   184→            </div>
   185→          </div>
   186→
   187→          <div class="job-stats">
   188→            <div class="job-stat">
   189→              <span class="stat-value">${job.views || 0}</span>
   190→              <span class="stat-label">Views</span>
   191→            </div>
   192→            <div class="job-stat">
   193→              <span class="stat-value">${job.application_count || 0}</span>
   194→              <span class="stat-label">Applications</span>
   195→            </div>
   196→            <div class="job-stat">
   197→              <span class="stat-value">${job.interviews || 0}</span>
   198→              <span class="stat-label">Interviews</span>
   199→            </div>
   200→          </div>
   201→
   202→          <div class="job-actions">
   203→            <a href="/recruiter-applications.html?job_id=${job.id}" class="btn btn-ghost btn-sm">View Applications</a>
   204→            <button class="btn btn-ghost btn-sm" onclick="toggleJobStatus(${job.id}, '${job.status}')">
   205→              ${job.status === 'active' ? 'Pause' : 'Activate'}
   206→            </button>
   207→            <button class="btn btn-ghost btn-sm" onclick="editJob(${job.id})">Edit</button>
   208→          </div>
   209→        </div>
   210→      `).join('');
   211→    }
   212→
   213→    async function toggleJobStatus(jobId, currentStatus) {
   214→      const newStatus = currentStatus === 'active' ? 'paused' : 'active';
   215→      try {
   216→        await apiCall(`/recruiter/jobs/${jobId}`, {
   217→          method: 'PUT',
   218→          body: JSON.stringify({ status: newStatus })
   219→        });
   220→        loadJobs();
   221→      } catch (err) {
   222→        alert('Failed to update job status');
   223→      }
   224→    }
   225→
   226→    let editingJob = null;
   227→
   228→    function editJob(jobId) {
   229→      // Find the job in the loaded data
   230→      editingJob = jobId;
   231→      // Navigate to job create page with edit param
   232→      window.location.href = `/job-create.html?edit=${jobId}`;
   233→    }
   234→
   235→    async function deleteJob(jobId) {
   236→      if (!confirm('Are you sure you want to delete this job posting? This cannot be undone.')) return;
   237→      try {
   238→        await apiCall(`/recruiter/jobs/${jobId}`, { method: 'PUT', body: JSON.stringify({ status: 'closed' }) });
   239→        loadJobs();
   240→      } catch (err) {
   241→        alert('Failed to close job');
   242→      }
   243→    }
