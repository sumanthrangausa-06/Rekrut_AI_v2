   275→    function applyFilters() {
   276→      const location = document.getElementById('filter-location').value.toLowerCase();
   277→      const type = document.getElementById('filter-type').value;
   278→      const minMatch = parseInt(document.getElementById('filter-match').value) || 0;
   279→
   280→      let filtered = allJobs.filter(job => {
   281→        if (location && job.location && !job.location.toLowerCase().includes(location)) return false;
   282→        if (type && job.job_type !== type) return false;
   283→        if (minMatch && (!job.match || job.match.match_score < minMatch)) return false;
   284→        return true;
   285→      });
   286→
   287→      renderJobs(filtered);
   288→    }

   506→    async function toggleSave(jobId) {
   507→      const isSaved = savedJobs.some(s => s.id === jobId);
   508→
   509→      try {
   510→        if (isSaved) {
   511→          await apiCall(`/candidate/jobs/${jobId}/save`, { method: 'DELETE' });
   512→          savedJobs = savedJobs.filter(s => s.id !== jobId);
   513→        } else {
   514→          await apiCall(`/candidate/jobs/${jobId}/save`, { method: 'POST' });
   515→          const job = allJobs.find(j => j.id === jobId);
   516→          if (job) savedJobs.push(job);
   517→        }
   518→
   519→        document.getElementById('saved-count').textContent = savedJobs.length;
   520→
   521→        // Re-render current view
   522→        if (currentTab === 'saved') {
   523→          renderSavedJobs();
   524→        } else {
   525→          renderJobs(allJobs);
   526→        }
   527→
   528→        // Update detail panel if open
   529→        if (selectedJobId === jobId) {
   530→          showJobDetail(jobId);
   531→        }
   532→      } catch (err) {
   533→        console.error('Toggle save error:', err);
   534→      }
   535→    }

   548→        (match.match_level || 'fair').charAt(0).toUpperCase() + (match.match_level || 'fair').slice(1) + ' Match';

   569→    async function submitApplication() {
   570→      const coverLetter = document.getElementById('cover-letter').value;
   571→      const btn = document.getElementById('submit-apply-btn');
   572→
   573→      btn.textContent = 'Submitting...';
   574→      btn.disabled = true;
   575→
   576→      try {
   577→        const data = await apiCall(`/candidate/jobs/${selectedJobId}/apply`, {
   578→          method: 'POST',
   579→          body: JSON.stringify({ cover_letter: coverLetter })
   580→        });
   581→
   582→        if (data.success) {
   583→          closeModal();
   584→          applications.push(data.application);
   585→          document.getElementById('applied-count').textContent = applications.length;
   586→
   587→          // Re-render and show success
   588→          if (currentTab === 'recommended' || currentTab === 'all') {
   589→            renderJobs(allJobs);
   590→          }
   591→
   592→          // Update detail panel
   593→          showJobDetail(selectedJobId);
   594→
   595→          alert('Application submitted successfully! 🎉');
   596→        }
   597→      } catch (err) {
   598→        console.error('Apply error:', err);
   599→        alert('Failed to submit application. Please try again.');
   600→      } finally {
   601→        btn.textContent = 'Submit Application';
   602→        btn.disabled = false;
   603→      }
   604→    }
