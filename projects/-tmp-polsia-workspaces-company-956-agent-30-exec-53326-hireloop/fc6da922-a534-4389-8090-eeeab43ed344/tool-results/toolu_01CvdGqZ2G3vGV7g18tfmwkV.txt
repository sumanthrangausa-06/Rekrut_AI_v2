     1→const jwt = require('jsonwebtoken');
     2→const crypto = require('crypto');
     3→const pool = require('./db');
     4→
     5→const JWT_SECRET = process.env.JWT_SECRET || 'hireloop-jwt-secret-change-in-prod';
     6→const ACCESS_TOKEN_EXPIRY = '15m';  // Short-lived access token
     7→const REFRESH_TOKEN_EXPIRY = 30 * 24 * 60 * 60 * 1000; // 30 days in ms
     8→
     9→// Generate access token (short-lived)
    10→function generateToken(user) {
    11→  return jwt.sign(
    12→    { id: user.id, email: user.email, role: user.role },
    13→    JWT_SECRET,
    14→    { expiresIn: ACCESS_TOKEN_EXPIRY }
    15→  );
    16→}
    17→
    18→// Generate long-lived token for backwards compatibility
    19→function generateLongToken(user) {
    20→  return jwt.sign(
    21→    { id: user.id, email: user.email, role: user.role },
    22→    JWT_SECRET,
    23→    { expiresIn: '7d' }
    24→  );
    25→}
    26→
    27→// Verify access token
    28→function verifyToken(token) {
    29→  try {
    30→    return jwt.verify(token, JWT_SECRET);
    31→  } catch (e) {
    32→    return null;
    33→  }
    34→}

    36→// Generate refresh token
    37→async function generateRefreshToken(userId) {
    38→  const token = crypto.randomBytes(40).toString('hex');
    39→  const tokenHash = crypto.createHash('sha256').update(token).digest('hex');
    40→  const familyId = crypto.randomBytes(16).toString('hex');
    41→  const expiresAt = new Date(Date.now() + REFRESH_TOKEN_EXPIRY);
    42→
    43→  await pool.query(`
    44→    INSERT INTO refresh_tokens (user_id, token_hash, family_id, expires_at)
    45→    VALUES ($1, $2, $3, $4)
    46→  `, [userId, tokenHash, familyId, expiresAt]);
    47→
    48→  return { token, familyId };
    49→}
    50→
    51→// Validate and rotate refresh token
    52→async function rotateRefreshToken(refreshToken) {
    53→  const tokenHash = crypto.createHash('sha256').update(refreshToken).digest('hex');
    54→
    55→  // Find the token
    56→  const result = await pool.query(`
    57→    SELECT rt.*, u.email, u.role, u.name
    58→    FROM refresh_tokens rt
    59→    JOIN users u ON rt.user_id = u.id
    60→    WHERE rt.token_hash = $1
    61→  `, [tokenHash]);
    62→
    63→  if (result.rows.length === 0) {
    64→    return { error: 'Invalid refresh token' };
    65→  }
    66→
    67→  const storedToken = result.rows[0];
    68→
    69→  // Check if revoked
    70→  if (storedToken.is_revoked) {
    71→    // Token reuse detected - revoke entire family
    72→    await pool.query(
    73→      'UPDATE refresh_tokens SET is_revoked = true WHERE family_id = $1',
    74→      [storedToken.family_id]
    75→    );
    76→    return { error: 'Token reuse detected - all tokens revoked' };
    77→  }
    78→
    79→  // Check expiration
    80→  if (new Date(storedToken.expires_at) < new Date()) {
    81→    return { error: 'Refresh token expired' };
    82→  }
    83→
    84→  // Revoke old token
    85→  await pool.query(
    86→    'UPDATE refresh_tokens SET is_revoked = true, last_used_at = NOW() WHERE id = $1',
    87→    [storedToken.id]
    88→  );
    89→
    90→  // Generate new tokens in same family
    91→  const newToken = crypto.randomBytes(40).toString('hex');
    92→  const newTokenHash = crypto.createHash('sha256').update(newToken).digest('hex');
    93→  const newExpiresAt = new Date(Date.now() + REFRESH_TOKEN_EXPIRY);
    94→
    95→  await pool.query(`
    96→    INSERT INTO refresh_tokens (user_id, token_hash, family_id, expires_at)
    97→    VALUES ($1, $2, $3, $4)
    98→  `, [storedToken.user_id, newTokenHash, storedToken.family_id, newExpiresAt]);
    99→
   100→  // Generate new access token
   101→  const user = {
   102→    id: storedToken.user_id,
   103→    email: storedToken.email,
   104→    role: storedToken.role,
   105→    name: storedToken.name
   106→  };
   107→  const accessToken = generateToken(user);
   108→
   109→  return {
   110→    accessToken,
   111→    refreshToken: newToken,
   112→    user
   113→  };
   114→}
   115→
   116→// Revoke all refresh tokens for a user (logout everywhere)
   117→async function revokeAllTokens(userId) {
   118→  await pool.query(
   119→    'UPDATE refresh_tokens SET is_revoked = true WHERE user_id = $1',
   120→    [userId]
   121→  );
   122→}

   124→// Auth middleware with refresh token support
   125→async function authMiddleware(req, res, next) {
   126→  const token = req.headers.authorization?.split(' ')[1] || req.session?.token;
   127→
   128→  if (!token) {
   129→    return res.status(401).json({ error: 'Authentication required' });
   130→  }
   131→
   132→  const decoded = verifyToken(token);
   133→  if (!decoded) {
   134→    return res.status(401).json({ error: 'Invalid or expired token', code: 'TOKEN_EXPIRED' });
   135→  }
   136→
   137→  try {
   138→    const result = await pool.query('SELECT * FROM users WHERE id = $1', [decoded.id]);
   139→    if (result.rows.length === 0) {
   140→      return res.status(401).json({ error: 'User not found' });
   141→    }
   142→    req.user = result.rows[0];
   143→    next();
   144→  } catch (err) {
   145→    console.error('Auth error:', err);
   146→    res.status(500).json({ error: 'Authentication error' });
   147→  }
   148→}
   149→
   150→async function optionalAuth(req, res, next) {
   151→  const token = req.headers.authorization?.split(' ')[1] || req.session?.token;
   152→
   153→  if (token) {
   154→    const decoded = verifyToken(token);
   155→    if (decoded) {
   156→      try {
   157→        const result = await pool.query('SELECT * FROM users WHERE id = $1', [decoded.id]);
   158→        if (result.rows.length > 0) {
   159→          req.user = result.rows[0];
   160→        }
   161→      } catch (err) {
   162→        console.error('Optional auth error:', err);
   163→      }
   164→    }
   165→  }
   166→  next();
   167→}
   168→
   169→function requireRole(...roles) {
   170→  return (req, res, next) => {
   171→    if (!req.user) {
   172→      return res.status(401).json({ error: 'Authentication required' });
   173→    }
   174→    if (!roles.includes(req.user.role)) {
   175→      return res.status(403).json({ error: 'Insufficient permissions' });
   176→    }
   177→    next();
   178→  };
   179→}
