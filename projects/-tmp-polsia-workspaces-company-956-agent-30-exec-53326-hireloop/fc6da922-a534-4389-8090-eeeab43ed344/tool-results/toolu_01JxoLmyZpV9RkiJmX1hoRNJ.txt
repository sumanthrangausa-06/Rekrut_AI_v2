   368→    function formatDate(dateStr) {
   369→      const date = new Date(dateStr);
   370→      const now = new Date();
   371→      const diff = now - date;
   372→      const days = Math.floor(diff / (1000 * 60 * 60 * 24));
   373→      if (days === 0) return 'Today';
   374→      if (days === 1) return 'Yesterday';
   375→      if (days < 7) return `${days} days ago`;
   376→      return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
   377→    }

   400→    function makeOffer(appId, candidateId, candidateName, candidateEmail, jobId, jobTitle) {
   401→      currentOfferData = { appId, candidateId, candidateName, candidateEmail, jobId, jobTitle };
   402→
   403→      // Pre-fill the modal
   404→      document.getElementById('offer-candidate-name').textContent = candidateName;
   405→      document.getElementById('offer-job-title').value = jobTitle;
   406→
   407→      // Set default start date to 2 weeks from now
   408→      const startDate = new Date();
   409→      startDate.setDate(startDate.getDate() + 14);
   410→      document.getElementById('offer-start-date').value = startDate.toISOString().split('T')[0];
   411→
   412→      // Show modal
   413→      document.getElementById('offer-modal').style.display = 'flex';
   414→    }

   422→    async function submitOffer(e) {
   423→      e.preventDefault();
   424→
   425→      if (!currentOfferData) return;
   426→
   427→      const offerData = {
   428→        candidate_id: currentOfferData.candidateId,
   429→        job_id: currentOfferData.jobId,
   430→        title: document.getElementById('offer-job-title').value,
   431→        salary: parseFloat(document.getElementById('offer-salary').value),
   432→        start_date: document.getElementById('offer-start-date').value,
   433→        benefits: document.getElementById('offer-benefits').value
   434→      };
   435→
   436→      try {
   437→        // Create the offer
   438→        const createResult = await apiCall('/onboarding/offers', {
   439→          method: 'POST',
   440→          body: JSON.stringify(offerData)
   441→        });
   442→
   443→        if (!createResult || createResult.error) {
   444→          throw new Error(createResult?.error || 'Failed to create offer');
   445→        }
   446→
   447→        // Send the offer immediately
   448→        const sendResult = await apiCall(`/onboarding/offers/${createResult.id}/send`, {
   449→          method: 'POST'
   450→        });
   451→
   452→        // Update application status
   453→        await apiCall(`/recruiter/applications/${currentOfferData.appId}/status`, {
   454→          method: 'PUT',
   455→          body: JSON.stringify({ status: 'offered' })
   456→        });
   457→
   458→        alert('Offer sent successfully to ' + currentOfferData.candidateName + '!');
   459→        closeOfferModal();
   460→        loadApplications();
   461→      } catch (err) {
   462→        console.error('Error creating offer:', err);
   463→        alert('Failed to create offer: ' + err.message);
   464→      }
   465→    }
