   298→    // Load available assessments
   299→    async function loadAssessments() {
   300→      try {
   301→        const response = await fetch('/api/assessments/available', {
   302→          headers: { 'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}` }
   303→        });
   304→
   305→        if (!response.ok) throw new Error('Failed to load assessments');
   306→
   307→        const data = await response.json();
   308→        const grid = document.getElementById('assessments-grid');
   309→
   310→        if (data.skills.length === 0) {
   311→          grid.innerHTML = `
   312→            <div class="empty-state">
   313→              <div class="empty-state-icon">📝</div>
   314→              <p>No skills found. Add skills to your profile to unlock assessments.</p>
   315→              <a href="/candidate-profile.html" class="btn btn-primary" style="margin-top: 16px;">Add Skills</a>
   316→            </div>
   317→          `;
   318→          return;
   319→        }
   320→
   321→        grid.innerHTML = data.skills.map(skill => `
   322→          <div class="assessment-card ${skill.is_verified ? 'verified' : ''}" onclick="startAssessment(${skill.id})">
   323→            <div class="assessment-header">
   324→              <div>
   325→                <div class="assessment-title">${skill.skill_name}</div>
   326→                <div class="assessment-category">${skill.category}</div>
   327→              </div>
   328→              ${skill.is_verified ? '<span class="verified-badge">✓ Verified</span>' : ''}
   329→            </div>
   330→
   331→            <div class="assessment-stats">
   332→              <div class="stat-item">
   333→                <span class="stat-value">${skill.assessment_count || 0}</span>
   334→                <span class="stat-label">Attempts</span>
   335→              </div>
   336→              <div class="stat-item">
   337→                <span class="stat-value">${skill.best_score || '-'}</span>
   338→                <span class="stat-label">Best Score</span>
   339→              </div>
   340→            </div>
   341→
   342→            <div class="assessment-actions">
   343→              <button class="btn ${skill.is_verified ? 'btn-ghost' : 'btn-primary'}" style="flex: 1;">
   344→                ${skill.is_verified ? '🔄 Retake' : '▶️ Start Test'}
   345→              </button>
   346→            </div>
   347→          </div>
   348→        `).join('');
   349→      } catch (error) {
   350→        console.error('Error loading assessments:', error);
   351→        document.getElementById('assessments-grid').innerHTML = `
   352→          <div class="empty-state">
   353→            <div class="empty-state-icon">❌</div>
   354→            <p>Failed to load assessments</p>
   355→          </div>
   356→        `;
   357→      }
   358→    }
   359→
   360→    // Load past results
   361→    async function loadResults() {
   362→      try {
   363→        const response = await fetch('/api/assessments/results', {
   364→          headers: { 'Authorization': `Bearer ${localStorage.getItem('rekrutai_token')}` }
   365→        });
   366→
   367→        if (!response.ok) throw new Error('Failed to load results');
   368→
   369→        const data = await response.json();
   370→        const list = document.getElementById('results-list');
   371→
   372→        if (data.results.length === 0) {
   373→          list.innerHTML = `
   374→            <div class="empty-state">
   375→              <div class="empty-state-icon">📊</div>
   376→              <p>No completed assessments yet</p>
   377→            </div>
   378→          `;
   379→          return;
   380→        }
   381→
   382→        list.innerHTML = data.results.map(result => {
   383→          const date = new Date(result.completed_at);
   384→          const hasWarnings = result.tab_switches > 0 || result.copy_paste_attempts > 0;
   385→
   386→          return `
   387→            <div class="result-item">
   388→              <div class="result-score">${result.score}</div>
   389→              <div class="result-info">
   390→                <div class="result-title">${result.skill_name}</div>
   391→                <div class="result-meta">
   392→                  Completed ${date.toLocaleDateString()} • Difficulty: ${result.max_difficulty_reached || 2}/5
   393→                </div>
   394→                <div class="result-badges">
   395→                  <span class="badge ${result.passed ? 'passed' : ''}">${result.passed ? '✓ Passed' : '✗ Failed'}</span>
   396→                  ${hasWarnings ? '<span class="badge warning">⚠️ Behavioral flags</span>' : '<span class="badge">✓ Clean attempt</span>'}
   397→                </div>
   398→              </div>
   399→              <a href="/assessment-results.html?id=${result.id}" class="btn btn-ghost">View Details</a>
   400→            </div>
   401→          `;
   402→        }).join('');
   403→      } catch (error) {
   404→        console.error('Error loading results:', error);
   405→      }
   406→    }
