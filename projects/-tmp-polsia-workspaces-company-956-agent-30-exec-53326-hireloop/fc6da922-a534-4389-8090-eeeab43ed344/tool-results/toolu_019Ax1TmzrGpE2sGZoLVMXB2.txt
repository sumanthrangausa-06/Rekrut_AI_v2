     9→// Handle OAuth callback tokens from URL
    10→function handleOAuthCallback() {
    11→  const params = new URLSearchParams(window.location.search);
    12→  const token = params.get('token');
    13→  const refresh = params.get('refresh');
    14→
    15→  if (token) {
    16→    localStorage.setItem(TOKEN_KEY, token);
    17→    if (refresh) {
    18→      localStorage.setItem(REFRESH_KEY, refresh);
    19→    }
    20→    // Clean URL
    21→    window.history.replaceState({}, document.title, window.location.pathname);
    22→  }
    23→}

    25→// Refresh access token using refresh token
    26→async function refreshAccessToken() {
    27→  const refreshToken = localStorage.getItem(REFRESH_KEY);
    28→  if (!refreshToken) return null;
    29→
    30→  try {
    31→    const response = await fetch('/api/auth/refresh', {
    32→      method: 'POST',
    33→      headers: { 'Content-Type': 'application/json' },
    34→      body: JSON.stringify({ refreshToken })
    35→    });
    36→
    37→    if (response.ok) {
    38→      const data = await response.json();
    39→      if (data.success) {
    40→        localStorage.setItem(TOKEN_KEY, data.accessToken);
    41→        localStorage.setItem(REFRESH_KEY, data.refreshToken);
    42→        return data.accessToken;
    43→      }
    44→    }
    45→  } catch (err) {
    46→    console.error('Token refresh failed:', err);
    47→  }
    48→
    49→  // Refresh failed - clear tokens
    50→  localStorage.removeItem(TOKEN_KEY);
    51→  localStorage.removeItem(REFRESH_KEY);
    52→  return null;
    53→}

    55→// Check auth status on page load
    56→// Uses a promise guard to prevent race conditions when called from multiple DOMContentLoaded handlers
    57→async function checkAuth() {
    58→  // If already running, return the existing promise (prevents double token refresh)
    59→  if (_checkAuthPromise) return _checkAuthPromise;
    60→
    61→  _checkAuthPromise = _doCheckAuth();
    62→  try {
    63→    return await _checkAuthPromise;
    64→  } finally {
    65→    _checkAuthPromise = null;
    66→  }
    67→}

    69→async function _doCheckAuth() {
    70→  // Handle OAuth callback first
    71→  handleOAuthCallback();
    72→
    73→  const token = localStorage.getItem(TOKEN_KEY);
    74→  if (!token) return null;
    75→
    76→  try {
    77→    const response = await fetch('/api/auth/me', {
    78→      headers: { 'Authorization': `Bearer ${token}` }
    79→    });
    80→
    81→    if (response.ok) {
    82→      const data = await response.json();
    83→      currentUser = data.user;
    84→      updateNavForAuth();
    85→      return data.user;
    86→    }
    87→
    88→    // Token expired - try refresh
    89→    if (response.status === 401) {
    90→      const newToken = await refreshAccessToken();
    91→      if (newToken) {
    92→        // Retry with new token (reset promise guard for retry)
    93→        _checkAuthPromise = null;
    94→        return await checkAuth();
    95→      }
    96→    }
    97→  } catch (err) {
    98→    console.error('Auth check failed:', err);
    99→  }
   100→  return null;
   101→}

   122→async function handleRegister(event) {

   126→  const errorEl = document.getElementById('register-error');

   132→  try {
   133→    const response = await fetch('/api/auth/register', {
   134→      method: 'POST',
   135→      headers: { 'Content-Type': 'application/json' },
   136→      body: JSON.stringify({
   137→        email: form.email.value,
   138→        password: form.password.value,
   139→        name: form.name?.value || '',
   140→        role: form.role?.value || 'candidate',
   141→        company_name: form.company_name?.value || ''
   142→      })
   143→    });
   144→
   145→    const data = await response.json();
   146→
   147→    if (data.success) {
   148→      localStorage.setItem(TOKEN_KEY, data.accessToken || data.token);
   149→      if (data.refreshToken) {
   150→        localStorage.setItem(REFRESH_KEY, data.refreshToken);
   151→      }
   152→      // Redirect based on role - all recruiter-type roles go to recruiter dashboard
   153→      const role = form.role?.value || 'candidate';
   154→      const recruiterRoles = ['recruiter', 'hiring_manager', 'employer', 'admin'];
   155→      window.location.href = recruiterRoles.includes(role)
   156→        ? '/recruiter-dashboard.html'
   157→        : '/candidate-dashboard.html';
   158→    } else {
   159→      if (errorEl) {
   160→        errorEl.textContent = data.error || 'Registration failed';
   161→        errorEl.style.display = 'block';
   162→      }
   163→    }
   164→  } catch (err) {
   165→    if (errorEl) {
   166→      errorEl.textContent = 'Network error. Please try again.';
   167→      errorEl.style.display = 'block';
   168→    }
   169→  } finally {
   170→    submitBtn.disabled = false;
   171→    submitBtn.textContent = 'Create Account';
   172→  }
   173→}

   243→// API helper with automatic token refresh
   244→async function apiCall(endpoint, options = {}) {
   245→  let token = localStorage.getItem(TOKEN_KEY);
   246→  const headers = {
   247→    'Content-Type': 'application/json',
   248→    ...(token && { 'Authorization': `Bearer ${token}` }),
   249→    ...options.headers
   250→  };
   251→
   252→  let response = await fetch(`/api${endpoint}`, {
   253→    ...options,
   254→    headers
   255→  });
   256→
   257→  // If 401 and we have a refresh token, try to refresh
   258→  if (response.status === 401) {
   259→    const newToken = await refreshAccessToken();
   260→    if (newToken) {
   261→      // Retry with new token
   262→      headers['Authorization'] = `Bearer ${newToken}`;
   263→      response = await fetch(`/api${endpoint}`, {
   264→        ...options,
   265→        headers
   266→      });
   267→    } else {
   268→      // Still unauthorized - redirect to login
   269→      window.location.href = '/login.html';
   270→      return null;
   271→    }
   272→  }
   273→
   274→  return response.json();
   275→}

   355→// Mobile Menu Toggle
   356→function initMobileMenu() {
   357→  const navLinks = document.getElementById('nav-links');
   358→  const navContainer = document.querySelector('.nav-container');
   359→
   360→  // Create hamburger button if it doesn't exist
   361→  let mobileToggle = document.querySelector('.mobile-menu-toggle');
   362→  if (!mobileToggle && navLinks && navContainer) {
   363→    mobileToggle = document.createElement('button');
   364→    mobileToggle.className = 'mobile-menu-toggle';
   365→    mobileToggle.innerHTML = '☰';
   366→    mobileToggle.setAttribute('aria-label', 'Toggle menu');
   367→
   368→    // Insert before nav-auth
   369→    const navAuth = document.getElementById('nav-auth');
   370→    if (navAuth) {
   371→      navContainer.insertBefore(mobileToggle, navAuth);
   372→    }
   373→  }
   374→
   375→  if (mobileToggle && navLinks) {
   376→    // Toggle mobile menu
   377→    mobileToggle.addEventListener('click', (e) => {
   378→      e.stopPropagation();
   379→      navLinks.classList.toggle('mobile-open');
   380→      mobileToggle.innerHTML = navLinks.classList.contains('mobile-open') ? '✕' : '☰';
   381→    });
   382→
   383→    // Close menu when clicking outside
   384→    document.addEventListener('click', (e) => {
   385→      if (!navLinks.contains(e.target) && !mobileToggle.contains(e.target)) {
   386→        navLinks.classList.remove('mobile-open');
   387→        mobileToggle.innerHTML = '☰';
   388→      }
   389→    });
   390→
   391→    // Handle dropdown toggles on mobile
   392→    const dropdowns = navLinks.querySelectorAll('.nav-dropdown');
   393→    dropdowns.forEach(dropdown => {
   394→      const toggle = dropdown.querySelector('.nav-dropdown-toggle');
   395→      if (toggle) {
   396→        toggle.addEventListener('click', (e) => {
   397→          if (window.innerWidth <= 768) {
   398→            e.preventDefault();
   399→            dropdown.classList.toggle('active');
   400→          }
   401→        });
   402→      }
   403→    });
   404→  }
   405→}
