   217→    // Check recruiter auth
   218→    document.addEventListener('DOMContentLoaded', async () => {
   219→      const user = await checkAuth();
   220→      if (!user) {
   221→        window.location.href = '/login.html';
   222→        return;
   223→      }
   224→
   225→      // Verify recruiter/employer role - must have company_id and appropriate role
   226→      const allowedRoles = ['employer', 'recruiter', 'hiring_manager', 'admin'];
   227→      if (!user.company_id || !allowedRoles.includes(user.role)) {
   228→        window.location.href = '/candidate-dashboard.html';
   229→        return;
   230→      }
   231→
   232→      // Update user info
   233→      const name = user.name || user.email.split('@')[0];
   234→      document.getElementById('user-name').textContent = name;
   235→      document.getElementById('user-avatar').textContent = name.charAt(0).toUpperCase();
   236→      document.getElementById('company-name').textContent = user.company_name || 'Company';
   237→      document.getElementById('dropdown-user-name').textContent = name;
   238→      document.getElementById('dropdown-user-email').textContent = user.email;
   239→
   240→      // Setup user menu dropdown
   241→      const userMenuToggle = document.getElementById('user-menu-toggle');
   242→      const userDropdownMenu = document.getElementById('user-dropdown-menu');
   243→      if (userMenuToggle && userDropdownMenu) {
   244→        userMenuToggle.addEventListener('click', (e) => {
   245→          e.stopPropagation();
   246→          userDropdownMenu.classList.toggle('active');
   247→        });
   248→
   249→        document.addEventListener('click', () => {
   250→          userDropdownMenu.classList.remove('active');
   251→        });
   252→
   253→        userDropdownMenu.addEventListener('click', (e) => {
   254→          e.stopPropagation();
   255→        });
   256→      }
   257→
   258→      // Load dashboard data
   259→      loadRecruiterDashboard();
   260→    });
   261→
   262→    async function loadRecruiterDashboard() {
   263→      try {
   264→        const data = await apiCall('/recruiter/dashboard');
   265→        if (!data) return;
   266→
   267→        // Update TrustScore
   268→        if (data.trust_score) {
   269→          document.getElementById('trustscore-value').textContent = data.trust_score.total_score;
   270→          document.getElementById('trustscore-tier').textContent = data.trust_score.tier_label;
   271→          document.getElementById('trustscore-tier').style.background = data.trust_score.tier_color;
   272→        }
   273→
   274→        // Update stats
   275→        document.getElementById('active-jobs').textContent = data.job_stats?.active_jobs || 0;
   276→        document.getElementById('total-applications').textContent = data.application_stats?.total_applications || 0;
   277→        document.getElementById('new-applications').textContent = data.application_stats?.new_applications || 0;
   278→
   279→        const interviewed = parseInt(data.application_stats?.interviewed) || 0;
   280→        const offered = parseInt(data.application_stats?.offered) || 0;
   281→        const ratio = interviewed > 0 ? ((offered / interviewed) * 100).toFixed(0) + '%' : '-';
   282→        document.getElementById('offer-ratio').textContent = ratio;
   283→
   284→        // Render upcoming interviews
   285→        renderUpcomingInterviews(data.upcoming_interviews || []);
   286→
   287→        // Render recent applications
   288→        renderRecentApplications(data.recent_applications || []);
   289→
   290→      } catch (err) {
   291→        console.error('Failed to load dashboard:', err);
   292→      }
   293→    }
   294→
   295→    function renderUpcomingInterviews(interviews) {
   296→      const container = document.getElementById('upcoming-interviews');
   297→
   298→      if (interviews.length === 0) {
   299→        container.innerHTML = `
   300→          <div class="empty-state">
   301→            <p>No upcoming interviews</p>
   302→            <a href="/recruiter-applications.html" class="btn btn-ghost btn-sm">Review applications</a>
   303→          </div>
   304→        `;
   305→        return;
   306→      }
   307→
   308→      container.innerHTML = interviews.map(i => `
   309→        <div class="interview-item">
   310→          <div class="interview-time">
   311→            <span class="date">${new Date(i.scheduled_at).toLocaleDateString()}</span>
   312→            <span class="time">${new Date(i.scheduled_at).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}</span>
   313→          </div>
   314→          <div class="interview-info">
   315→            <span class="candidate-name">${i.candidate_name}</span>
   316→            <span class="job-title">${i.job_title}</span>
   317→          </div>
   318→          <span class="interview-type-badge ${i.interview_type}">${i.interview_type}</span>
   319→        </div>
   320→      `).join('');
   321→    }
   322→
   323→    function renderRecentApplications(applications) {
   324→      const container = document.getElementById('recent-applications');
   325→
   326→      if (applications.length === 0) {
   327→        container.innerHTML = `
   328→          <div class="empty-state">
   329→            <p>No applications yet</p>
   330→            <a href="/job-create.html" class="btn btn-ghost btn-sm">Post a job</a>
   331→          </div>
   332→        `;
   333→        return;
   334→      }
   335→
   336→      container.innerHTML = applications.map(a => `
   337→        <div class="application-item">
   338→          <div class="candidate-info">
   339→            <span class="candidate-name">${a.candidate_name}</span>
   340→            <span class="job-title">${a.job_title}</span>
   341→          </div>
   342→          <div class="application-meta">
   343→            ${a.omniscore_at_apply ? `<span class="omniscore-badge">${a.omniscore_at_apply} OmniScore</span>` : ''}
   344→            <span class="status-badge ${a.status}">${a.status}</span>
   345→          </div>
   346→        </div>
   347→      `).join('');
   348→    }
