   429→  // ─── Circuit Breaker ──────────────────────────────────────────
   430→
   431→  isCircuitOpen(providerKey, modality) {
   432→    const key = `${providerKey}:${modality}`;
   433→    const failure = this.failures[key];
   434→    if (!failure) return false;
   435→
   436→    const elapsed = Date.now() - failure.lastFailure;
   437→    if (elapsed > CIRCUIT_BREAK_MS) {
   438→      // Circuit closed — enough time passed, allow retry
   439→      delete this.failures[key];
   440→      return false;
   441→    }
   442→    return true;
   443→  }
   444→
   445→  recordFailure(providerKey, modality, error) {
   446→    const key = `${providerKey}:${modality}`;
   447→    const existing = this.failures[key] || { count: 0 };
   448→    const status = error.status || error.statusCode || null;
   449→
   450→    this.failures[key] = {
   451→      count: existing.count + 1,
   452→      lastFailure: Date.now(),
   453→      error: error.message || String(error),
   454→      status,
   455→    };
   456→
   457→    this.stats.totalFailovers++;
   458→
   459→    const logEntry = {
   460→      timestamp: new Date().toISOString(),
   461→      event: 'provider_failure',
   462→      provider: providerKey,
   463→      modality,
   464→      error: error.message || String(error),
   465→      status,
   466→    };
   467→    this.logs.push(logEntry);
   468→    if (this.logs.length > 100) this.logs.shift();
   469→
   470→    console.log(`[ai-provider] ❌ ${providerKey} failed for ${modality}: [${status || 'ERR'}] ${error.message || error}`);
   471→  }

   475→      timestamp: new Date().toISOString(),

   477→      from: fromProvider,
   478→      to: toProvider,
   479→      modality,
   480→    };
   481→    this.logs.push(logEntry);
   482→    if (this.logs.length > 100) this.logs.shift();
   483→    console.log(`[ai-provider] 🔄 ${modality} failover: ${fromProvider} → ${toProvider}`);
   484→    activityLogger.logFailover(modality, fromProvider, toProvider);
   485→  }
   486→
   487→  recordSuccess(providerKey, modality, tokenCount = 0) {
   488→    this.activeProviders[modality] = providerKey;
   489→    const key = `${providerKey}:${modality}`;
   490→    delete this.failures[key];
   491→
   492→    this.stats.totalCalls++;
   493→    this.stats.providerCalls[providerKey] = (this.stats.providerCalls[providerKey] || 0) + 1;
   494→
   495→    // Track OpenAI token usage for budget enforcement
   496→    if (this._isOpenAIProvider(providerKey) && tokenCount > 0) {
   497→      tokenBudget.recordUsage(modality, tokenCount);
   498→    }
   499→  }
   500→
   501→  isRetryable(err) {
   502→    // Network/timeout errors
   503→    if (err.code === 'ETIMEDOUT' || err.code === 'ECONNREFUSED' || err.code === 'ECONNRESET' || err.code === 'ENOTFOUND') return true;
   504→    if (err.message && (err.message.includes('timeout') || err.message.includes('ETIMEDOUT'))) return true;
   505→
   506→    // HTTP status codes
   507→    const status = err.status || err.statusCode || (err.response && err.response.status);
   508→    if (status && RETRYABLE_STATUS.has(status)) return true;
   509→
   510→    // OpenAI SDK-specific error types
   511→    if (err.type === 'insufficient_quota' || err.code === 'insufficient_quota') return true;
   512→    if (err.type === 'rate_limit_exceeded' || err.code === 'rate_limit_exceeded') return true;
   513→
   514→    // Anthropic-specific
   515→    if (err.error && err.error.type === 'overloaded_error') return true;
   516→
   517→    return false;
   518→  }
   519→
   520→  // Check if a provider key is an OpenAI provider (not Anthropic, not NIM)
   521→  _isOpenAIProvider(providerKey) {
   522→    return providerKey === 'openai' || providerKey.startsWith('openai_');
   523→  }

   525→  // Get available providers for a modality (skipping unavailable + circuit-broken + budget-exhausted OpenAI)
   526→  getAvailableChain(modality, variant = 'default') {
   527→    const budgetExhausted = tokenBudget.isOpenAIBudgetExhausted();
   528→    return this.getChain(modality, variant).filter(key => {
   529→      if (!this.isProviderAvailable(key)) return false;
   530→      if (this.isCircuitOpen(key, modality)) return false;
   531→      // Skip OpenAI providers when daily token budget is exhausted
   532→      if (budgetExhausted && this._isOpenAIProvider(key)) return false;
   533→      return true;
   534→    });
   535→  }

   537→  // ─── Core Execution with Fallback ────────────────────────────
   538→
   539→  /**
   540→   * Execute an AI call with automatic fallback.
   541→   * @param {string} modality - 'llm', 'vision', 'tts', 'asr', 'embedding', 'reranking', 'safety'
   542→   * @param {Function} executeFn - async (providerKey) => result
   543→   * @param {string} variant - Chain variant (default, quality, efficient, reasoning)
   544→   * @returns {*} Result from the first successful provider
   545→   */
   546→  async execute(modality, executeFn, variant = 'default') {
   547→    const chain = this.getAvailableChain(modality, variant);
   548→
   549→    if (chain.length === 0) {
   550→      const allChain = this.getChain(modality, variant);
   551→      const reasons = allChain.map(k => {
   552→        if (!this.isProviderAvailable(k)) return `${k}: not configured`;
   553→        if (this.isCircuitOpen(k, modality)) return `${k}: circuit open`;
   554→        return `${k}: unknown`;
   555→      });
   556→      throw new Error(`No available providers for ${modality}. ${reasons.join(', ')}`);
   557→    }
   558→
   559→    // Per-provider timeout: 8s for TTS/ASR (audio processing), 15s for LLM, 10s default
   560→    const timeoutMs = modality === 'tts' || modality === 'asr' ? 8000
   561→      : modality === 'llm' ? 15000
   562→      : 10000;
   563→
   564→    let lastError;
   565→    let failedProvider = null;  // Track from→to for failover logging
   566→    for (const providerKey of chain) {
   567→      try {
   568→        // Wrap each provider call with a timeout to prevent hanging
   569→        const result = await Promise.race([
   570→          executeFn(providerKey),
   571→          new Promise((_, reject) => setTimeout(() => {
   572→            const err = new Error(`${providerKey} timed out after ${timeoutMs}ms for ${modality}`);
   573→            err.code = 'ETIMEDOUT';
   574→            reject(err);
   575→          }, timeoutMs))
   576→        ]);
   577→        // Log successful failover transition if we had a prior failure
   578→        if (failedProvider) {
   579→          this._logFailover(failedProvider, providerKey, modality);
   580→        }
   581→        // Extract token count if result is wrapped with _tokenCount
   582→        let tokenCount = 0;
   583→        let actualResult = result;
   584→        if (result && typeof result === 'object' && result._tokenCount !== undefined) {
   585→          tokenCount = result._tokenCount;
   586→          actualResult = result._result;
   587→        }
   588→        this.recordSuccess(providerKey, modality, tokenCount);
   589→        return actualResult;
   590→      } catch (err) {
   591→        lastError = err;
   592→        // Provider not applicable for this request (e.g., voice cloning TTS without audio prompt)
   593→        // Skip silently without circuit-breaking — provider is fine, just wrong for this request
   594→        if (err.notApplicable) {
   595→          continue;
   596→        }
   597→        if (this.isRetryable(err)) {
   598→          failedProvider = providerKey;
   599→          this.recordFailure(providerKey, modality, err);
   600→          continue; // try next provider
   601→        }
   602→        // Non-retryable error (e.g., bad request, invalid input) — don't try others
   603→        throw err;
   604→      }
   605→    }
   606→
   607→    // All providers exhausted
   608→    console.error(`[ai-provider] All ${modality} providers exhausted.`);
   609→    throw lastError || new Error(`All ${modality} providers failed`);
   610→  }

   659→  async _anthropicChat(messages, options) {
   660→    const systemMsgs = messages.filter(m => m.role === 'system');
   661→    const nonSystemMsgs = messages.filter(m => m.role !== 'system');
   662→
   663→    const systemPrompt = options.system || (systemMsgs.length > 0 ? systemMsgs[0].content : undefined);
   664→    const msgs = nonSystemMsgs.map(m => ({
   665→      role: m.role,
   666→      content: typeof m.content === 'string' ? m.content : String(m.content),
   667→    }));
   668→    if (msgs.length === 0) msgs.push({ role: 'user', content: 'Hello' });
   669→
   670→    const response = await this.clients.anthropic.messages.create({
   671→      max_tokens: options.maxTokens || options.max_tokens || 8192,
   672→      messages: msgs,
   673→      system: systemPrompt,
   674→    }, {
   675→      headers: options.subscriptionId ? { 'X-Subscription-ID': options.subscriptionId } : {},
   676→    });
   677→
   678→    return response.content[0].text;
   679→  }

   681→  async _openaiChat(model, messages, options, isPolsia = false) {
   682→    const client = this.clients.openai;
   683→    const msgs = [];
   684→
   685→    if (options.system) {
   686→      msgs.push({ role: 'system', content: options.system });
   687→    }
   688→    for (const m of messages) {
   689→      if (m.role === 'system' && options.system) continue; // already added
   690→      msgs.push({ role: m.role, content: typeof m.content === 'string' ? m.content : String(m.content) });
   691→    }
   692→    if (msgs.length === 0) msgs.push({ role: 'user', content: 'Hello' });

   695→      model,
   696→      messages: msgs,
   697→      max_tokens: options.maxTokens || options.max_tokens || 8192,
   698→      temperature: options.temperature || 0.7,
   699→    };

   701→    // Polsia proxy requires task field
   702→    if (isPolsia) params.task = 'chat-fallback';

   704→
   705→    const response = await client.chat.completions.create(params);
   706→    const text = response.choices[0].message.content;

   708→    const tokenCount = (response.usage && response.usage.total_tokens)
   709→      ? response.usage.total_tokens
   710→      : tokenBudget.estimateTokensFromText(text) + tokenBudget.estimateTokensFromText(msgs.map(m => m.content).join(' '));
   711→    return { _result: text, _tokenCount: tokenCount };
   712→  }

   750→    const moduleConfig = MODULE_CHAINS[options.module];
   751→    const variant = (moduleConfig && moduleConfig.vision) || 'default';
   752→
   753→    return this.execute('vision', async (providerKey) => {
   754→      let client, model;
   755→
   756→      switch (providerKey) {
   757→        case 'openai_vision':
   758→          client = this.clients.openai;

   761→        case 'nim_cosmos':

   808→          ? response.usage.total_tokens
   809→          : tokenBudget.estimateVisionTokens(imageUrls.length, prompt.length);

   853→  async _openaiTTS(text, options) {
   854→    const voice = options.voice || 'nova';
   855→    const speed = options.speed || 1.0;
   856→    const truncatedText = text.length > 4000 ? text.substring(0, 4000) : text;
   857→
   858→    const baseUrl = this.openaiBaseUrl || 'https://api.openai.com/v1';
   859→    const apiKey = this.openaiApiKey;
   860→
   861→    const headers = {
   862→      'Authorization': `Bearer ${apiKey}`,
   863→      'Content-Type': 'application/json',
   864→    };
   865→    if (options.subscriptionId) headers['X-Subscription-ID'] = options.subscriptionId;
   866→
   867→    const res = await fetch(`${baseUrl}/audio/speech`, {
   868→      method: 'POST',
   869→      headers,
   870→      body: JSON.stringify({
   871→        model: 'tts-1',
   872→        input: truncatedText,
   873→        voice,
   874→        response_format: 'mp3',
   875→        speed,
   876→        task: 'tts-interview',
   877→      }),
   878→    });
   879→
   880→    if (!res.ok) {
   881→      const errText = await res.text();
   882→      const err = new Error(`TTS API error: ${res.status} ${errText.substring(0, 200)}`);
   883→      err.status = res.status;
   884→      throw err;
   885→    }
   886→
   887→    const arrayBuffer = await res.arrayBuffer();
   888→    const buffer = Buffer.from(arrayBuffer);
   889→    // Track TTS token usage
   890→    const ttsTokens = tokenBudget.estimateTTSTokens(truncatedText);
   891→    return { _result: buffer, _tokenCount: ttsTokens };
   892→  }

   936→  /**
   937→   * NIM Magpie/FastPitch TTS — Generic adapter for all NIM TTS models.
   938→   * Uses OpenAI-compatible /v1/audio/speech endpoint with model parameter.
   939→   * Voice-cloning models (Flow, Zeroshot) require an audio prompt — if not provided,
   940→   * they throw a notApplicable error and the fallback system skips them silently.
   941→   * @param {string} configKey - Key into NIM_TTS_CONFIGS
   942→   * @param {string} text - Text to synthesize
   943→   * @param {Object} options - { language, nimVoice, audioPrompt, audioPromptTranscript }
   944→   * @returns {Buffer} Audio buffer (WAV format)
   945→   */
   946→  async _nimMagpieTTS(configKey, text, options) {
   947→    const config = NIM_TTS_CONFIGS[configKey];
   948→    if (!config) throw new Error(`Unknown NIM TTS config: ${configKey}`);
   949→
   950→    // Voice-cloning models require audio prompts — skip gracefully if not provided
   951→    if (config.requiresAudioPrompt && !options.audioPrompt) {
   952→      const err = new Error(`${config.name} requires an audio prompt for voice cloning — skipping`);
   953→      err.notApplicable = true;
   954→      throw err;
   955→    }
   956→
   957→    const truncatedText = text.length > (config.maxTextLength || 4000)
   958→      ? text.substring(0, config.maxTextLength || 4000)
   959→      : text;
   960→
   961→    // Map config keys to NIM model IDs
   962→    const modelMap = {
   963→      nim_magpie_multilingual: NIM_MODELS.tts_magpie_multilingual,
   964→      nim_magpie_flow: NIM_MODELS.tts_magpie_flow,
   965→      nim_magpie_zeroshot: NIM_MODELS.tts_magpie_zeroshot,
   966→      nim_fastpitch: NIM_MODELS.tts_fastpitch,
   967→    };
   968→    const model = modelMap[configKey];
   969→    if (!model) throw new Error(`No NIM model ID for TTS config: ${configKey}`);
   970→
   971→    // Use model-specific base URL if configured, otherwise default NIM base
   972→    const baseUrl = config.baseUrl || process.env.NIM_TTS_BASE_URL || this.nimBaseUrl;
   973→
   974→    // Use OpenAI-compatible /v1/audio/speech endpoint (JSON body)
   975→    const voice = config.defaultVoice || options.nimVoice || 'default';
   976→
   977→    const res = await fetch(`${baseUrl}/audio/speech`, {
   978→      method: 'POST',
   979→      headers: {
   980→        'Authorization': `Bearer ${this.nimApiKey}`,
   981→        'Content-Type': 'application/json',
   982→        'Accept': 'audio/wav',
   983→      },
   984→      body: JSON.stringify({
   985→        model,
   986→        input: truncatedText,
   987→        voice,
   988→        response_format: 'wav',
   989→      }),
   990→    });
   991→
   992→    if (!res.ok) {
   993→      const errText = await res.text();
   994→      const err = new Error(`NIM ${config.name} error: ${res.status} ${errText.substring(0, 200)}`);
   995→      err.status = res.status;
   996→      throw err;
   997→    }
   998→
   999→    const arrayBuffer = await res.arrayBuffer();
  1000→    const buffer = Buffer.from(arrayBuffer);
  1001→
  1002→    if (buffer.length < 100) {
  1003→      throw new Error(`NIM ${config.name} returned empty/invalid audio`);
  1004→    }
  1005→
  1006→    return buffer;
  1007→  }

  1045→

  1048→

  1056→      method: 'POST',
  1057→      headers,

  1061→    if (!res.ok) {

  1130→  async _openaiEmbedding(text, options) {

  1162→  /**
  1163→   * Rerank search results for quality improvement.
  1164→   * Uses NIM NV-RerankQA model to reorder candidate/job matches.
  1165→   * @param {string} query - The search query
  1166→   * @param {string[]} documents - Array of document texts to rerank
  1167→   * @param {Object} options - { topN }
  1168→   * @returns {Array} Reranked results with scores
  1169→   */
  1170→  async rerank(query, documents, options = {}) {
  1171→    if (!this.nimAvailable) {
  1172→      // No NIM = return original order with default scores
  1173→      console.log('[ai-provider] Reranking skipped (NIM not available), returning original order');
  1174→      return documents.map((doc, i) => ({ index: i, score: 1 - (i * 0.01), text: doc }));
  1175→    }
  1176→
  1177→    try {
  1178→      return await this.execute('reranking', async (providerKey) => {
  1179→        let model;
  1180→        switch (providerKey) {
  1181→          case 'nim_rerank_qa':
  1182→            model = NIM_MODELS.rerank_qa;
  1183→            break;
  1184→          case 'nim_rerank_vl':
  1185→            model = NIM_MODELS.rerank_vl;
  1186→            break;
  1187→          default:
  1188→            throw new Error(`Unknown reranking provider: ${providerKey}`);
  1189→        }
  1190→
  1191→        const topN = options.topN || documents.length;
  1192→
  1193→        // NIM reranking endpoint
  1194→        const res = await fetch(`${this.nimBaseUrl}/ranking`, {
  1195→          method: 'POST',
  1196→          headers: {
  1197→            'Authorization': `Bearer ${this.nimApiKey}`,
  1198→            'Content-Type': 'application/json',
  1199→          },
  1200→          body: JSON.stringify({
  1201→            model,
  1202→            query: { text: query },
  1203→            passages: documents.map(doc => ({ text: doc })),
  1204→            top_n: topN,
  1205→          }),
  1206→        });
  1207→
  1208→        if (!res.ok) {
  1209→          const errText = await res.text();
  1210→          const err = new Error(`NIM Reranking error: ${res.status} ${errText.substring(0, 200)}`);
  1211→          err.status = res.status;
  1212→          throw err;
  1213→        }
  1214→
  1215→        const result = await res.json();
  1216→        return (result.rankings || []).map(r => ({
  1217→          index: r.index,
  1218→          score: r.logit || r.score || 0,
  1219→          text: documents[r.index],
  1220→        }));
  1221→      });
  1222→    } catch (err) {
  1223→      console.error('[ai-provider] Reranking failed, returning original order:', err.message);
  1224→      return documents.map((doc, i) => ({ index: i, score: 1 - (i * 0.01), text: doc }));
  1225→    }
  1226→  }

  1236→  async moderateContent(content, options = {}) {
  1237→    if (!this.nimAvailable) {
  1238→      // No NIM = allow all (no moderation available)

  1248→            break;
  1249→          case 'nim_safety_reasoning':

  1251→            break;
  1252→          default:
  1253→            throw new Error(`Unknown safety provider: ${providerKey}`);

  1255→
  1256→        // NemoGuard uses chat completions format with safety-specific system prompt
  1257→        const response = await this.clients.nim.chat.completions.create({
  1258→          model,
  1259→          messages: [
  1260→            {
  1261→              role: 'system',
  1262→              content: 'You are a content safety classifier. Analyze the following text for safety violations including: harassment, hate speech, violence, sexual content, self-harm, dangerous content, and professional misconduct. Return a JSON object: {"safe": true/false, "categories": {"harassment": false, "hate_speech": false, "violence": false, "sexual_content": false, "self_harm": false, "dangerous": false, "unprofessional": false}, "reason": "explanation if unsafe"}'
  1263→            },
  1264→            { role: 'user', content: content.substring(0, 4000) }
  1265→          ],
  1266→          max_tokens: 512,
  1267→          temperature: 0.1,
  1268→        });
  1269→
  1270→        const text = response.choices[0].message.content;
  1271→        try {
  1272→          const jsonMatch = text.match(/\{[\s\S]*\}/);
  1273→          if (jsonMatch) return JSON.parse(jsonMatch[0]);
  1274→        } catch (e) { /* fall through */ }
  1275→
  1276→        // Default safe if parsing fails
  1277→        return { safe: true, categories: {}, reason: 'Safety check completed (parse fallback)' };
  1278→      });
  1279→    } catch (err) {
  1280→      console.error('[ai-provider] Safety moderation failed:', err.message);
  1281→      return { safe: true, categories: {}, reason: 'Safety check failed (allowing by default)' };
  1282→    }
  1283→  }
