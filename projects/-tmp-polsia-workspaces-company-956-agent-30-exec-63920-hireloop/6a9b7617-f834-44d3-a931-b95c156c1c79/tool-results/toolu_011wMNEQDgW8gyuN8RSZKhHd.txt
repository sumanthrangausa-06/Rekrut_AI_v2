    14→// Helper: Race a promise against a timeout to prevent hanging when AI providers are slow.
    15→// The AI provider chain can take 9×15s = 135s worst case; Render kills requests at ~30s.
    16→// This ensures we hit the scripted fallback BEFORE the request is killed.
    17→function withTimeout(promise, ms, label = 'Operation') {
    18→  return Promise.race([
    19→    promise,
    20→    new Promise((_, reject) => setTimeout(() => {
    21→      reject(new Error(`${label} timed out after ${ms}ms`));
    22→    }, ms))
    23→  ]);
    24→}

   198→// Complete interview and get overall feedback
   199→router.post('/:id/complete', authMiddleware, async (req, res) => {
   200→  try {
   201→    const interview = await pool.query(
   202→      'SELECT * FROM interviews WHERE id = $1 AND user_id = $2',
   203→      [req.params.id, req.user.id]
   204→    );
   205→
   206→    if (interview.rows.length === 0) {
   207→      return res.status(404).json({ error: 'Interview not found' });
   208→    }
   209→
   210→    const interviewData = interview.rows[0];
   211→    const responses = interviewData.responses || [];
   212→    const validResponses = responses.filter(r => r && r.analysis);
   213→
   214→    if (validResponses.length === 0) {
   215→      return res.status(400).json({ error: 'No responses to evaluate' });
   216→    }
   217→
   218→    // Generate overall feedback
   219→    const overallFeedback = await generateOverallFeedback(
   220→      validResponses.map(r => r.analysis),
   221→      { subscriptionId: req.user.stripe_subscription_id }
   222→    );
   223→
   224→    // Calculate duration
   225→    const startTime = new Date(interviewData.created_at);
   226→    const duration = Math.floor((Date.now() - startTime.getTime()) / 1000);
   227→
   228→    // Update interview record
   229→    await pool.query(
   230→      `UPDATE interviews SET
   231→        status = 'completed',
   232→        ai_feedback = $1,
   233→        overall_score = $2,
   234→        duration_seconds = $3,
   235→        completed_at = NOW()
   236→       WHERE id = $4`,
   237→      [JSON.stringify(overallFeedback), overallFeedback.overall_score, duration, req.params.id]
   238→    );
   239→
   240→    // Update OmniScore with interview results
   241→    let omniscoreUpdate = null;
   242→    try {
   243→      omniscoreUpdate = await omniscoreService.addInterviewComponent(
   244→        req.user.id,
   245→        req.params.id,
   246→        overallFeedback.overall_score
   247→      );
   248→
   249→      // Update role-specific score if job title is available
   250→      if (interviewData.job_id) {
   251→        const job = await pool.query('SELECT title FROM jobs WHERE id = $1', [interviewData.job_id]);
   252→        if (job.rows.length > 0) {
   253→          await omniscoreService.updateRoleScore(
   254→            req.user.id,
   255→            job.rows[0].title,
   256→            overallFeedback.overall_score
   257→          );
   258→        }
   259→      }
   260→    } catch (scoreErr) {
   261→      console.error('OmniScore update error:', scoreErr);
   262→      // Don't fail the request if OmniScore update fails
   263→    }
   264→
   265→    res.json({
   266→      success: true,
   267→      overall_feedback: overallFeedback,
   268→      duration_seconds: duration,
   269→      response_count: validResponses.length,
   270→      omniscore: omniscoreUpdate
   271→    });
   272→  } catch (err) {
   273→    console.error('Complete interview error:', err);
   274→    res.status(500).json({ error: 'Failed to complete interview' });
   275→  }

   702→               COUNT(*) OVER () as total

   706→      SELECT
   707→        AVG(CASE WHEN rn <= total/2 THEN score END) as first_half_avg,

   716→      const secondHalf = parseFloat(improvement.rows[0].second_half_avg);
   717→      improvementPercent = ((secondHalf - firstHalf) / firstHalf) * 100;
   718→    }
   719→
   720→    // Calculate day streak
   721→    const streakResult = await pool.query(
   722→      `WITH RECURSIVE date_series AS (
   723→        SELECT CURRENT_DATE::date as check_date, 0 as days_back
   724→        UNION ALL
   725→        SELECT (check_date - INTERVAL '1 day')::date, days_back + 1
   726→        FROM date_series
   727→        WHERE days_back < 30
   728→      )

   732→        SELECT 1 FROM practice_sessions
   733→        WHERE user_id = $1 AND DATE(created_at) = d.check_date
   734→      )
   735→      AND d.check_date <= CURRENT_DATE
   736→      AND NOT EXISTS (
   737→        SELECT 1 FROM date_series d2
   738→        WHERE d2.check_date > d.check_date
   739→        AND d2.check_date <= CURRENT_DATE
   740→        AND NOT EXISTS (
   741→          SELECT 1 FROM practice_sessions
   742→          WHERE user_id = $1 AND DATE(created_at) = d2.check_date
